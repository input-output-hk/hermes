2//!

18/// Runtime Extension needs Initialization during Event Dispatch by the Node.
19///
20/// *MUST* be used with:
21///
22/// ```ignore
23/// #[traitreg::register]
24/// impl RteInitModule for MyRte {
25///   // implementation goes here
26/// }
27/// ```
28pub(crate) trait RteInitEvent {
29    /// Initialize the Runtime extension at Event Dispatch.
30    /// If it errors, the node does not crash, but the App will not
31    /// load.
32    ///
33    /// Note: `self` is not required to be used by the implemented function.
34    /// It is required because of the dynamic initialization logic.
35    ///
36    /// If `self` is used, it is consumed.  This allows the underlying state
37    /// to be stored in a global static, or the like.
38    #[allow(
39        unused_variables,
40        reason = "Preserves the API structure for better documentation."
41    )]
42    fn init(
43        self: Box<Self>,
44        ctx: &HermesRuntimeContext,
45    ) -> RteInitResult {
46        Ok(())
47    }
48
49    /// Returns the priority to be used when calling `init` and `fini`.
50    ///
51    /// IF the priority is the same, the order of initialization is non-deterministic.
52    /// Higher priorities run before lower ones.
53    ///
54    /// Two values are returned
55    ///
56    /// 0 is the default priority, and should be used for most runtime extensions that do
57    /// not care about the order of initialization.
58    fn priority(
59        &self,
60        init: bool,
61    ) -> i32 {
62        RteInitPriority::default().priority(init)
63    }
64
65    /// Finalize the Runtime extension when the event is complete.
66    /// Can Error, but will not cause the Node to terminate.
67    ///
68    /// Note: `self` is not required to be used by the implemented function.
69    /// It is required because of the dynamic initialization logic.
70    ///
71    /// If `self` is used, it is consumed.  This allows the underlying state
72    /// to be stored in a global static, or the like.
73    #[allow(
74        unused_variables,
75        reason = "Preserves the API structure for better documentation."
76    )]
77    fn fini(
78        self: Box<Self>,
79        ctx: &HermesRuntimeContext,
80    ) -> RteInitResult {
81        Ok(())
82    }
83}
84
85#[traitreg::registry(RteInitEvent)]
86static RTE_INIT_EVENT_REGISTRY: () = ();
87
88/// Locks the Initialized `DashSet` to stop race conditions, but only for specific apps.
89/// TODO (SJ): Replace with a true key locked map.
90/// <https://github.com/input-output-hk/catalyst-internal-docs/issues/39#issuecomment-3240936500>
91static IS_RTE_EVENT_LOCK: LazyLock<KeyedLock<String>> = LazyLock::new(KeyedLock::<String>::new);
92
93/// Is the RTE Initialized
94static IS_RTE_EVENT_INITIALIZED: LazyLock<DashSet<String>> = LazyLock::new(DashSet::new);
95
96/// How all Module Init is called.
97///
98/// Note: We use the same trait as the runtime extensions to enforce
99/// uniformity in the call, but this struct IS NOT, and MUST NEVER BE
100/// placed in the registry.
101///
102/// This is how the node interacts with the RTE App Initialization.
103pub(crate) struct RteEvent;
104
105impl RteEvent {
106    /// Create a new instance of the `RteApp` being initialized.
107    #[allow(
108        clippy::unnecessary_box_returns,
109        reason = "Its not unnecessary, the init and fini need it boxed."
110    )]
111    pub fn new() -> Box<Self>
112    where Self: std::marker::Sized {
113        Box::new(Self)
114    }
115
116    /// Create a unique lock ID for an App/Module pair
117    fn lock_id(ctx: &HermesRuntimeContext) -> String {
118        format!(
119            "{}-{}-{}-{}",
120            ctx.app_name(),
121            ctx.module_id(),
122            ctx.event_name(),
123            ctx.exc_counter()
124        )
125        .to_string()
126    }
127}
128
129impl RteInitEvent for RteEvent {
130    fn init(
131        self: Box<Self>,
132        ctx: &HermesRuntimeContext,
133    ) -> RteInitResult {
134        let lock_id = Self::lock_id(ctx);
135
136        // Prevents init or fini running simultaneously for the same app.
137        let _guard = IS_RTE_EVENT_LOCK.lock(lock_id.clone());
138
139        if !IS_RTE_EVENT_INITIALIZED.insert(lock_id) {
140            error!(
141                name=%ctx.app_name(),
142                module=%ctx.module_id(),
143                event=%ctx.event_name(),
144                exc_count=%ctx.exc_counter(),
145                "Multiple attempts to initialize event..  This does not cause problems, but don't do it.");
146            return Ok(()); // Not an error which should stop us running.
147        }
148
149        let errors = run_init_fini!(
150            init = true,
151            registry = RTE_INIT_EVENT_REGISTRY,
152            rte_trait = RteInitEvent,
153            span_label = "Runtime Extension Event Initialization Span",
154            (ctx)
155        );
156
157        errors
158    }
159
160    fn fini(
161        self: Box<Self>,
162        ctx: &HermesRuntimeContext,
163    ) -> RteInitResult {
164        let lock_id = Self::lock_id(ctx);
165
166        // Prevents init or fini running simultaneously for the same app.
167        let _guard = IS_RTE_EVENT_LOCK.lock(lock_id.clone());
168
169        if IS_RTE_EVENT_INITIALIZED.remove(&lock_id).is_none() {
170            error!(
171                name=%ctx.app_name(),
172                module=%ctx.module_id(),
173                event=%ctx.event_name(),
174                exc_count=%ctx.exc_counter(),
175                "Multiple attempts to finalize event (or event never initialized).  This does not cause problems, but don't do it.");
176            return Ok(()); // Not an error which should stop us running.
177        }
178
179        let errors = run_init_fini!(
180            init = true,
181            registry = RTE_INIT_EVENT_REGISTRY,
182            rte_trait = RteInitEvent,
183            span_label = "Runtime Extension Event Finalization Span",
184            (ctx)
185        );
186
187        errors
188    }
189}
