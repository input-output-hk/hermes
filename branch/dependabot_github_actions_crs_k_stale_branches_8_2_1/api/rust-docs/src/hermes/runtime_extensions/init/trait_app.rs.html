1//! Hermes runtime extensions key traits (Application Pre Initialization).
2//!
3//! Runtime extensions must implement these traits if they require code to execute for
4//! resource management or other purposes at various phases of the node and application
5//! life cycle.
6use std::sync::LazyLock;

8use dashmap::DashSet;
9use keyed_lock::sync::KeyedLock;

12use crate::{
13    app::ApplicationName,
14    run_init_fini,
15    runtime_extensions::init::{errors::RteInitResult, priority::RteInitPriority},
16};

18/// Runtime Extension needs Initialization before an Application is loaded by the Node.
19///
20/// *MUST* be used with:
21///
22/// ```ignore
23/// #[traitreg::register]
24/// impl RteInitApp for MyRte {
25///   // implementation goes here
26/// }
27/// ```
28pub(crate) trait RteInitApp {
29    /// Initialize the Runtime extension at App startup (before any modules are
30    /// initialized). If it errors, the node does not crash, but the App will not
31    /// load.
32    ///
33    /// Note: `self` is not required to be used by the implemented function.
34    /// It is required because of the dynamic initialization logic.
35    ///
36    /// If `self` is used, it is consumed.  This allows the underlying state
37    /// to be stored in a global static, or the like.
38    fn init(
39        self: Box<Self>,
40        #[allow(
41            unused_variables,
42            reason = "Preserves the API structure for better documentation."
43        )]
44        name: &ApplicationName,
45    ) -> RteInitResult {
46        Ok(())
47    }
48
49    /// Returns the priority to be used when calling `init` and `fini`.
50    ///
51    /// IF the priority is the same, the order of initialization is non-deterministic.
52    /// Higher priorities run before lower ones.
53    ///
54    /// Two values are returned
55    ///
56    /// 0 is the default priority, and should be used for most runtime extensions that do
57    /// not care about the order of initialization.
58    fn priority(
59        &self,
60        init: bool,
61    ) -> i32 {
62        RteInitPriority::default().priority(init)
63    }
64
65    /// Finalize the Runtime extension when the application is fully stopped.
66    /// Can Error, but will not cause the Node to terminate.
67    ///
68    /// Note: `self` is not required to be used by the implemented function.
69    /// It is required because of the dynamic initialization logic.
70    ///
71    /// If `self` is used, it is consumed.  This allows the underlying state
72    /// to be stored in a global static, or the like.
73    fn fini(
74        self: Box<Self>,
75        #[allow(
76            unused_variables,
77            reason = "Preserves the API structure for better documentation."
78        )]
79        name: &ApplicationName,
80    ) -> RteInitResult {
81        Ok(())
82    }
83}
84
85#[traitreg::registry(RteInitApp)]
86static RTE_INIT_APP_REGISTRY: () = ();
87
88/// Locks the Initialized `DashSet` to stop race conditions, but only for specific apps.
89/// TODO (SJ): Replace with a true key locked map.
90/// <https://github.com/input-output-hk/catalyst-internal-docs/issues/39#issuecomment-3240936500>
91static IS_RTE_APP_LOCK: LazyLock<KeyedLock<String>> = LazyLock::new(KeyedLock::<String>::new);
92
93/// Is the RTE Initialized
94static IS_RTE_APP_INITIALIZED: LazyLock<DashSet<ApplicationName>> = LazyLock::new(DashSet::new);
95
96/// How all Application Init is called.
97///
98/// Note: We use the same trait as the runtime extensions to enforce
99/// uniformity in the call, but this struct IS NOT, and MUST NEVER BE
100/// placed in the registry.
101///
102/// This is how the node interacts with the RTE App Initialization.
103pub(crate) struct RteApp;
104
105impl RteApp {
106    /// Create a new instance of the `RteApp` being initialized.
107    #[allow(
108        clippy::unnecessary_box_returns,
109        reason = "Its not unnecessary, the init and fini need it boxed."
110    )]
111    pub fn new() -> Box<Self>
112    where Self: std::marker::Sized {
113        Box::new(Self)
114    }
115}
116
117impl RteInitApp for RteApp {
118    fn init(
119        self: Box<Self>,
120        name: &ApplicationName,
121    ) -> RteInitResult {
122        // Prevents init or fini running simultaneously for the same app.
123        let _guard = IS_RTE_APP_LOCK.lock(name.to_string());
124
125        if !IS_RTE_APP_INITIALIZED.insert(name.clone()) {
126            error!(name=%name,"Multiple attempts to initialize application..  This does not cause problems, but don't do it.");
127            return Ok(()); // Not an error which should stop us running.
128        }
129
130        let errors = run_init_fini!(
131            init = true,
132            registry = RTE_INIT_APP_REGISTRY,
133            rte_trait = RteInitApp,
134            span_label = "Runtime Extension Node Initialization Span",
135            (name)
136        );
137
138        errors
139    }
140
141    fn fini(
142        self: Box<Self>,
143        name: &ApplicationName,
144    ) -> RteInitResult {
145        // Prevents init or fini running simultaneously for the same app.
146        let _guard = IS_RTE_APP_LOCK.lock(name.to_string());
147
148        if IS_RTE_APP_INITIALIZED.remove(name).is_none() {
149            error!(name=%name,"Multiple attempts to finalize application (or application never initialized).  This does not cause problems, but don't do it.");
150            return Ok(()); // Not an error which should stop us running.
151        }
152
153        let errors = run_init_fini!(
154            init = true,
155            registry = RTE_INIT_APP_REGISTRY,
156            rte_trait = RteInitApp,
157            span_label = "Runtime Extension Node Initialization Span",
158            (name)
159        );
160
161        errors
162    }
163}
