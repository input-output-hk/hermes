1//! Internal state implementation for the `SQLite` module.
2
3use once_cell::sync::Lazy;
4
5use crate::runtime_extensions::{
6    bindings::hermes::sqlite::api::{Sqlite, Statement},
7    resource_manager::ApplicationResourceStorage,
8};
9
10/// The object pointer used specifically with C objects like `sqlite3` or `sqlite3_stmt`.
11pub(super) type ObjectPointer = usize;
12
13/// Map of app name to db resource holder
14pub(super) type DbState = ApplicationResourceStorage<Sqlite, ObjectPointer>;
15
16/// Map of app name to db statement resource holder
17pub(super) type StatementState = ApplicationResourceStorage<Statement, ObjectPointer>;
18
19/// Global state to hold `SQLite` db resources.
20static SQLITE_DB_STATE: Lazy<DbState> = Lazy::new(DbState::new);
21
22/// Global state to hold `SQLite` statement resources.
23static SQLITE_STATEMENT_STATE: Lazy<StatementState> = Lazy::new(StatementState::new);
24
25/// Get the global state of `SQLite` db resources.
26pub(super) fn get_db_state() -> &'static DbState {
27    &SQLITE_DB_STATE
28}
29
30/// Get the global state of `SQLite` statement resources.
31pub(super) fn get_statement_state() -> &'static StatementState {
32    &SQLITE_STATEMENT_STATE
33}
