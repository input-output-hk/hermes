14/// Get public key from the given extended private key.
15///
16/// # Arguments
17///
18/// - `xprivate_key`: An extended private key of type `XPrv`.
19///
20/// # Returns
21///
22/// Returns a tuple of u64 values with length 4 representing the public key.
23pub(crate) fn get_public_key(xprivate_key: &XPrv) -> Bip32Ed25519PublicKey {
24    let xpub = xprivate_key.public().public_key();
25    array_u8_32_to_tuple(&xpub)
26}
27
28/// Sign data with the given extended private key.
29///
30/// # Arguments
31///
32/// - `xprivate_key`: An extended private key of type `XPrv`.
33/// - `data`: The data to sign.
34///
35/// # Returns
36/// Returns a tuple of u64 values with length 8 representing the signature.
37pub(crate) fn sign_data(
38    xprivate_key: &XPrv,
39    data: &Bstr,
40) -> Bip32Ed25519Signature {
41    let sig: Signature<Bstr> = xprivate_key.sign(data);
42    let sig_bytes = sig.to_bytes();
43    array_u8_64_to_tuple(sig_bytes)
44}
45
46/// Check the signature on the given data.
47///
48/// # Arguments
49///
50/// - `xprivate_key`: An extended private key of type `XPrv`.
51/// - `data`: The data to sign.
52/// - `signature`: The signature to check.
53///
54/// # Returns
55/// Returns a boolean value indicating if the signature match the sign data
56/// from `xprivate_key` and data.
57/// True if the signature is valid and match the sign data, false otherwise.
58pub(crate) fn check_signature(
59    xprivate_key: &XPrv,
60    data: &Bstr,
61    signature: Bip32Ed25519Signature,
62) -> bool {
63    let sig_array = b512_u64_tuple_to_u8_array(&signature);
64    // Verify the signature.
65    let signature: Signature<Bstr> = match Signature::from_slice(&sig_array) {
66        Ok(sig) => sig,
67        // Invalid signature
68        Err(_) => return false,
69    };
70    xprivate_key.verify(data, &signature)
71}

73/// Derive a new extended private key from the given extended private key.
74/// - V2 derivation scheme is used as it is mention in [SLIP-0023](https://github.com/satoshilabs/slips/blob/master/slip-0023.md).
75/// - More information about child key derivation can be found in [BIP32-Ed25519](https://input-output-hk.github.io/adrestia/static/Ed25519_BIP.pdf).
76///  
77/// # Arguments
78///
79/// - `xprivate_key`: An extended private key of type `XPrv`.
80/// - `path`: Derivation path. eg. m/0/2'/3 where ' represents hardened derivation.
81///
82/// # Returns
83///
84/// Returns the `XPrv` extended private key as a `Result`.
85/// If the derivation path is successful, it returns `Ok` with the extended private key
86/// (`XPrv`).
87///
88/// # Errors
89///
90/// Returns an `InvalidDerivationalPath` if the derivation path is invalid.
91pub(crate) fn derive_new_private_key(
92    xprivate_key: XPrv,
93    path: &str,
94) -> Result<XPrv, Errno> {
95    let Ok(derivation_path) = path.parse::<DerivationPath>() else {
96        return Err(Errno::InvalidDerivationalPath);
97    };
98    let key = derivation_path
99        .iter()
100        .fold(xprivate_key, |xprv, child_num| {
101            if child_num.is_hardened() {
102                xprv.derive(DerivationScheme::V2, child_num.index() | 0x80_00_00_00)
103            } else {
104                xprv.derive(DerivationScheme::V2, child_num.index())
105            }
106        });
107    Ok(key)
108}
