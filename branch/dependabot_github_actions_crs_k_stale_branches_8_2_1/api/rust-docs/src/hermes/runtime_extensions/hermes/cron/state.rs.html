24/// Cron Internal State
25static CRON_INTERNAL_STATE: Lazy<InternalState> = Lazy::new(|| {
26    let sender = if let Ok(runtime) = Builder::new_current_thread().enable_all().build() {
27        let (sender, receiver) = mpsc::channel(1);
28
29        let _handle = std::thread::spawn(move || {
30            runtime.block_on(async move {
31                let h = tokio::spawn(cron_queue_task(receiver));
32                drop(tokio::join!(h));
33            });
34            std::process::exit(0);
35        });
36        Some(sender)
37    } else {
38        // TODO (@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
39        // Failed to start the queue task
40        None
41    };
42
43    InternalState::new(sender)
44});

46/// Internal State.
47struct InternalState {
48    /// The send events to the crontab queue.
49    cron_queue: CronEventQueue,
50}

52impl InternalState {
53    /// Create a new `InternalState`.
54    fn new(sender: Option<mpsc::Sender<CronJob>>) -> Self {
55        Self {
56            cron_queue: CronEventQueue::new(sender),
57        }
58    }

60    /// Add a new crontab entry.
61    ///
62    /// Allows for management of scheduled cron events queue.
63    ///
64    /// Cron events will be delivered to the `on-cron` event handler.
65    ///
66    /// ## Parameters
67    ///
68    /// - `app_name`:  `HermesAppName`. The name of the application that owns the crontab.
69    /// - `entry`:  `CronTagged`. The crontab entry to add.
70    /// - `retrigger`:  `bool`. If `true`, the event will re-trigger every time the
71    ///   crontab entry matches until cancelled.
72    ///
73    /// ## Returns
74    ///
75    /// - `true`: Crontab added successfully.
76    /// - `false`: Crontab failed to be added.
77    fn add_crontab(
78        &self,
79        app_name: &ApplicationName,
80        entry: CronTagged,
81        retrigger: bool,
82    ) -> bool {
83        let crontab = OnCronEvent {
84            tag: entry,
85            last: !retrigger,
86        };
87        let (cmd_tx, cmd_rx) = oneshot::channel();
88        drop(
89            self.cron_queue
90                .spawn_cron_job(CronJob::Add(app_name.clone(), crontab, cmd_tx)),
91        );
92        cmd_rx.blocking_recv().unwrap_or(false)
93    }
94
95    /// Schedule a single cron event after a fixed delay.
96    ///
97    /// Allows for easy timed wait events to be delivered without
98    /// requiring datetime calculations or formatting cron entries.
99    ///
100    /// ## Parameters
101    ///
102    /// - `app_name`:  `HermesAppName`. The name of the application that owns the crontab.
103    /// - `duration`: `Instant`. How many nanoseconds to delay.  The delay will be AT
104    ///   LEAST this long.
105    /// - `tag`:  `CronEventTag`. A tag which will accompany the triggered event.
106    ///
107    /// ## Returns
108    ///
109    /// - `Ok(true)`: Crontab added successfully.
110    /// - `Ok(false)`: Crontab failed to be added.
111    /// - `Err`: Returns error if the duration is invalid for generating a crontab entry.
112    fn delay_crontab(
113        &self,
114        app_name: &ApplicationName,
115        duration: Instant,
116        tag: CronEventTag,
117    ) -> wasmtime::Result<bool> {
118        let cron_delay = mkdelay_crontab(duration, tag)?;
119        let (cmd_tx, cmd_rx) = oneshot::channel();
120        drop(
121            self.cron_queue
122                .spawn_cron_job(CronJob::Delay(app_name.clone(), cron_delay, cmd_tx)),
123        );
124        if let Ok(resp) = cmd_rx.blocking_recv() {
125            Ok(resp)
126        } else {
127            // TODO (@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
128            Ok(false)
129        }
130    }
131
132    /// List crontabs for an application.
133    ///
134    /// Allows for management of scheduled cron events queue.
135    /// If `tag` is `none` then all crontabs are listed.
136    /// Otherwise, only the crontabs with the specified tag are listed.
137    ///
138    /// ## Parameters
139    ///
140    /// - `tag`: Optional, the tag to limit the list to.  If `none` then all crons listed.
141    ///
142    /// ## Returns
143    ///
144    /// - A list of tuples containing the scheduled crontabs and their tags, along with
145    ///   the current retrigger flag.  `Vec<(CronEventTag, bool)>` The list is sorted from
146    ///   most crontab that will trigger soonest to latest. Crontabs are only listed once,
147    ///   in the case where a crontab may be scheduled may times before a later one.
148    fn ls_crontabs(
149        &self,
150        app_name: &ApplicationName,
151        tag: Option<CronEventTag>,
152    ) -> Vec<(CronTagged, bool)> {
153        let (cmd_tx, cmd_rx) = oneshot::channel();
154        drop(
155            self.cron_queue
156                .spawn_cron_job(CronJob::List(app_name.clone(), tag, cmd_tx)),
157        );
158        // TODO (@@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
159        cmd_rx.blocking_recv().unwrap_or_default()
160    }
161
162    /// Remove the requested crontab.
163    ///
164    /// Allows for management of scheduled cron events.
165    ///
166    /// ## Parameters
167    ///
168    /// - `when`: The crontab entry to add.  Standard crontab format.
169    /// - `tag`: A tag which will accompany the triggered event.
170    ///
171    /// ## Returns
172    ///
173    /// - `true`: The requested crontab was deleted and will not trigger.
174    /// - `false`: The requested crontab does not exist.
175    fn rm_crontab(
176        &self,
177        app_name: &ApplicationName,
178        entry: CronTagged,
179    ) -> bool {
180        let (cmd_tx, cmd_rx) = oneshot::channel();
181        drop(
182            self.cron_queue
183                .spawn_cron_job(CronJob::Remove(app_name.clone(), entry, cmd_tx)),
184        );
185        cmd_rx.blocking_recv().unwrap_or(false)
186    }
187}

189impl Hash for CronTagged {
190    fn hash<H: Hasher>(
191        &self,
192        state: &mut H,
193    ) {
194        self.tag.hash(state);
195        self.when.hash(state);
196    }
197}

199/// Add a crontab to the cron queue.
200pub(crate) fn cron_queue_add(
201    app_name: &ApplicationName,
202    entry: CronTagged,
203    retrigger: bool,
204) -> bool {
205    CRON_INTERNAL_STATE.add_crontab(app_name, entry, retrigger)
206}

208/// List crontabs from the cron queue.
209pub(crate) fn cron_queue_ls(
210    app_name: &ApplicationName,
211    tag: Option<CronEventTag>,
212) -> Vec<(CronTagged, bool)> {
213    CRON_INTERNAL_STATE.ls_crontabs(app_name, tag)

252/// The crontab queue task runs in the background.
253async fn cron_queue_task(mut queue_rx: mpsc::Receiver<CronJob>) {
254    while let Some(cron_job) = queue_rx.recv().await {
255        match cron_job {
256            CronJob::Add(app_name, on_cron_event, response_tx) => {
257                handle_add_cron_job(app_name, on_cron_event, response_tx);
258                // Trigger the cron queue
259                if let Err(_err) = cron_queue_trigger() {
260                    // TODO (@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
261                }
262            },
263            CronJob::List(app_name, tag, response_tx) => {
264                handle_ls_cron_job(&app_name, tag.as_ref(), response_tx);
265            },
266            CronJob::Delay(app_name, cron_job_delay, response_tx) => {
267                handle_delay_cron_job(app_name, cron_job_delay, response_tx);
268                // Trigger the cron queue
269                if let Err(_err) = cron_queue_trigger() {
270                    // TODO (@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
271                }
272            },
273            CronJob::Remove(app_name, cron_tagged, response_tx) => {
274                handle_rm_cron_job(&app_name, &cron_tagged, response_tx);
275                // Trigger the cron queue
276                if let Err(_err) = cron_queue_trigger() {
277                    // TODO (@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
278                }
279            },
280        }
281    }
282}

298/// Handle the `CronJob::Add` command.
299fn handle_add_cron_job(
300    app_name: ApplicationName,
301    on_cron_event: OnCronEvent,
302    response_tx: oneshot::Sender<bool>,
303) {
304    // Check if the event will trigger by getting the next immediate timestamp.
305    let response = if let Some(timestamp) = on_cron_event.tick_from(None) {
306        // add the event to the queue
307        CRON_INTERNAL_STATE
308            .cron_queue
309            .add_event(app_name, timestamp, on_cron_event);
310        true
311    } else {
312        // The event will not trigger any timestamp. This can happen when setting
313        // impossible combinations such as `day = 31` and `month = 2`, since it is not possible
314        // for February to have 31 days.
315
316        // TODO (@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
317        // debug!("Event will not trigger: {on_cron_event:?}");
318        false
319    };
320    if let Err(_err) = response_tx.send(response) {
321        // TODO (@saibatizoku): log error https://github.com/input-output-hk/hermes/issues/15
322        // error!("Error sending response to `handle_add_cron_job`: {_err:?}");
323    }
324}

365    // triggers every minute, three days from now
366    fn crontab_future_dow(
367        tag: &str,
368        days_from_now: i64,
369    ) -> CronTagged {
370        #[allow(clippy::arithmetic_side_effects)] // Ok in tests.
371        let dow = (chrono::Utc::now() + chrono::TimeDelta::try_days(days_from_now).unwrap())
372            .weekday()
373            .number_from_monday();
374        CronTagged {
375            when: format!("* * * * {dow}"),
376            tag: tag.into(),
377        }
378    }
379    // triggers every minute, three days from now
380    fn crontab_example_1() -> CronTagged {
381        crontab_future_dow("Example1", 3)
382    }
383    // triggers every minute, four days from now
384    fn crontab_example_2() -> CronTagged {
385        crontab_future_dow("Example2", 4)
386    }
387    // triggers every minute, two days from now
388    fn crontab_example_3() -> CronTagged {
389        crontab_future_dow("Example3", 2)
390    }
391    // triggers every minute, two days from now
392    fn crontab_other_1() -> CronTagged {
393        crontab_future_dow("Other1", 2)
394    }

429    #[test]
430    // **NOTE**: in order to test the `cron_queue_*` functions,
431    // custom `CronTagged`s are used, by setting the `dow` to be at least 2 days from now
432    // and never more than 6 days ahead. This way, the events won't be dispatched for the
433    // duration of the test.
434    fn test_cron_state_multi_thread_add_and_list_and_delay_and_remove_crontabs_without_triggering()
435    {
436        let app_name = hermes_app_name(APP_NAME);
437        // Initial state for any `HermesAppName` is always empty
438        assert!(cron_queue_ls(&app_name, None).is_empty());
439
440        // inserting returns true
441        assert!(cron_queue_add(
442            &hermes_app_name(APP_NAME),
443            crontab_example_1(),
444            RETRIGGER_YES
445        ));
446
447        // inserting separate thread
448        let h = std::thread::spawn(move || {
449            let app_name = hermes_app_name(APP_NAME);
450            cron_queue_add(&app_name, crontab_example_1(), RETRIGGER_NO)
451        });
452        assert!(h.join().unwrap());
453
454        let queue_ls = cron_queue_ls(&app_name, None);
455        assert!(queue_ls.contains(&(crontab_example_1(), IS_NOT_LAST)));
456        assert!(queue_ls.contains(&(crontab_example_1(), IS_LAST)));
457
458        assert!(cron_queue_add(
459            &app_name,
460            crontab_example_2(),
461            RETRIGGER_YES
462        ));
463
464        let h = std::thread::spawn(move || {
465            let app_name = hermes_app_name(APP_NAME);
466            cron_queue_add(&app_name.clone(), crontab_example_2(), RETRIGGER_YES)
467        });
468        assert!(h.join().unwrap());
469
470        let h = std::thread::spawn(move || {
471            let app_name = hermes_app_name(APP_NAME);
472            cron_queue_ls(&app_name.clone(), None)
473        });
474        let queue_ls = h.join().unwrap();
475        assert!(queue_ls.contains(&(crontab_example_1(), IS_NOT_LAST)));
476        assert!(queue_ls.contains(&(crontab_example_1(), IS_LAST)));
477        assert!(queue_ls.contains(&(crontab_example_2(), IS_NOT_LAST)));
478
479        assert!(cron_queue_add(
480            &app_name,
481            crontab_example_3(),
482            RETRIGGER_YES
483        ));
484        assert!(cron_queue_add(&app_name, crontab_other_1(), RETRIGGER_YES));
485
486        // List
487        let queue_ls = cron_queue_ls(&app_name, None);
488        assert!(queue_ls.contains(&(crontab_example_1(), IS_NOT_LAST)));
489        assert!(queue_ls.contains(&(crontab_example_3(), IS_NOT_LAST)));
490        assert!(queue_ls.contains(&(crontab_other_1(), IS_NOT_LAST)));
491        assert!(queue_ls.contains(&(crontab_example_1(), IS_NOT_LAST)));
492        assert!(queue_ls.contains(&(crontab_example_1(), IS_LAST)));
493        assert!(queue_ls.contains(&(crontab_example_2(), IS_NOT_LAST)));
494
495        // Adding a delayed crontab from another thread
496        let h = std::thread::spawn(move || {
497            let duration = 3_600_000_000_000;
498            let delayed_tag = "Delayed1".to_string();
499            let CronJobDelay {
500                timestamp: _,
501                event,
502            } = mkdelay_crontab(duration, delayed_tag.clone()).unwrap();
503            assert!(
504                cron_queue_delay(&hermes_app_name(APP_NAME), duration, delayed_tag.clone())
505                    .unwrap()
506            );
507
508            CronTagged {
509                when: event.tag.when.clone(),
510                tag: delayed_tag,
511            }
512        });
513        let expected_crontagged = h.join().unwrap();
514
515        assert!(cron_queue_ls(&app_name, None).contains(&(expected_crontagged.clone(), IS_LAST)));
516
517        // Start clearing the queue in current thread
518        assert!(cron_queue_rm(&app_name, expected_crontagged));
519
520        // Remove everything else from another thread
521        let h = std::thread::spawn(move || {
522            let app_name = hermes_app_name(APP_NAME);
523            assert!(cron_queue_rm(&app_name, crontab_example_1()));
524            assert!(cron_queue_rm(&app_name, crontab_example_2()));
525            assert!(cron_queue_rm(&app_name, crontab_example_3()));
526            assert!(cron_queue_rm(&app_name, crontab_other_1()));
527        });
528        h.join().unwrap();
529
530        // Run final test in another thread
531        let h = std::thread::spawn(move || {
532            let app_name = hermes_app_name(APP_NAME);
533            assert!(cron_queue_ls(&app_name, None).is_empty());
534        });
535        h.join().unwrap();
536    }
