23/// Hermes application playground
24///
25/// Loads each wasm component from the arguments as a standalone Hermes application
26/// consisting of a single module with a temporary VFS.
27///
28/// This allows to run Wasm components without properly packaging them.
29///
30/// Returns an exit code that can be inspected for custom values issued by Wasm
31/// components.
32///
33/// If an internal error occurred returns 101.
34#[cfg_attr(debug_assertions, derive(Debug))]
35#[derive(clap::Args)]
36pub struct Playground {
37    /// Wasm components to load as apps in this example.
38    components: Vec<PathBuf>,
39
40    /// Shutdown the playground after the timeout (milliseconds)
41    #[arg(long)]
42    timeout_ms: Option<u64>,
43}
44
45impl Playground {
46    /// Run playground CLI command
47    pub fn exec(self) -> anyhow::Result<Exit> {
48        println!("{} Running a playground...", Emoji::new("‚ú®", ""));
49
50        let temp_dir = TempDir::new()?;
51
52        let apps = collect_apps(&self.components, &temp_dir)?;
53        ensure!(!apps.is_empty(), "At least one app is required to run");
54
55        tracing::info!("{} Bootstrapping IPFS node", console::Emoji::new("üñß", ""),);
56        init_ipfs(&temp_dir)?;
57
58        let exit_lock = reactor::init()?;
59
60        println!(
61            "{} Loading {} application(s)...",
62            Emoji::new("üõ†Ô∏è", ""),
63            apps.len(),
64        );
65        for app in apps {
66            reactor::load_app(app)?;
67        }
68
69        let exit = if let Some(timeout_ms) = self.timeout_ms {
70            exit_lock.wait_timeout(Duration::from_millis(timeout_ms))
71        } else {
72            exit_lock.wait()
73        };
74
75        Ok(exit)
76    }
77}

79/// Initialize the IPFS node
80fn init_ipfs(temp_dir: &TempDir) -> anyhow::Result<()> {
81    let ipfs_dir = create_temp_dir_child(temp_dir, Path::new("ipfs"))?;
82    // disable bootstrapping the IPFS node to default addresses for testing
83    let default_bootstrap = false;
84    ipfs::bootstrap(ipfs_dir.as_path(), default_bootstrap)
85}

87/// Collects `.wasm` files in the current directory or sub-directories of the current
88/// directory. Return a [`String`] module name along with each compiled [`Module`].
89fn collect_modules(components: &[PathBuf]) -> anyhow::Result<Vec<(String, Module)>> {
90    // All wasm components in a directory.
91    let mut modules = Vec::new();
92
93    // Collect component files
94    for file_path in components {
95        let name = file_path
96            .file_name()
97            .and_then(OsStr::to_str)
98            .ok_or_else(|| anyhow!("Provided path is invalid: {}", file_path.display()))?
99            .to_string();
100        let wasm_buf = fs::read(file_path)?;
101        let app_name = ApplicationName::new(&name);
102        let module = Module::from_bytes(&app_name, &wasm_buf)?;
103        modules.push((name, module));
104    }
105
106    Ok(modules)
107}

109/// Create one-module application with temp directory VFS.
110fn create_one_module_app(
111    name: String,
112    vfs_dir_path: &Path,
113    module: Module,
114) -> anyhow::Result<Application> {
115    let vfs_name = [name.as_str(), "_vfs"].concat();
116    let vfs = VfsBootstrapper::new(vfs_dir_path, vfs_name).bootstrap()?;
117    let app = Application::new(name, vfs, vec![module]);
118
119    Ok(app)
120}

122/// Create a temp subdirectory.
123fn create_temp_dir_child(
124    temp_dir: &TempDir,
125    child_path: &Path,
126) -> anyhow::Result<PathBuf> {
127    let child_absolute_path = temp_dir.path().join(child_path);
128    fs::create_dir_all(child_absolute_path.as_path())?;
129    Ok(child_absolute_path)
130}

132/// Collects `.wasm` files in the current directory or sub-directories of the current
133/// directory. Then creates one-module applications out of each of them.
134fn collect_apps(
135    components: &[PathBuf],
136    temp_dir: &TempDir,
137) -> anyhow::Result<Vec<Application>> {
138    let modules = collect_modules(components)?;
139    let mut apps = Vec::with_capacity(modules.len());
140    for (module_name, module) in modules {
141        let vfs_dir_path =
142            create_temp_dir_child(temp_dir, Path::new("vfs").join(&module_name).as_path())?;
143        let app = create_one_module_app(module_name, vfs_dir_path.as_path(), module)?;
144        apps.push(app);
145    }
146    Ok(apps)
147}
