32/// Hermes IPFS Internal Node
33///
34/// This is a wrapper around `HermesIpfsNode` which provides a singleton instance of the
35/// IPFS state.
36///
37/// This is done to ensure the IPFS Node is initialized only once when the
38/// `HermesIpfsNode` is first used. This is done to avoid any issues that may arise if
39/// the IPFS Node is initialized multiple times.
40///
41/// The IPFS Node is initialized in a separate thread and the sender channel is stored in
42/// the `HermesIpfsNode`.
43pub(crate) static HERMES_IPFS: OnceCell<HermesIpfsNode> = OnceCell::new();
44
45/// Bootstrap `HERMES_IPFS` node.
46///
47/// ## Errors
48///
49/// Returns errors if IPFS node fails to start.
50pub fn bootstrap(
51    base_dir: &Path,
52    default_bootstrap: bool,
53) -> anyhow::Result<()> {
54    let ipfs_data_path = base_dir.join("ipfs");
55    if !ipfs_data_path.exists() {
56        tracing::info!("creating IPFS repo directory: {}", ipfs_data_path.display());
57        std::fs::create_dir_all(&ipfs_data_path)?;
58    }
59    let ipfs_node = HermesIpfsNode::init(
60        IpfsBuilder::new()
61            .with_default()
62            .set_default_listener()
63            .set_disk_storage(ipfs_data_path.clone()),
64        default_bootstrap,
65    )?;
66    HERMES_IPFS
67        .set(ipfs_node)
68        .map_err(|_| anyhow::anyhow!("failed to start IPFS node"))?;
69    Ok(())
70}

81    /// Create, initialize, and bootstrap a new `HermesIpfsNode`
82    pub(crate) fn init(
83        builder: IpfsBuilder,
84        default_bootstrap: bool,
85    ) -> anyhow::Result<Self> {
86        let runtime = Builder::new_current_thread().enable_all().build()?;
87        let (sender, receiver) = mpsc::channel(1);
88        let _handle = std::thread::spawn(move || {
89            // Build and start IPFS node
90            let _unused = runtime.block_on(async move {
91                let node = builder.start().await?;
92                if default_bootstrap {
93                    // Add default addresses for bootstrapping
94                    let addresses = node.default_bootstrap().await?;
95                    // Connect to bootstrap nodes.
96                    node.bootstrap().await?;
97                    tracing::debug!(
98                        "Bootstrapped IPFS node with default addresses: {:?}",
99                        addresses
100                    );
101                }
102                let hermes_node: HermesIpfs = node.into();
103                let h = tokio::spawn(ipfs_command_handler(hermes_node, receiver));
104                let (..) = tokio::join!(h);
105                Ok::<(), anyhow::Error>(())
106            });
107            std::process::exit(0);
108        });
109        Ok(Self {
110            sender: Some(sender),
111            apps: AppIpfsState::new(),
112        })
113    }

115    /// Add file
116    ///
117    /// Returns the IPFS path of the added file
118    ///
119    /// ## Parameters
120    /// - `contents`: The content to add
121    ///
122    /// ## Errors
123    /// - `Errno::FileAddError`: Failed to add the content
124    fn file_add(
125        &self,
126        contents: IpfsFile,
127    ) -> Result<hermes_ipfs::IpfsPath, Errno> {
128        let (cmd_tx, cmd_rx) = oneshot::channel();
129        self.sender
130            .as_ref()
131            .ok_or(Errno::FileAddError)?
132            .blocking_send(IpfsCommand::AddFile(
133                AddIpfsFile::Stream((None, contents)),
134                cmd_tx,
135            ))
136            .map_err(|_| Errno::FileAddError)?;
137        cmd_rx.blocking_recv().map_err(|_| Errno::FileAddError)?
138    }

300/// IPFS app state
301struct AppIpfsState {
302    /// List of pinned files per app.
303    pinned_files: DashMap<ApplicationName, HashSet<Cid>>,
304    /// List of DHT values per app.
305    dht_keys: DashMap<ApplicationName, HashSet<DhtKey>>,
306    /// List of subscriptions per app.
307    topic_subscriptions: DashMap<PubsubTopic, HashSet<ApplicationName>>,
308    /// Collection of stream join handles per topic subscription.
309    subscriptions_streams: DashMap<PubsubTopic, JoinHandle<()>>,
310    /// List of evicted peers per app.
311    evicted_peers: DashMap<ApplicationName, HashSet<PeerId>>,
312}
313
314impl AppIpfsState {
315    /// Create new `AppIpfsState`
316    fn new() -> Self {
317        Self {
318            pinned_files: DashMap::default(),
319            dht_keys: DashMap::default(),
320            topic_subscriptions: DashMap::default(),
321            subscriptions_streams: DashMap::default(),
322            evicted_peers: DashMap::default(),
323        }
324    }
325
326    /// Keep track of `ipfs_path` of file pinned by an app.
327    fn pinned_file(
328        &self,
329        app_name: ApplicationName,
330        ipfs_path: &IpfsPath,
331    ) -> Result<(), Errno> {
332        let ipfs_path: BaseIpfsPath = ipfs_path.parse().map_err(|_| Errno::InvalidIpfsPath)?;
333        let cid = ipfs_path.root().cid().ok_or(Errno::InvalidCid)?;
334        self.pinned_files
335            .entry(app_name)
336            .or_default()
337            .value_mut()
338            .insert(*cid);
339        Ok(())
340    }
341
342    /// Un-pin a file with `ipfs_path` pinned by an app.
343    fn unpinned_file(
344        &self,
345        app_name: &ApplicationName,
346        ipfs_path: &IpfsPath,
347    ) -> Result<(), Errno> {
348        let ipfs_path: BaseIpfsPath = ipfs_path.parse().map_err(|_| Errno::InvalidIpfsPath)?;
349        let cid = ipfs_path.root().cid().ok_or(Errno::InvalidCid)?;
350        self.pinned_files
351            .entry(app_name.clone())
352            .or_default()
353            .value_mut()
354            .remove(cid);
355        self.pinned_files.remove_if(app_name, |_, v| v.is_empty());
356        Ok(())
357    }
358
359    /// List of pinned files by an app.
360    pub(crate) fn list_pinned_files(
361        &self,
362        app_name: &ApplicationName,
363    ) -> Vec<String> {
364        self.pinned_files.get(app_name).map_or(vec![], |cids| {
365            cids.iter().map(std::string::ToString::to_string).collect()
366        })
367    }
368
369    /// Keep track of `dht_key` of DHT value added by an app.
370    fn added_dht_key(
371        &self,
372        app_name: ApplicationName,
373        dht_key: DhtKey,
374    ) {
375        self.dht_keys
376            .entry(app_name)
377            .or_default()
378            .value_mut()
379            .insert(dht_key);
380    }
381
382    /// Keep track of `topic` subscription added by an app.
383    fn added_app_topic_subscription(
384        &self,
385        app_name: ApplicationName,
386        topic: PubsubTopic,
387    ) {
388        self.topic_subscriptions
389            .entry(topic)
390            .or_default()
391            .value_mut()
392            .insert(app_name);
393    }
394
395    /// Keep track of `topic` stream handle.
396    fn added_topic_stream(
397        &self,
398        topic: PubsubTopic,
399        handle: JoinHandle<()>,
400    ) {
401        self.subscriptions_streams.entry(topic).insert(handle);
402    }
403
404    /// Check if a topic subscription already exists.
405    fn topic_subscriptions_contains(
406        &self,
407        topic: &PubsubTopic,
408    ) -> bool {
409        self.topic_subscriptions.contains_key(topic)
410    }
411
412    /// Returns a list of apps subscribed to a topic.
413    fn subscribed_apps(
414        &self,
415        topic: &PubsubTopic,
416    ) -> Vec<ApplicationName> {
417        self.topic_subscriptions
418            .get(topic)
419            .map_or(vec![], |apps| apps.value().iter().cloned().collect())
420    }
421
422    /// Add `peer_id` of evicted peer by an app.
423    fn evicted_peer(
424        &self,
425        app_name: ApplicationName,
426        peer_id: PeerId,
427    ) {
428        self.evicted_peers
429            .entry(app_name)
430            .or_default()
431            .value_mut()
432            .insert(peer_id);
433    }
434}
