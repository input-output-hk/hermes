1//! `SQLite` host implementation for WASM runtime.
2
3use super::{core, state::get_db_state};
4use crate::{
5    runtime_context::HermesRuntimeContext,
6    runtime_extensions::bindings::hermes::sqlite::api::{Errno, Host, Sqlite},
7};
8
9impl Host for HermesRuntimeContext {
10    /// Opens a connection to a new or existing `SQLite` database.
11    ///
12    /// ## Parameters
13    ///
14    /// - `readonly`: If set to true, the database is opened in read-only mode. An error
15    ///   is returned if the database doesn't already exist.
16    /// - `memory`: If set to true, the database will be opened as an in-memory database.
17    ///
18    /// ## Returns
19    ///
20    /// If the database is opened (and/or created) successfully, then the `sqlite3` object
21    /// is returned. Otherwise an error code is returned.
22    fn open(
23        &mut self,
24        readonly: bool,
25        memory: bool,
26    ) -> wasmtime::Result<Result<wasmtime::component::Resource<Sqlite>, Errno>> {
27        match core::open(readonly, memory, self.app_name().clone()) {
28            Ok(db_ptr) => {
29                let app_state = get_db_state().get_app_state(self.app_name())?;
30                let db_id = app_state.create_resource(db_ptr as _);
31
32                Ok(Ok(db_id))
33            },
34            Err(err) => Ok(Err(err)),
35        }
36    }
37}
