19impl HostBip32Ed25519 for HermesRuntimeContext {
20    /// Create a new ED25519-BIP32 Crypto resource
21    ///
22    /// **Parameters**
23    ///
24    /// - `mnemonic-phrase` : BIP39 mnemonic.
25    /// - `passphrase` : Optional BIP39 passphrase.
26    fn new(
27        &mut self,
28        mnemonic: MnemonicPhrase,
29        passphrase: Option<Passphrase>,
30    ) -> wasmtime::Result<wasmtime::component::Resource<Bip32Ed25519>> {
31        let passphrase = passphrase.unwrap_or_default();
32        // TODO(bkioshn): https://github.com/input-output-hk/hermes/issues/183
33        let xprv = mnemonic_to_xprv(&mnemonic.join(" "), &passphrase.join(" "))
34            .map_err(|e| wasmtime::Error::msg(e.to_string()))?;
35
36        let app_state = get_state().get_app_state(self.app_name())?;
37        Ok(app_state.create_resource(xprv))
38    }
39
40    /// Get the public key for this private key.
41    fn public_key(
42        &mut self,
43        resource: wasmtime::component::Resource<Bip32Ed25519>,
44    ) -> wasmtime::Result<Bip32Ed25519PublicKey> {
45        let mut app_state = get_state().get_app_state(self.app_name())?;
46        let private_key = app_state.get_object(&resource)?;
47        let public_key = get_public_key(&private_key);
48        Ok(public_key)
49    }
50
51    /// Sign data with the Private key, and return it.
52    ///
53    /// **Parameters**
54    ///
55    /// - `data` : The data to sign.
56    fn sign_data(
57        &mut self,
58        resource: wasmtime::component::Resource<Bip32Ed25519>,
59        data: Bstr,
60    ) -> wasmtime::Result<Bip32Ed25519Signature> {
61        let mut app_state = get_state().get_app_state(self.app_name())?;
62        let private_key = app_state.get_object(&resource)?;
63        let sig = sign_data(&private_key, &data);
64        Ok(sig)
65    }
66
67    /// Check a signature on a set of data.
68    ///
69    /// **Parameters**
70    ///
71    /// - `data` : The data to check.
72    /// - `sig`  : The signature to check.
73    ///
74    /// **Returns**
75    ///
76    /// - `true` : Signature checked OK.
77    /// - `false` : Signature check failed.
78    fn check_sig(
79        &mut self,
80        resource: wasmtime::component::Resource<Bip32Ed25519>,
81        data: Bstr,
82        sig: Bip32Ed25519Signature,
83    ) -> wasmtime::Result<bool> {
84        let mut app_state = get_state().get_app_state(self.app_name())?;
85        let private_key = app_state.get_object(&resource)?;
86        let check_sig = check_signature(&private_key, &data, sig);
87        Ok(check_sig)
88    }
89
90    /// Derive a new private key from the current private key.
91    ///
92    /// **Parameters**
93    ///
94    /// - `path` : Derivation path.
95    ///
96    /// Note: uses BIP32 HD key derivation.
97    fn derive(
98        &mut self,
99        resource: wasmtime::component::Resource<Bip32Ed25519>,
100        path: Path,
101    ) -> wasmtime::Result<wasmtime::component::Resource<Bip32Ed25519>> {
102        let mut app_state = get_state().get_app_state(self.app_name())?;
103
104        let private_key = app_state.get_object(&resource)?;
105        // TODO(bkioshn): https://github.com/input-output-hk/hermes/issues/183
106        let new_private_key = derive_new_private_key(private_key.clone(), &path)
107            .map_err(|_| wasmtime::Error::msg("Error deriving new private key"))?;
108        drop(private_key);
109        Ok(app_state.create_resource(new_private_key))
110    }
111
112    fn drop(
113        &mut self,
114        res: wasmtime::component::Resource<Bip32Ed25519>,
115    ) -> wasmtime::Result<()> {
116        let app_state = get_state().get_app_state(self.app_name())?;
117        app_state.delete_resource(res)?;
118        Ok(())
119    }
120}

122impl Host for HermesRuntimeContext {
123    /// # Generate BIP39 Mnemonic Function
124    ///
125    /// Generate a new BIP39 mnemonic phrase with the given
126    /// size, prefix and language.
127    ///
128    /// ## Parameters
129    ///
130    /// `size` : The size of the mnemonic. Must be a multiple of 3 and in the range of 12
131    /// - 24.
132    /// `prefix` : The prefix for the mnemonic. Must be a list of 1 - 3 words.
133    /// `language` : Optional. The language to use for the mnemonic.
134    ///              If not provided, the default language is used.
135    ///
136    /// ## Returns
137    ///
138    /// - Either a list of mnemonic words.
139    /// - Or an error if the mnemonic could not be generated:
140    ///     - `prefix-too-long` : The prefix is longer than the maximum allowed length,
141    ///       max is 3.
142    ///     - `invalid-mnemonic-length` : The mnemonic length is not a multiple of 3 or
143    ///       not in the range of 12 - 24.
144    ///     - `word-not-found` : A word in the mnemonic is not found in the word list.
145    ///     - `generate-entropy-failed` : Failed to generate entropy.
146    fn generate_mnemonic(
147        &mut self,
148        size: u8,
149        prefix: Vec<String>,
150        language: Option<String>,
151    ) -> wasmtime::Result<Result<Vec<String>, Errno>> {
152        Ok(generate_new_mnemonic(size.into(), prefix, language))
153    }
154}
