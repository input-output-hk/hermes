21/// HTTP scheme.
22const HTTP: &str = "http://";
23/// HTTPS scheme.
24const HTTPS: &str = "https://";

26/// Represents a command that can be sent to the Tokio runtime task.
27pub enum Command {
28    /// Command to send an HTTP request.
29    Send {
30        /// The payload containing the request details.
31        payload: Payload,
32        /// A channel to send the response back to the caller.
33        response_tx: oneshot::Sender<Result<(), ErrorCode>>,
34    },
35}
36
37/// Represents a sender for commands sent to the Tokio runtime task.
38type CommandSender = tokio::sync::mpsc::Sender<Command>;
39/// Represents a receiver for commands sent to the Tokio runtime task.
40type CommandReceiver = tokio::sync::mpsc::Receiver<Command>;
41
42/// Tokio runtime task handle for sending HTTP requests.
43pub struct TokioTaskHandle {
44    /// Command sender for sending commands to the Tokio runtime task.
45    cmd_tx: CommandSender,
46}
47
48impl TokioTaskHandle {
49    /// Sends a command to the Tokio runtime task.
50    pub fn send(
51        &self,
52        payload: Payload,
53    ) -> Result<(), ErrorCode> {
54        let (response_tx, response_rx) = tokio::sync::oneshot::channel();
55        self.cmd_tx
56            .blocking_send(Command::Send {
57                payload,
58                response_tx,
59            })
60            .map_err(|err| {
61                tracing::warn!(%err, "failed to send command via channel");
62                ErrorCode::Internal
63            })?;
64        response_rx.blocking_recv().map_err(|err| {
65            tracing::warn!(%err, "failed to receive command result via channel");
66            ErrorCode::Internal
67        })?
68    }
69}

71/// Initializes the Rustls crypto provider for TLS connections.
72static INIT_RUSTLS_CRYPTO: LazyLock<Result<(), ErrorCode>> = LazyLock::new(|| {
73    rustls::crypto::ring::default_provider()
74        .install_default()
75        .map_err(|_| {
76            tracing::error!("Failed to install default crypto provider for Rustls");
77            ErrorCode::Internal
78        })
79});

81/// Initializes the Rustls TLS connector with the default root certificates.
82fn init_rustls_connector() -> TlsConnector {
83    let mut root_store = RootCertStore::empty();
84    root_store.extend(TLS_SERVER_ROOTS.iter().cloned());
85
86    let config = ClientConfig::builder()
87        .with_root_certificates(root_store)
88        .with_no_client_auth();
89    let config = Arc::new(config);
90    TlsConnector::from(config)
91}

93/// Spawns a Tokio runtime task for handling HTTP requests.
94pub fn spawn() -> TokioTaskHandle {
95    let (cmd_tx, cmd_rx) = tokio::sync::mpsc::channel(1);
96    std::thread::spawn(move || {
97        executor(cmd_rx);
98    });
99
100    TokioTaskHandle { cmd_tx }
101}

103/// Represents a connection to an HTTP or HTTPS server.
104#[allow(clippy::large_enum_variant)]
105enum Connection {
106    /// Represents a plain HTTP connection.
107    Http(TcpStream),
108    /// Represents a secure HTTP connection using TLS.
109    Https(client::TlsStream<TcpStream>),
110}
111
112impl Connection {
113    /// Creates a new connection to the specified HTTP or HTTPS server.
114    async fn new<S>(
115        addr: S,
116        port: u16,
117    ) -> Result<Self, ErrorCode>
118    where
119        S: AsRef<str> + Into<String> + core::fmt::Display,
120    {
121        if let Some(sliced_addr) = addr.as_ref().to_ascii_lowercase().strip_prefix(HTTP) {
122            let stream = TcpStream::connect((sliced_addr, port))
123                .await
124                .map_err(|err| {
125                    tracing::debug!(%err, %sliced_addr, "Failed to connect to HTTP server");
126                    ErrorCode::HttpConnectionFailed
127                })?;
128            tracing::trace!(%addr, port, "connected over HTTP");
129            return Ok(Connection::Http(stream));
130        } else if let Some(sliced_addr) = addr.as_ref().to_ascii_lowercase().strip_prefix(HTTPS) {
131            (*INIT_RUSTLS_CRYPTO)?;
132
133            let tcp_stream = TcpStream::connect((sliced_addr, port))
134                .await
135                .map_err(|err| {
136                    tracing::debug!(%err, %sliced_addr, "tcp stream to HTTPs server");
137                    ErrorCode::HttpsConnectionFailed
138                })?;
139
140            let server_name = ServerName::try_from(sliced_addr.to_string()).map_err(|err| {
141                tracing::debug!(%err, %sliced_addr, "invalid server name when connecting to HTTPs server");
142                ErrorCode::HttpsConnectionFailed
143            })?;
144
145            let stream = STATE
146                .tls_connector
147                .get_or_init(init_rustls_connector)
148                .connect(server_name, tcp_stream)
149                .await
150                .map_err(|err| {
151                    tracing::debug!(%err, %sliced_addr, "TLS handshake failed");
152                    ErrorCode::HttpsConnectionFailed
153                })?;
154
155            tracing::trace!(%addr, port, "connected over HTTPs");
156            return Ok(Connection::Https(stream));
157        }
158        tracing::debug!(%addr, "missing scheme");
159        Err(ErrorCode::MissingScheme)
160    }
161
162    /// Sends the HTTP request body and returns the response.
163    // TODO[RC]: Timeout or more complex task management needed here
164    async fn send(
165        &mut self,
166        body: &[u8],
167    ) -> Result<Vec<u8>, ErrorCode> {
168        let mut response = Vec::new();
169        match self {
170            Connection::Http(ref mut tcp_stream) => {
171                tcp_stream.write_all(body).await.map_err(|err| {
172                    error!("failed to send HTTP request: {err}");
173                    ErrorCode::HttpSendFailed
174                })?;
175                read_to_end_ignoring_unexpected_eof(tcp_stream, &mut response)
176                    .await
177                    .map_err(|err| {
178                        tracing::debug!(%err, "failed to read from HTTP server");
179                        ErrorCode::HttpSendFailed
180                    })?;
181                tracing::debug!(length_bytes = response.len(), "got response");
182            },
183            Connection::Https(tls_stream) => {
184                tls_stream.write_all(body).await.map_err(|err| {
185                    tracing::error!(%err, "failed to connect to HTTPs server");
186                    ErrorCode::HttpsSendFailed
187                })?;
188                read_to_end_ignoring_unexpected_eof(tls_stream, &mut response)
189                    .await
190                    .map_err(|err| {
191                        tracing::debug!(%err, "failed to read from HTTPs server");
192                        ErrorCode::HttpsSendFailed
193                    })?;
194                tracing::debug!(length_bytes = response.len(), "got response");
195            },
196        }
197        Ok(response)
198    }
199}

201/// Reads all bytes from the reader until EOF, ignoring `UnexpectedEof` errors.
202async fn read_to_end_ignoring_unexpected_eof<R>(
203    reader: &mut R,
204    buf: &mut Vec<u8>,
205) -> std::io::Result<usize>
206where
207    R: AsyncRead + Unpin,
208{
209    // TODO[RC]: This won't work for payloads that do not include "Connection: close", we need
210    // a more sophisticated processing.
211    match reader.read_to_end(buf).await {
212        Ok(0) => Ok(0),
213        Ok(n) => Ok(n),
214        Err(e) if e.kind() == std::io::ErrorKind::UnexpectedEof => {
215            tracing::debug!("HTTPs connection closed unexpectedly, but no big deal");
216            Ok(buf.len())
217        },
218        Err(e) => Err(e),
219    }
220}

222/// Sends an HTTP request and puts the response event on the queue.
223async fn send_http_request(payload: Payload) -> Result<(), ErrorCode> {
224    // TODO[RC]: Make sure there are no stray threads left running
225    let mut conn = Connection::new(payload.host_uri, payload.port).await?;
226    let response = conn.send(&payload.body).await?;
227
228    let on_http_response_event = super::event::OnHttpResponseEvent {
229        request_id: payload.request_id,
230        response,
231    };
232
233    crate::event::queue::send(HermesEvent::new(
234        on_http_response_event,
235        TargetApp::All,
236        TargetModule::All,
237    ))
238    .map_err(|err| {
239        tracing::error!(%err, "queue failure");
240        ErrorCode::Internal
241    })?;
242    Ok(())
243}

245/// Handles a command sent to the Tokio runtime task.
246async fn handle_command(cmd: Command) {
247    match cmd {
248        Command::Send {
249            payload,
250            response_tx,
251        } => {
252            let sending_result = send_http_request(payload).await;
253            let _ = response_tx.send(sending_result);
254        },
255    }
256}

258/// Tokio runtime task for handling HTTP requests.
259fn executor(mut cmd_rx: CommandReceiver) {
260    let res = tokio::runtime::Builder::new_current_thread()
261        .enable_io()
262        .build();
263
264    let rt = match res {
265        Ok(rt) => rt,
266        Err(err) => {
267            error!(error = ?err, "failed to start Http Request Runtime Extension background thread");
268            return;
269        },
270    };
271
272    trace!("Created Tokio runtime for Http Request Runtime Extension");
273
274    rt.block_on(async move {
275        while let Some(cmd) = cmd_rx.recv().await {
276            tokio::spawn(handle_command(cmd));
277        }
278    });
