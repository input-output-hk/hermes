1//! Regex Specificity Scoring and Sorting
2//!
3//! This module provides functionality to heuristically score and sort
4//! regular expressions by their "specificity."
5//!
6//! ### Specificity Heuristics
7//! - **More literal characters → more specific**
8//! - **Anchors (`^`, `$`) → more specific**
9//! - **Exact quantifiers (`{n}`) → more specific**
10//! - **Wildcards (`.`, `.*`) and open quantifiers (`+`, `*`, `?`) → less specific**
11//! - **Alternation (`|`) → less specific**
12//!
13//! ⚠️ Note: This is **heuristic-based**. Regex semantics can be arbitrarily
14//! complex, so this does not guarantee mathematically correct ordering,
15//! but is usually good enough for routing/matching use cases.
16
17use std::sync::LazyLock;
18
19use catalyst_types::conversion::from_saturating;
20use regex::Regex;
21
22use crate::runtime_extensions::utils::mul_add::SaturatingMulAdd;

24// Scoring weights (tweakable constants).
25/// Weight of an alphanumeric literal
26const LITERAL_WEIGHT: i32 = 1;
27/// Weight of an Anchor (`^` or `$`) in the Regex
28const ANCHOR_WEIGHT: i32 = 5;
29/// Weight of a single `.` in the Regex
30const DOT_WEIGHT: i32 = -2;
31/// Weight of a `.*` in the Regex
32const DOTSTAR_WEIGHT: i32 = -10;
33/// Weight of a `+` in the Regex
34const PLUS_WEIGHT: i32 = -11;
35/// Weight of a `*` in the Regex
36const STAR_WEIGHT: i32 = -3;
37/// Weight of a `?` in the Regex
38const QMARK_WEIGHT: i32 = -2;
39/// Weight of a quantifier (`{n}`) in the Regex
40const EXACT_QUANT_WEIGHT: i32 = 4;
41
42/// Regex used to match and count exact number of quantifiers in a regex.
43static EXACT_QUANT: LazyLock<Regex> = LazyLock::new(|| {
44    #[allow(clippy::unwrap_used)]
45    Regex::new(r"\{\d+\}").unwrap()
46});
47
48/// Compute a "specificity score" for a regex string.
49/// Higher scores mean the regex is considered more specific.
50/// `input` is anything that can be converted to a string reference.
51#[allow(dead_code)]
52pub(crate) fn regex_specificity_score<T: AsRef<str>>(input: T) -> i32 {
53    let re = input.as_ref();
54
55    let mut score: i32 = 0;
56
57    // split on alternations, if present, and weight them distinctly, and use the total of
58    // weights divided by the number of alternations.
59    // Theory being the specificity is the average of all the alternatives specificity.
60    //let alts = re.bytes().filter(|c| *c == b'|').count();
61    let alts: Vec<&str> = re.split('|').collect();
62    let alts_len = alts.len();
63    if alts_len > 1 {
64        for x in alts {
65            score.mul_add(1, regex_specificity_score(x));
66        }
67        #[allow(clippy::arithmetic_side_effects)]
68        return score.saturating_div(from_saturating(alts_len));
69    }
70
71    // Count literal (alphanumeric) characters
72    score.mul_add(
73        re.chars().filter(|c| c.is_alphanumeric()).count(),
74        LITERAL_WEIGHT,
75    );
76
77    // Anchors
78    if re.contains('^') {
79        score.mul_add(1, ANCHOR_WEIGHT);
80    }
81    if re.contains('$') {
82        score.mul_add(1, ANCHOR_WEIGHT);
83    }
84
85    // Wildcards
86    let dot_stars = re.matches(".*").count();
87    if dot_stars > 0 {
88        score.mul_add(dot_stars, DOTSTAR_WEIGHT);
89    } else {
90        score.mul_add(re.matches('.').count(), DOT_WEIGHT);
91    }
92
93    // Quantifiers
94    score.mul_add(re.matches('+').count(), PLUS_WEIGHT);
95    score.mul_add(re.matches('*').count(), STAR_WEIGHT);
96    score.mul_add(re.matches('?').count(), QMARK_WEIGHT);
97
98    // Exact quantifiers `{n}`
99    score.mul_add(EXACT_QUANT.find_iter(re).count(), EXACT_QUANT_WEIGHT);
100
101    score
102}
103
104/// Sort a list of regex strings from most specific to least specific.
105#[allow(dead_code)]
106pub(crate) fn sort_regexes_by_specificity<T: AsRef<str>>(regexes: Vec<T>) -> Vec<T> {
107    let mut scored: Vec<_> = regexes
108        .into_iter()
109        .map(|re| (regex_specificity_score(&re), re))
110        .collect();
111
112    // Sort descending by score (most specific first)
113    scored.sort_by(|a, b| b.0.cmp(&a.0));
114
115    scored.into_iter().map(|(_score, re)| re).collect()
116}

118#[cfg(test)]
119mod tests {
120    use super::*;
121
122    #[test]
123    fn test_literal_specificity() {
124        let s1 = regex_specificity_score("hello");
125        let s2 = regex_specificity_score(".*");
126        assert!(s1 > s2, "Literal should be more specific than wildcard");
127    }
128
129    #[test]
130    fn test_anchor_bonus() {
131        let anchored = regex_specificity_score("^abc$");
132        let unanchored = regex_specificity_score("abc");
133        assert!(anchored > unanchored, "Anchored regex should score higher");
134    }
135
136    #[test]
137    fn test_quantifiers() {
138        let exact = regex_specificity_score(r"a{5}");
139        let open = regex_specificity_score(r"a+");
140        assert!(exact > open, "Exact quantifier should be more specific");
141    }
142
143    #[test]
144    fn test_alternation_penalty() {
145        let alt = regex_specificity_score("a|b|c");
146        let literal = regex_specificity_score("abc");
147        assert!(
148            literal > alt,
149            "Literal sequence should be more specific than alternation"
150        );
151    }
152
153    #[test]
154    fn test_sorting_order() {
155        let regexes = vec![r".*", r"^hello$", r"abc\d+", r"^foo.*bar$", r"a|b|c"];
156        let expected = vec![r"^hello$", r"^foo.*bar$", r"a|b|c", r"abc\d+", r".*"];
157
158        let sorted = sort_regexes_by_specificity(regexes);
159
160        assert_eq!(sorted, expected);
161    }
162}
