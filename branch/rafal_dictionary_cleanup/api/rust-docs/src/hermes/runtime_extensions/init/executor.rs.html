3use traitreg::RegisteredImplWrapper;
4
5/// Generic data for sorting initialization/finalization of runtime extensions.
6pub(crate) struct InstanceData<'a, T: ?Sized + 'static> {
7    /// RTE Registration Data
8    pub register: &'a RegisteredImplWrapper<Box<T>>,
9    /// RTE Instance
10    pub instance: Box<T>,
11}
12
13/// Helper to make running `init` and `fini` for a Trait registry easy and consistent.
14#[macro_export]
15macro_rules! run_init_fini {
16    (
17        init = $init:expr,
18        registry = $registry:ident,
19        rte_trait = $rte_trait:ident,
20        span_label = $span_label:expr,
21        ( $( $arg:expr ),* $(,)? )
22    ) => {{
23        use $crate::runtime_extensions::init::errors::RuntimeExtensionErrors;
24
25        /// Data we use to help sort initialization and finalization of runtime extensions.
26        type InstanceData<'a> = $crate::runtime_extensions::init::executor::InstanceData<'a, dyn $rte_trait>;
27
28        let errors = RuntimeExtensionErrors::new();
29
30        // TODO: (SJ) Creating the list of sorted (and grouped) instances could in theory
31        // be done only once, and saved inside a OnceLock, however, the instances are not
32        // Send + Sync, so this currently isn't possible as they can't be moved between threads.
33        // Would require an update to the traitreg crate most likely.
34        // Suggestion would be to make creating any instance wrapped in an Arc<Box<>> perhaps.
35        // Needs more investigation.
36        // For now, this is enough, if the number of registered runtime extensions grows,
37        // or the work they need to do gets more complex, then this can be revisited.
38        let mut instances: Vec<InstanceData> = Vec::new();
39
40        for registered in $registry.iter() {
41            // First we collect all the instances.
42            if let Some(instance) = registered.instanciate() {
43                instances.push(InstanceData {
44                    register: registered,
45                    instance,
46                });
47            } else {
48                error!(
49                    name = registered.name(),
50                    path = registered.path(),
51                    file = registered.file(),
52                    trait_name = registered.trait_name(),
53                    module_path = registered.module_path(),
54                    "Missing Constructor in the registered runtime extension."
55                );
56            }
57        }
58
59        instances.sort_by_key(|rte| std::cmp::Reverse(rte.instance.priority($init)));
60
61        for rte in instances {
62            // TODO: (SJ) This could be executed in parallel.
63            // For example, instead of a sorted list, we used a sorted list of groups with the
64            // same priorities.  The individual groups can be executed sequentially, with
65            // the runtime extensions at the same priority being executed in parallel.
66            // this would likely improve performance, especially for the event dispatch initializer/finalizer.
67            // However, it isn't currently possible because the instances from the registry are not Send + Sync
68            // so they can't be moved between threads.
69            // This would also need the grouped and sorted list of runtime extensions to be held in a once
70            // lock to eliminate the requirement to constantly re-calculate them.
71            // Would most likely require an update/fork of the `traitreg` crate to make instances thread safe.
72            // Currently this is not a big problem, but could become bigger if the runtime extension event
73            // initialization became bigger or more complex.  Runtime, App and Module initialization are not time
74            // sensitive, so would not be hurt, but would likely not help performance to be parallel executed to
75            // any noticeable degree.
76            // Runtime Initialization dispatch is designed to be thread safe, so they can be executed in parallel
77            // Individual runtime extensions can rely on the initializers not calling them more than once or
78            // in a way which would not be thread safe if executed concurrently.
79            span!(
80                Level::DEBUG,
81                $span_label,
82                priority = rte.instance.priority($init),
83                name = rte.register.name(),
84                path = rte.register.path(),
85                file = rte.register.file(),
86                trait_name = rte.register.trait_name(),
87                module_path = rte.register.module_path(),
88            )
89            .in_scope(|| {
90                if $init {
91                errors.maybe(rte.instance.init( $( $arg ),* ));
92                } else {
93                errors.maybe(rte.instance.fini( $( $arg ),* ));
94
95                }
96            });
97        }
98
99        if errors.is_empty() {
100            Ok(())
101        } else {
102            Err(errors)
103        }
104    }};
105}
