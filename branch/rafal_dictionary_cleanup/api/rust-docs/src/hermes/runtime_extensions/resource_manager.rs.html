13/// `ResourceStorage` struct.
14/// - `WitType` represents the type from the wit file definitions and which will appear in
15///   the `wasmtime::component::Resource<WitType>` object.
16/// - `RustType` actually the type which is bound to the `WitType` and holds all the data
17///   needed for the `WitType`.
18pub(crate) struct ResourceStorage<WitType, RustType> {
19    /// Map of id to resource object.
20    state: DashMap<u32, RustType>,
21    /// Next available address id of the resource.
22    available_address: AtomicU32,
23    /// `WitType` type phantom.
24    _phantom: std::marker::PhantomData<WitType>,
25}
26
27impl<WitType, RustType> ResourceStorage<WitType, RustType>
28where WitType: 'static
29{
30    /// Creates new `ResourceStorage` instance.
31    pub(crate) fn new() -> Self {
32        Self {
33            state: DashMap::new(),
34            available_address: AtomicU32::default(),
35            _phantom: std::marker::PhantomData,
36        }
37    }
38
39    /// Creates a new owned resource from the given object.
40    /// Stores a resources link to the original object in the resource manager.
41    pub(crate) fn create_resource(
42        &self,
43        object: RustType,
44    ) -> wasmtime::component::Resource<WitType> {
45        let available_address = self.available_address.load(Ordering::Acquire);
46        self.state.insert(available_address, object);
47
48        // Increment the value of the available address to 1,
49        // so that it can be used for the next resource.
50        // Under the assumption that `ResourceStorage` will not handle too many resources at once,
51        // and will not hold resources for too long, saturating increment is safe.
52        self.available_address
53            .store(available_address.saturating_add(1), Ordering::Release);
54
55        wasmtime::component::Resource::new_own(available_address)
56    }
57
58    /// Creates a new owned resource from the given object.
59    /// Stores a resources link to the original object in the resource manager.
60    ///
61    /// NOTE: `&mut self` is used to enable the `rustc` borrow checker and prevent
62    /// potential deadlocking of the `DashMap`.
63    /// As `ResourceStorage` not supposed to be used in any other cases rather than as a
64    /// field of `ApplicationResourceStorage` (so no any `static` variable of this
65    /// type will be created), it is fine to add `&mut self` for this method.
66    pub(crate) fn get_object<'a>(
67        &'a mut self,
68        resource: &wasmtime::component::Resource<WitType>,
69    ) -> wasmtime::Result<impl DerefMut<Target = RustType> + 'a> {
70        self.state
71            .get_mut(&resource.rep())
72            .ok_or(Self::resource_not_found_err())
73    }
74
75    /// Removes the resource from the resource manager.
76    /// Similar to the `drop` function, resource is releasing and consumed by this
77    /// function, thats why it is passed by value.
78    #[allow(clippy::needless_pass_by_value)]
79    pub(crate) fn delete_resource(
80        &self,
81        resource: wasmtime::component::Resource<WitType>,
82    ) -> anyhow::Result<RustType> {
83        self.delete_resource_rep(resource.rep())
84    }
85
86    /// Removes the resource from the resource manager by its representation.
87    /// The resource is properly removed from internal storage.
88    /// Returns an error if the resource was not found.
89    pub(crate) fn delete_resource_rep(
90        &self,
91        rep: u32,
92    ) -> anyhow::Result<RustType> {
93        self.state
94            .remove(&rep)
95            .map(|(_, v)| v)
96            .ok_or(Self::resource_not_found_err())
97    }
98
99    /// Resource not found error message.
100    fn resource_not_found_err() -> wasmtime::Error {
101        let msg = format!(
102            "Resource <{}, {}> not found, need to add resource first by calling `add_resource`",
103            type_name::<WitType>(),
104            type_name::<RustType>()
105        );
106        wasmtime::Error::msg(msg)
107    }
108}

110/// `ApplicationResourceStorage` struct.
111/// - `WitType` represents the type from the wit file definitions and which will appear in
112///   the `wasmtime::component::Resource<WitType>` object.
113/// - `RustType` actually the type which is bound to the `WitType` and holds all the data
114///   needed for the `WitType`.
115pub(crate) struct ApplicationResourceStorage<WitType, RustType> {
116    /// Map of app name to resources.
117    state: DashMap<ApplicationName, ResourceStorage<WitType, RustType>>,
118}
119
120impl<WitType, RustType> ApplicationResourceStorage<WitType, RustType>
121where WitType: 'static
122{
123    /// Creates new `ApplicationResourceStorage` instance.
124    pub(crate) fn new() -> Self {
125        Self {
126            state: DashMap::new(),
127        }
128    }
129
130    /// Adds new application to the resource manager.
131    /// If the application state already exists, do nothing.
132    pub(crate) fn add_app(
133        &self,
134        app_name: ApplicationName,
135    ) {
136        if !self.state.contains_key(&app_name) {
137            self.state.insert(app_name, ResourceStorage::new());
138        }
139    }
140
141    /// Get application state from the resource manager.
142    /// To increase performance and reduce locking time, it's better to call
143    /// `drop(app_state)` immediately when the `app_state` is not needed anymore and don't
144    /// wait until it will be released by the compiler.
145    ///
146    /// **Locking behavior:** May deadlock if called when holding any sort of reference
147    /// into the map.
148    pub(crate) fn get_app_state<'a>(
149        &'a self,
150        app_name: &ApplicationName,
151    ) -> anyhow::Result<impl DerefMut<Target = ResourceStorage<WitType, RustType>> + 'a> {
152        self.state
153            .get_mut(app_name)
154            .ok_or_else(|| anyhow::anyhow!(Self::app_not_found_err()))
155    }
156
157    /// Removes application and all associated resources from the resource manager.
158    #[allow(dead_code)]
159    pub(crate) fn remove_app(
160        &self,
161        app_name: &ApplicationName,
162    ) {
163        self.state.remove(app_name);
164    }
165
166    /// Application not found error message.
167    fn app_not_found_err() -> wasmtime::Error {
168        let msg = format!(
169        "Application not found for resource <{}, {}>, need to add application first by calling `add_app`",
170        type_name::<WitType>(),
171        type_name::<RustType>()
172    );
173        wasmtime::Error::msg(msg)
174    }
175}

177#[cfg(all(test, debug_assertions))]
178mod tests {
179    use super::*;
180
181    struct WitType;
182
183    #[test]
184    fn test_resource_storage() {
185        let mut resource_manager = ResourceStorage::<WitType, u32>::new();
186
187        let object = 100;
188        let resource = resource_manager.create_resource(object);
189        let copied_resource = wasmtime::component::Resource::new_borrow(resource.rep());
190
191        assert_eq!(*resource_manager.get_object(&resource).unwrap(), object);
192        assert!(resource_manager.delete_resource(resource).is_ok());
193        assert!(resource_manager.get_object(&copied_resource).is_err());
194        assert!(resource_manager.delete_resource(copied_resource).is_err());
195    }
196
197    #[test]
198    fn test_app_resource_storage() {
199        let resource_manager = ApplicationResourceStorage::<WitType, u32>::new();
200        let app_name_1 = ApplicationName("app_1".to_string());
201
202        {
203            assert!(resource_manager.get_app_state(&app_name_1).is_err());
204            resource_manager.add_app(app_name_1.clone());
205            assert!(resource_manager.get_app_state(&app_name_1).is_ok());
206            resource_manager.remove_app(&app_name_1);
207            assert!(resource_manager.get_app_state(&app_name_1).is_err());
208        }
209
210        {
211            // Check preserving app state when instantiating the same app twice
212            let object = 100;
213            resource_manager.add_app(app_name_1.clone());
214            let app_state = resource_manager.get_app_state(&app_name_1).unwrap();
215            let res = app_state.create_resource(object);
216
217            drop(app_state);
218            resource_manager.add_app(app_name_1.clone());
219            let mut app_state = resource_manager.get_app_state(&app_name_1).unwrap();
220            assert!(app_state.get_object(&res).is_ok());
221        }
222    }
