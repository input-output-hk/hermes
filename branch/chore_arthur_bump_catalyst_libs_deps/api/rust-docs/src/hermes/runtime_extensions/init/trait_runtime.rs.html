20/// Runtime Extension needs Initialization at Node Startup
21///
22/// *MUST* be used with:
23///
24/// ```ignore
25/// #[traitreg::register(new)]
26/// impl RteInitRuntime for MyRte {
27///   // implementation goes here
28/// }
29/// ```
30pub(crate) trait RteInitRuntime {
31    /// Initialize the Runtime extension at node startup.
32    /// This SHOULD be infallible, but if it does fail, then the node must
33    /// present the errors reported to the user in a nice and actionable way BEFORE
34    /// it exits.
35    ///
36    /// Note: `self` is not required to be used by the implemented function.
37    /// It is required because of the dynamic initialization logic.
38    ///
39    /// If `self` is used, it is consumed.  This allows the underlying state
40    /// to be stored in a global static, or the like.
41    fn init(self: Box<Self>) -> RteInitResult {
42        Ok(())
43    }
44
45    /// Returns the priority to be used when calling `init` and `fini`.
46    ///
47    /// IF the priority is the same, the order of initialization is non-deterministic.
48    /// Higher priorities run before lower ones.
49    ///
50    /// Two values are returned
51    ///
52    /// 0 is the default priority, and should be used for most runtime extensions that do
53    /// not care about the order of initialization.
54    fn priority(
55        &self,
56        init: bool,
57    ) -> i32 {
58        RteInitPriority::default().priority(init)
59    }
60
61    /// Finalize the Runtime extension at node startup.
62    /// Is infallible, because runtime extensions should not fail to initialize.
63    ///
64    /// Note: `self` is not required to be used by the implemented function.
65    /// It is required because of the dynamic initialization logic.
66    ///
67    /// If `self` is used, it is consumed.  This allows the underlying state
68    /// to be stored in a global static, or the like.
69    fn fini(self: Box<Self>) -> RteInitResult {
70        Ok(())
71    }
72}
73
74#[traitreg::registry(RteInitRuntime)]
75static RTE_INIT_RUNTIME_REGISTRY: () = ();
76
77/// Is the RTE Initialized
78static IS_RTE_RUNTIME_INITIALIZED: LazyLock<Mutex<bool>> = LazyLock::new(|| Mutex::new(false));
79/// Is the RTE finalized
80static IS_RTE_RUNTIME_FINALIZED: LazyLock<Mutex<bool>> = LazyLock::new(|| Mutex::new(false));
81
82/// How all RTE Runtime Init is called.
83///
84/// Note: We use the same trait as the runtime extensions to enforce
85/// uniformity in the call, but this struct IS NOT, and MUST NEVER BE
86/// placed in the registry.
87///
88/// This is how the node interacts with the RTE Runtime Initialization.
89pub(crate) struct RteRuntime;
90
91impl RteRuntime {
92    /// Create a new instance of the `RteRuntime` being initialized.
93    #[allow(
94        clippy::unnecessary_box_returns,
95        reason = "Its not unnecessary, the init and fini need it boxed."
96    )]
97    pub fn new() -> Box<Self>
98    where Self: std::marker::Sized {
99        Box::new(Self)
100    }
101}
102
103impl RteInitRuntime for RteRuntime {
104    fn init(self: Box<Self>) -> RteInitResult {
105        match IS_RTE_RUNTIME_INITIALIZED.lock() {
106            Ok(mut initialized) => {
107                if *initialized {
108                    error!("Multiple calls to RTE Node `init()`.  This does not cause problems, but don't do it.");
109                    return Ok(()); // Not an error which should stop us running.
110                }
111
112                let errors = run_init_fini!(
113                    init = true,
114                    registry = RTE_INIT_RUNTIME_REGISTRY,
115                    rte_trait = RteInitRuntime,
116                    span_label = "Runtime Extension Node Initialization Span",
117                    ()
118                );
119
120                *initialized = true;
121
122                errors
123            },
124            Err(e) => {
125                let msg = "Poisoned `IS_RTE_RUNTIME_INITIALIZED` on RTE Runtime Initialization. Should never happen.";
126                error!(
127                    error = ?e,
128                    "Failed to acquire lock on RTE Runtime Initialization. Should never happen."
129                );
130
131                let errors = RuntimeExtensionErrors::new();
132                add_rte_error!(errors, RteMetadata::none(), ImpossibleError {
133                    description: msg.to_string(),
134                });
135                Err(errors)
136            },
137        }
138    }
139
140    fn fini(self: Box<Self>) -> RteInitResult {
141        // Needs to be fully initialized or we won't run finalize.
142        match IS_RTE_RUNTIME_INITIALIZED.lock() {
143            Ok(initialized) => {
144                if !*initialized {
145                    error!("RTE Node `fini()` called but runtimes are not initialized.  This does not cause problems by itself, but you probably did something very wrong.");
146                    return Ok(()); // Not an error which stops us ending ok.
147                }
148
149                match IS_RTE_RUNTIME_FINALIZED.lock() {
150                    Ok(mut finalized) => {
151                        if *finalized {
152                            error!("`runtime_fini()` called multiple times.  This does not cause problems by itself, but you probably did something very wrong.");
153                            return Ok(()); // Not fatal, but still wrong.
154                        }
155
156                        let errors = run_init_fini!(
157                            init = false,
158                            registry = RTE_INIT_RUNTIME_REGISTRY,
159                            rte_trait = RteInitRuntime,
160                            span_label = "Runtime Extension Node Finalization Span",
161                            ()
162                        );
163
164                        *finalized = true;
165                        errors
166                    },
167                    Err(e) => {
168                        let msg = "Poisoned `IS_RTE_RUNTIME_FINALIZED` on RTE Runtime Finalization. Should never happen.";
169
170                        error!(
171                            error = ?e,
172                            msg
173                        );
174
175                        let errors = RuntimeExtensionErrors::new();
176                        add_rte_error!(errors, RteMetadata::none(), ImpossibleError {
177                            description: msg.to_string(),
178                        });
179                        Err(errors)
180                    },
181                }
182            },
183            Err(e) => {
184                let msg = "Poisoned `IS_RTE_RUNTIME_INITIALIZED` on RTE Runtime Finalization. Should never happen.";
185                error!(
186                    error = ?e,
187                    msg
188                );
189
190                let errors = RuntimeExtensionErrors::new();
191                add_rte_error!(errors, RteMetadata::none(), ImpossibleError {
192                    description: msg.to_string(),
193                });
194                Err(errors)
195            },
196        }
197    }
198}
