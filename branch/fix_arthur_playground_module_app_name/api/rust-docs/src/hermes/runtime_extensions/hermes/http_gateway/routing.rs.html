50/// Main HTTP request router that processes incoming requests by hostname
51///
52/// Takes an HTTP request and routes it to the appropriate handler based on the
53/// hostname in the Host header. Each hostname corresponds to a different application.
54///
55/// ## Process:
56/// 1. Generates a unique request ID for tracking
57/// 2. Validates the hostname against allowed hosts
58/// 3. Routes valid requests to the Hermes application handler
59/// 4. Tracks connection state in the connection manager
60///
61///
62/// ## Example:
63/// A request with `Host: app.hermes.local` gets routed to the "app" application
64/// if "hermes.local" is in the valid hosts list.
65pub(crate) async fn router(
66    req: Request<Incoming>,
67    connection_manager: ConnectionManager,
68    ip: SocketAddr,
69    config: Config,
70) -> anyhow::Result<Response<Full<Bytes>>> {
71    let unique_request_id = EventUID(rusty_ulid::generate_ulid_string());
72    let client_ip = ClientIPAddr(ip);
73
74    // Register connection start
75    connection_manager.insert(
76        unique_request_id.clone(),
77        (client_ip.clone(), Processed(false), LiveConnection(true)),
78    );
79
80    #[cfg(debug_assertions)]
81    debug!("connection manager {:?}", connection_manager);
82
83    // Parse the Host header to extract the application name and resolved hostname
84    // This determines which application should handle the request
85    // For example: "app.hermes.local" -> app_name="app", resolved_host="hermes.local"
86    let (app_name, resolved_host) = host_resolver(req.headers())?;
87
88    // Check if the resolved hostname is in our list of valid/allowed hosts
89    // This is a security measure to prevent routing to unauthorized applications
90    let response = if config
91        .valid_hosts
92        .iter()
93        .any(|host| host.0 == resolved_host.0)
94    {
95        // If hostname is valid, route the request to the Hermes WASM runtime
96        // The app_name determines which specific application will handle it
97        route_to_hermes(req, app_name.clone()).await?
98    } else {
99        // If hostname is not in the valid list, reject with an error response
100        // This prevents potential security issues from unauthorized hosts
101        return error_response("Invalid hostname");
102    };
103
104    // Mark connection as processed
105    connection_manager.insert(
106        unique_request_id,
107        (client_ip, Processed(true), LiveConnection(false)),
108    );
109
110    #[cfg(debug_assertions)]
111    debug!("connection manager {connection_manager} app {app_name}");
112
113    Ok(response)
114}

134/// Extractor that resolves the hostname of the request.
135/// Hostname is resolved through the Host header
136pub(crate) fn host_resolver(headers: &HeaderMap) -> anyhow::Result<(ApplicationName, Hostname)> {
137    let host = headers
138        .get("Host")
139        .and_then(|h| h.to_str().ok())
140        .ok_or_else(|| anyhow!("Missing or invalid Host header"))?;
141
142    // Strip port if present (e.g., "app.hermes.local:5000" -> "app.hermes.local")
143    let host_without_port = host.split(':').next().unwrap_or(host);
144
145    // Parse <app.name>.hermes.local format
146    let (app, hostname) = host_without_port
147        .split_once('.')
148        .ok_or_else(|| anyhow!("Malformed Host header: expected format 'app.domain'"))?;
149
150    Ok((
151        ApplicationName(app.to_owned()),
152        Hostname(hostname.to_owned()),
153    ))
154}

156/// Main HTTP request router that processes incoming requests and delegates to appropriate
157/// handlers
158///
159/// This function serves as the central routing hub, taking validated HTTP requests and
160/// directing them to either WebAssembly modules or static file handlers based on the
161/// request path and routing rules.
162async fn route_to_hermes(
163    req: Request<Incoming>,
164    app_name: ApplicationName,
165) -> anyhow::Result<Response<Full<Bytes>>> {
166    // Extract the URI for route analysis - this contains path and query parameters
167    let uri = req.uri().to_owned();
168
169    // Analyze the request path and determine which type of handler should process it
170    // This applies our routing rules to classify the request appropriately
171    let route_type = determine_route(&uri, &req).await?;
172
173    match route_type {
174        // API endpoints need WebAssembly module processing
175        // These requests go through the event queue to WASM components
176        RouteType::WebAssembly(path, module_id) => {
177            handle_webasm_request(req, path, module_id, app_name).await
178            // Static files are served directly from the virtual file system
179        },
180        RouteType::StaticFile(path) => serve_static_web_content(&path, &app_name),
181    }
182}

184/// Route classification for incoming HTTP requests
185///
186/// This enum categorizes requests into two main types that require different handling:
187/// - API requests that need WebAssembly module processing
188/// - Static file requests that serve assets directly from the filesystem
189enum RouteType {
190    /// API endpoints that should be processed by WebAssembly modules
191    /// Contains the full path including query parameters for forwarding
192    WebAssembly(String, Option<String>),
193    /// Static file requests that serve assets (HTML, CSS, JS, images, etc.)
194    /// Contains the normalized file path for filesystem lookup
195    StaticFile(String),
196}

198/// Analyzes an HTTP request URI and determines the appropriate handler type
199///
200/// This function implements the core routing logic by examining the request path
201/// and applying routing rules to classify the request type.
202async fn determine_route(
203    uri: &hyper::Uri,
204    req: &Request<Incoming>,
205) -> anyhow::Result<RouteType> {
206    // Extract the full path including query parameters for accurate forwarding
207    // This preserves important data like ?asat=SLOT:95022059 that WASM modules need
208    let path = uri
209        .path_and_query()
210        .map_or(uri.path(), hyper::http::uri::PathAndQuery::as_str);
211
212    // Check if this is an API endpoint that needs WebAssembly processing
213    // API routes are identified by the /api prefix (exact match or with additional path)
214    let method = req.method().as_str();
215    let content_type = req
216        .headers()
217        .get("content-type")
218        .and_then(|ct| ct.to_str().ok());
219
220    if let Some(subscription) =
221        find_global_endpoint_subscription(method, uri.path(), content_type).await
222    {
223        info!(
224            "Found subscription for {} {}: module {}",
225            method,
226            uri.path(),
227            subscription.module_id
228        );
229        return Ok(RouteType::WebAssembly(
230            path.to_string(),
231            Some(subscription.module_id),
232        ));
233    }
234
235    // Check if this is an API endpoint that needs WebAssembly processing
236    // API routes are identified by the /api prefix (exact match or with additional path)
237    if uri.path() == WEBASM_ROUTE || uri.path().starts_with(&format!("{WEBASM_ROUTE}/")) {
238        Ok(RouteType::WebAssembly(path.to_string(), None))
239        // Check if this is a valid static file path
240        // Uses regex validation to ensure path safety and prevent directory traversal
241    } else if is_valid_path(uri.path()).is_ok() {
242        Ok(RouteType::StaticFile(path.to_string()))
243    } else {
244        // Reject invalid or potentially dangerous paths
245        Err(anyhow!("Invalid route: {}", uri.path()))
246    }
247}

249/// Processes HTTP requests for WebAssembly module execution
250///
251/// Converts incoming HTTP requests into WASM-compatible format and handles
252/// the async communication with WebAssembly modules.
253///
254/// ## Returns:
255/// HTTP response from the WebAssembly module
256async fn handle_webasm_request(
257    req: Request<Incoming>,
258    path: String,
259    module_id: Option<String>,
260    app_name: ApplicationName,
261) -> anyhow::Result<Response<Full<Bytes>>> {
262    let (lambda_send, lambda_recv_answer) = channel();
263    let method = req.method().to_string();
264
265    let headers: HeadersKV = req
266        .headers()
267        .iter()
268        .map(|(name, value)| {
269            let key = name.to_string();
270            let values = vec![value.to_str().unwrap_or_default().to_string()];
271            (key, values)
272        })
273        .collect();
274
275    let (_parts, body) = req.into_parts();
276    let body_bytes = body.collect().await?.to_bytes();
277
278    let request_params = HttpRequestParams {
279        method,
280        headers,
281        body: body_bytes,
282        path,
283        sender: lambda_send,
284    };
285
286    compose_http_event(request_params, &lambda_recv_answer, module_id, &app_name)
287}

303/// Composes and processes HTTP events for WebAssembly modules
304///
305/// This function handles the complete lifecycle of HTTP request processing:
306/// 1. Creates an HTTP event from request parameters
307/// 2. Determines target module(s) for routing
308/// 3. Sends event to the global queue
309/// 4. Waits for and processes the response
310fn compose_http_event<B>(
311    params: HttpRequestParams,
312    receiver: &Receiver<HTTPEventMsg>,
313    module_id: Option<String>,
314    app_name: &ApplicationName,
315) -> anyhow::Result<Response<B>>
316where
317    B: Body + From<Vec<u8>> + From<String>,
318{
319    let http_event = create_http_event(params);
320    let target_module = resolve_target_module(module_id, app_name)?;
321    let hermes_event = HermesEvent::new(http_event, TargetApp::All, target_module);
322
323    send_event_and_await_response(hermes_event, receiver)
324}

326/// Creates an `HTTPEvent` from the provided request parameters
327fn create_http_event(params: HttpRequestParams) -> HTTPEvent {
328    HTTPEvent {
329        headers: params.headers,
330        method: params.method,
331        path: params.path,
332        body: params.body,
333        sender: params.sender,
334    }
335}

337/// Resolves the target module(s) for routing the HTTP request
338///
339/// Returns:
340/// - `TargetModule::List` with specific module if found
341/// - `TargetModule::All` if module not found or not specified
342fn resolve_target_module(
343    module_id: Option<String>,
344    app_name: &ApplicationName,
345) -> anyhow::Result<TargetModule> {
346    let app = reactor::get_app(app_name)?;
347    let modules = app.get_module_registry();
348
349    let target_module = match module_id {
350        None => {
351            info!("Broadcasting HTTP request to all modules (no specific subscription)");
352            TargetModule::All
353        },
354        Some(target_module_str) => {
355            info!(
356                "Routing HTTP request to specific module: {}",
357                target_module_str
358            );
359
360            if let Some(found_module_id) = modules.get(&target_module_str) {
361                TargetModule::List(vec![found_module_id.clone()])
362            } else {
363                info!(
364                    "Module '{}' not found in available modules: {:?}, broadcasting to all",
365                    target_module_str,
366                    modules.keys().collect::<Vec<_>>()
367                );
368                TargetModule::All
369            }
370        },
371    };
372
373    Ok(target_module)
374}

376/// Sends the event to the global queue and waits for a response
377fn send_event_and_await_response<B>(
378    event: HermesEvent,
379    receiver: &Receiver<HTTPEventMsg>,
380) -> anyhow::Result<Response<B>>
381where
382    B: Body + From<Vec<u8>> + From<String>,
383{
384    crate::event::queue::send(event)?;
385
386    let timeout = Duration::from_secs(EVENT_TIMEOUT);
387    match receiver.recv_timeout(timeout)? {
388        HTTPEventMsg::HttpEventResponse((status_code, headers, body)) => {
389            let mut response_builder = Response::builder().status(status_code);
390
391            // Add headers to response
392            for (key, values) in headers {
393                for value in values {
394                    response_builder = response_builder.header(&key, value);
395                }
396            }
397            let response = response_builder.body(body.into())?;
398            Ok(response)
399        },
400        HTTPEventMsg::HTTPEventReceiver => error_response("Invalid HTTP event message received"),
401    }
402}

404/// Compiled regex for validating static file request paths
405///
406/// Lazily compiled from `VALID_PATH` pattern for efficient path matching.
407/// Returns `None` if the regex pattern is invalid.
408static VALID_PATH_REGEX: LazyLock<Option<Regex>> = LazyLock::new(|| Regex::new(VALID_PATH).ok());
409
410/// Validates HTTP request paths for static file serving
411///
412/// Uses regex pattern matching to ensure paths are safe for VFS access.
413/// Allows root path "/" and absolute paths "/path/to/resource".
414///
415/// Primary security relies on VFS sandboxing; this provides basic validation.
416fn is_valid_path(path: &str) -> anyhow::Result<()> {
417    VALID_PATH_REGEX
418        .as_ref()
419        .ok_or_else(|| anyhow::anyhow!("Path validation unavailable: invalid regex pattern"))?
420        .is_match(path)
421        .then_some(())
422        .ok_or_else(|| anyhow::anyhow!("Invalid path format: {path:?}"))
423}

425/// Serves static web assets for Flutter applications with comprehensive error handling
426///
427/// This is the primary function for handling static file requests in the Hermes web
428/// server. It manages the complete lifecycle of serving web assets from the Virtual File
429/// System (VFS), including MIME type detection, caching headers, security policies, and
430/// fallback handling.
431///
432/// ## Core Functionality:
433///
434/// ### 1. Path Resolution & VFS Access
435/// - Converts HTTP request paths to VFS file paths using `resolve_static_file_path()`
436/// - Accesses the application's Virtual File System through the reactor
437/// - Handles both direct file requests and Flutter routing scenarios
438fn serve_static_web_content<B>(
439    path: &str,
440    app_name: &ApplicationName,
441) -> anyhow::Result<Response<B>>
442where
443    B: Body + From<Vec<u8>>,
444{
445    let file_path = resolve_static_file_path(path);
446    let app = reactor::get_app(app_name)?;
447
448    app.vfs()
449        .read(&file_path)
450        .and_then(|contents| serve_existing_asset(contents, &file_path))
451        .or_else(|_| handle_missing_asset(&file_path, &app))
452}

454/// Default file path for root requests - Flutter application entry point
455const DEFAULT_INDEX_PATH: &str = "www/index.html";
456/// Document root directory in the VFS where static web assets are stored
457const DOCUMENT_ROOT: &str = "www";
458
459/// Resolves incoming HTTP request paths to actual file paths within the Virtual File
460/// System (VFS)
461///
462/// This function performs path normalization and translation for serving static files in
463/// a Flutter web application. It implements the standard web server convention where the
464/// document root maps to a specific VFS directory.
465///
466/// ## Path Resolution Rules:
467///
468/// ### Root Path Handling:
469/// - **Input**: `"/"` (root/index request)
470/// - **Output**: `"www/index.html"`
471/// - **Purpose**: Serves the main Flutter application entry point
472/// - **Example**: `GET /` → serves `www/index.html` from VFS
473///
474/// ### Static File Path Translation:
475/// - **Input**: `"/assets/app.js"` (any non-root path)
476/// - **Process**: Remove leading `/`, prepend `www/`
477/// - **Output**: `"www/assets/app.js"`
478/// - **Purpose**: Maps URL paths to VFS file locations
479///
480/// ## VFS Structure Context:
481/// VFS Root
482/// |-- www/                    <- Document root directory
483///     |-- index.html         <- Main application entry point
484///     |-- flutter.js         <- Flutter framework loader
485///     |-- `flutter_service_worker.js`  <- Service worker
486///     |-- assets/            <- Static assets directory
487///         |-- fonts/         <- Font files
488///         |-- images/        <- Image assets
489///         |-- packages/      <- Dart package assets
490///
491/// ## Security Considerations:
492/// - **Path traversal protection**: This function doesn't validate against `../` attacks
493///   (assumes upstream validation by HTTP router/framework)
494/// - **VFS isolation**: All resolved paths are contained within the `www/` directory
495/// - **No filesystem access**: Works purely with VFS paths, not real filesystem paths
496fn resolve_static_file_path(path: &str) -> String {
497    if path == "/" {
498        DEFAULT_INDEX_PATH.to_string()
499    } else {
500        let clean_path = path.strip_prefix('/').unwrap_or(path);
501        format!("{DOCUMENT_ROOT}/{clean_path}")
502    }
503}

505/// Serves an existing asset file with appropriate headers
506fn serve_existing_asset<B>(
507    file_contents: Vec<u8>,
508    file_path: &str,
509) -> anyhow::Result<Response<B>>
510where
511    B: Body + From<Vec<u8>>,
512{
513    let mut response = Response::new(file_contents.into());
514
515    if let Some(extension) = get_file_extension(file_path) {
516        let content_type = get_flutter_content_type(extension);
517        response
518            .headers_mut()
519            .insert("Content-Type", content_type.parse()?);
520    }
521
522    add_security_headers(response)
523}

525/// Handles missing asset files with appropriate fallback logic
526fn handle_missing_asset<B>(
527    file_path: &str,
528    app: &Application,
529) -> anyhow::Result<Response<B>>
530where
531    B: Body + From<Vec<u8>>,
532{
533    if is_critical_asset(file_path) {
534        error!("Critical asset missing: {}", file_path);
535        return Ok(Response::builder()
536            .status(StatusCode::NOT_FOUND)
537            .body("Asset not found".as_bytes().to_vec().into())?);
538    }
539
540    // Fall back to index.html for navigation routes
541    serve_index_html_fallback(file_path, app)
542}

552/// Adds security headers to HTTP responses for Flutter web applications
553///
554/// This function implements Cross-Origin Isolation by setting two critical security
555/// headers that work together to create a secure browsing context.
556///
557/// ## Headers Added:
558///
559/// ### Cross-Origin-Opener-Policy (COOP): "same-origin"
560/// - **Purpose**: Isolates the browsing context from cross-origin windows
561/// - **Effect**: Prevents other origins from accessing this window object
562/// - **Security**: Protects against certain types of cross-origin attacks
563/// - **Compatibility**: Allows same-origin popups/windows to communicate normally
564///
565/// ### Cross-Origin-Embedder-Policy (COEP): "require-corp"
566/// - **Purpose**: Requires all embedded resources to explicitly opt-in to cross-origin
567///   loading
568/// - **Effect**: Blocks cross-origin resources without proper CORS or
569///   Cross-Origin-Resource-Policy headers
570/// - **Security**: Prevents malicious resource injection from untrusted origins
571/// - **Requirement**: All cross-origin assets need `crossorigin` attribute or CORP
572///   headers
573fn add_security_headers<B>(mut response: Response<B>) -> anyhow::Result<Response<B>>
574where B: Body {
575    let headers = response.headers_mut();
576
577    // Enable Cross-Origin Isolation for advanced web features
578    headers.insert(
579        "Cross-Origin-Opener-Policy",
580        "same-origin"
581            .parse()
582            .map_err(|e| anyhow!("Invalid COOP header value: {e}"))?,
583    );
584    headers.insert(
585        "Cross-Origin-Embedder-Policy",
586        "require-corp"
587            .parse()
588            .map_err(|e| anyhow!("Invalid COEP header value: {e}"))?,
589    );
590
591    Ok(response)
592}

599/// Returns the appropriate MIME type for Flutter web assets based on file extension
600///
601/// Maps common web asset file extensions to their corresponding MIME types
602/// for proper browser handling. Defaults to `application/octet-stream` for
603/// unknown extensions.
604fn get_flutter_content_type(extension: &str) -> &'static str {
605    match extension {
606        "html" => "text/html",
607        "js" => "application/javascript",
608        "wasm" => "application/wasm",
609        "css" => "text/css",
610        "json" => "application/json",
611        "png" => "image/png",
612        "jpg" | "jpeg" => "image/jpeg",
613        "svg" => "image/svg+xml",
614        "ico" => "image/x-icon",
615        "woff" | "woff2" => "font/woff2",
616        "ttf" => "font/ttf",
617        "otf" => "font/otf",
618        _ => "application/octet-stream",
619    }
620}

622/// HTTP Content-Type header value for HTML documents
623///
624/// Used when serving HTML files, particularly the index.html fallback for Flutter
625/// single-page applications. This MIME type tells browsers to parse the content
626/// as HTML markup.
627const CONTENT_TYPE_HTML: &str = "text/html";
628
629/// Cache-Control header directive to prevent caching
630///
631/// Instructs browsers and intermediate caches to:
632/// - `no-cache`: Always revalidate with server before using cached content
633/// - `no-store`: Never store the response in any cache
634/// - `must-revalidate`: Require revalidation of stale cache entries
635///
636/// Used for index.html to ensure users always get the latest version of the
637/// Flutter application, preventing issues with cached outdated app shells.
638const NO_CACHE_DIRECTIVE: &str = "no-cache, no-store, must-revalidate";
639
640/// Serves index.html as fallback for navigation routes
641fn serve_index_html_fallback<B>(
642    file_path: &str,
643    app: &Application,
644) -> anyhow::Result<Response<B>>
645where
646    B: Body + From<Vec<u8>>,
647{
648    match file_path {
649        "www/index.html" => not_found(),
650        _ => {
651            match app.vfs().read("www/index.html") {
652                Ok(index_contents) => {
653                    let mut response = Response::new(index_contents.into());
654                    response
655                        .headers_mut()
656                        .insert("Content-Type", CONTENT_TYPE_HTML.parse()?);
657                    response
658                        .headers_mut()
659                        .insert("Cache-Control", NO_CACHE_DIRECTIVE.parse()?);
660                    response = add_security_headers(response)?;
661                    Ok(response)
662                },
663                Err(_) => not_found(),
664            }
665        },
666    }
667}

677    fn test_valid_paths_regex() {
678        // ^ and $: Match the entire string/line
679        // (/[a-zA-Z0-9-_]+)+: One or more directories, starting with slash, separated by
680        // slashes; each directory must consist of one or more characters of your charset.
681        // (...)+|/: Explicitly allow just a single slash
682        let regex = Regex::new(VALID_PATH).unwrap();
683
684        let example_one = "/abc/def";
685        let example_two = "/hello_1/world";
686        let example_three = "/three/directories/abc";
687        let example_four = "/";
688        let valid_path = vec![example_one, example_two, example_three, example_four];
689
690        for valid in valid_path {
691            if let Some(captures) = regex.captures(valid) {
692                assert_eq!(captures.get(0).unwrap().as_str(), valid);
693            }
694
695            assert!(is_valid_path(valid).is_ok());
696        }
697
698        let example_one = "/abc/def/";
699        let example_two = "/abc//def";
700        let example_three = "//";
701        let example_four = "abc/def";
702        let example_five = "/abc/def/file.txt";
703        let invalids = vec![
704            example_one,
705            example_two,
706            example_three,
707            example_four,
708            example_five,
709        ];
710
711        for invalid in invalids {
712            if let Some(captures) = regex.captures(invalid) {
713                assert!(captures.len() == 0);
714            }
715
716            assert!(is_valid_path(invalid).is_err());
717        }
718    }
