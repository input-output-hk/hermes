1//! Blake2b-256 hash implementation.
2
3/// Blake2b-256 hash size.
4const HASH_SIZE: usize = 32;
5
6/// Blake2b-256 hasher instance.
7pub(crate) struct Blake2b256Hasher(blake2b_simd::State);
8
9impl Blake2b256Hasher {
10    /// Create a new `Blake2b256Hasher`.
11    pub(crate) fn new() -> Self {
12        Self(
13            blake2b_simd::Params::new()
14                .hash_length(HASH_SIZE)
15                .to_state(),
16        )
17    }
18
19    /// Incrementally add bytes to the hasher.
20    pub(crate) fn update(
21        &mut self,
22        bytes: &[u8],
23    ) {
24        self.0.update(bytes);
25    }
26
27    ///  Finalize the state and return a `Hash`.
28    pub(crate) fn finalize(self) -> Blake2b256 {
29        let hash = self.0.finalize();
30        Blake2b256::from_bytes_unchecked(hash.as_bytes())
31    }
32}
33
34/// Blake2b-256 hash instance.
35#[cfg_attr(debug_assertions, derive(Debug))]
36#[derive(Clone, PartialEq, Eq, Hash)]
37pub(crate) struct Blake2b256([u8; HASH_SIZE]);
38
39impl Blake2b256 {
40    /// Create a new `Blake2b256` from bytes.
41    /// It does not validate the size of the bytes, so all checks should be done by
42    /// the caller.
43    fn from_bytes_unchecked(bytes: &[u8]) -> Self {
44        let mut hash_bytes = [0; HASH_SIZE];
45        hash_bytes.copy_from_slice(bytes);
46
47        Self(hash_bytes)
48    }
49
50    /// Calculate a new `Blake2b256` from bytes.
51    pub(crate) fn hash(bytes: &[u8]) -> Self {
52        let hash = blake2b_simd::Params::new()
53            .hash_length(HASH_SIZE)
54            .hash(bytes);
55
56        Self::from_bytes_unchecked(hash.as_bytes())
57    }
58
59    /// Convert the hash to a hexadecimal string.
60    pub(crate) fn to_hex(&self) -> String {
61        hex::encode(self.0)
62    }
63
64    /// Create a new `Blake2b256` from hex string.
65    pub(crate) fn from_hex(s: &str) -> anyhow::Result<Self> {
66        let bytes = hex::decode(s)?;
67        Self::from_bytes(&bytes)
68    }
69
70    /// Return the hash bytes.
71    pub(crate) fn to_bytes(&self) -> [u8; HASH_SIZE] {
72        self.0
73    }
74
75    /// Create a new `Blake2b256` from bytes.
76    pub(crate) fn from_bytes(bytes: &[u8]) -> anyhow::Result<Self> {
77        anyhow::ensure!(
78            bytes.len() == HASH_SIZE,
79            "Invalid hash length: expected {}, provided {}.",
80            HASH_SIZE,
81            bytes.len()
82        );
83        let hash = Self::from_bytes_unchecked(bytes);
84        Ok(hash)
85    }
86}

88#[cfg(all(test, debug_assertions))]
89mod tests {
90    use super::*;
91
92    #[test]
93    fn hash_serde_test() {
94        let hash = Blake2b256::hash(b"test");
95
96        let hex = hash.to_hex();
97        assert_eq!(
98            "928b20366943e2afd11ebc0eae2e53a93bf177a4fcf35bcc64d503704e65e202",
99            hex
100        );
101
102        let decoded_hash = Blake2b256::from_hex(&hex).unwrap();
103        assert_eq!(hash, decoded_hash);
104    }
105
106    #[test]
107    fn hasher_test() {
108        let mut hasher = Blake2b256Hasher::new();
109        hasher.update(b"test");
110        let hash = hasher.finalize();
111        assert_eq!(
112            "928b20366943e2afd11ebc0eae2e53a93bf177a4fcf35bcc64d503704e65e202",
113            hash.to_hex()
114        );
115    }
