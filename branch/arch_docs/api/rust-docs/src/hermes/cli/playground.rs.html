24/// Hermes application playground
25///
26/// Loads each wasm component from the arguments as a standalone Hermes application
27/// consisting of a single module with a temporary VFS.
28///
29/// This allows to run Wasm components without properly packaging them.
30///
31/// Returns an exit code that can be inspected for custom values issued by Wasm
32/// components.
33///
34/// If an internal error occurred returns 101.
35#[cfg_attr(debug_assertions, derive(Debug))]
36#[derive(clap::Args)]
37pub struct Playground {
38    /// Wasm components to load as apps in this example
39    components: Vec<PathBuf>,
40
41    /// Select the name for the app.
42    #[arg(long, default_value = "playground-app")]
43    app_name: String,
44
45    /// Shutdown the playground after the timeout (milliseconds)
46    #[arg(long)]
47    timeout_ms: Option<u64>,
48}
49
50impl Playground {
51    /// Run playground CLI command
52    pub fn exec(self) -> anyhow::Result<Exit> {
53        let exit_lock = reactor::init()?;
54
55        println!("{} Running a playground...", Emoji::new("‚ú®", ""));
56
57        ensure!(
58            !self.components.is_empty(),
59            "At least one component is required to run"
60        );
61
62        let temp_dir = TempDir::new()?;
63
64        let app =
65            create_and_init_app_with_temp_dir_vfs(self.app_name, &self.components, &temp_dir)?;
66
67        tracing::info!("{} Bootstrapping IPFS node", console::Emoji::new("üñß", ""),);
68        init_ipfs(&temp_dir)?;
69
70        pool::init()?;
71        println!("{} Loading an application...", Emoji::new("üõ†Ô∏è", ""),);
72
73        reactor::load_app(app)?;
74
75        let exit = if let Some(timeout_ms) = self.timeout_ms {
76            exit_lock.wait_timeout(Duration::from_millis(timeout_ms))
77        } else {
78            exit_lock.wait()
79        };
80
81        // Wait for scheduled tasks to be finished.
82        pool::terminate();
83        reactor::drop_all_apps()?;
84        Ok(exit)
85    }
86}

88/// Initialize the IPFS node
89fn init_ipfs(temp_dir: &TempDir) -> anyhow::Result<()> {
90    let ipfs_dir = create_temp_dir_child(temp_dir, Path::new("ipfs"))?;
91    // disable bootstrapping the IPFS node to default addresses for testing
92    let default_bootstrap = false;
93    ipfs::bootstrap(ipfs_dir.as_path(), default_bootstrap)
94}

96/// Collects `.wasm` files in the current directory or sub-directories of the current
97/// directory. Return a [`String`] module name along with each compiled [`Module`].
98fn collect_modules(
99    app_name: &str,
100    components: &[PathBuf],
101) -> anyhow::Result<Vec<(String, Module)>> {
102    // All wasm components in a directory.
103    let mut modules = Vec::new();
104
105    // Collect component files
106    for file_path in components {
107        let name = file_path
108            .file_stem()
109            .and_then(OsStr::to_str)
110            .ok_or_else(|| anyhow!("Provided path is invalid: {}", file_path.display()))?
111            .to_string();
112        let wasm_buf = fs::read(file_path)?;
113        let module = Module::from_bytes(&ApplicationName::new(app_name), &wasm_buf)?;
114        modules.push((name, module));
115    }
116
117    Ok(modules)
118}

120/// Create an application with provided VFS.
121fn create_app(
122    app_name: String,
123    vfs_dir_path: &Path,
124    named_modules: Vec<(String, Module)>,
125) -> anyhow::Result<Application> {
126    let vfs_name = [&app_name, "_vfs"].concat();
127    let vfs = VfsBootstrapper::new(vfs_dir_path, vfs_name).bootstrap()?;
128    let module_registry = named_modules
129        .iter()
130        .map(|(name, module)| (name.to_owned(), module.id().clone()))
131        .collect();
132    let modules = named_modules
133        .into_iter()
134        .map(|(_, module)| module)
135        .collect();
136    let app = Application::new(ApplicationName(app_name), vfs, modules, module_registry);
137
138    Ok(app)
139}

141/// Create a temp subdirectory.
142fn create_temp_dir_child(
143    temp_dir: &TempDir,
144    child_path: &Path,
145) -> anyhow::Result<PathBuf> {
146    let child_absolute_path = temp_dir.path().join(child_path);
147    fs::create_dir_all(child_absolute_path.as_path())?;
148    Ok(child_absolute_path)
149}

151/// Collects `.wasm` files in the current directory or sub-directories of the current
152/// directory. Then creates an applications out of all of them.
153fn create_and_init_app_with_temp_dir_vfs(
154    app_name: String,
155    components: &[PathBuf],
156    temp_dir: &TempDir,
157) -> anyhow::Result<Application> {
158    let named_modules = collect_modules(&app_name, components)?;
159
160    let vfs_dir_path = create_temp_dir_child(temp_dir, Path::new("vfs"))?;
161    let app = create_app(app_name, &vfs_dir_path, named_modules)?;
162
163    RteApp::new().init(app.name())?;
164
165    Ok(app)
166}
