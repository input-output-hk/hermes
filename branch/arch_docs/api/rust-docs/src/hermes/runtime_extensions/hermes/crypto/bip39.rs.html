14/// Generate a new extended private key (`XPrv`) from a mnemonic and passphrase.
15///
16/// # Arguments
17///
18/// - `mnemonic`: A string representing the mnemonic.
19/// - `passphrase`: An optional string representing the passphrase.
20///
21/// # Returns
22///
23/// Returns the `XPrv` extended private key as a `Result`.
24/// If the conversion is successful, it returns `Ok` with the extended private key
25/// (`XPrv`).
26///
27/// # Errors
28///
29/// - `InvalidMnemonic`: If the mnemonic should be either 12, 15, 18, 21, or 24.
30pub(crate) fn mnemonic_to_xprv(
31    mnemonic: &str,
32    passphrase: &str,
33) -> Result<XPrv, Errno> {
34    /// 4096 is the number of iterations for PBKDF2.
35    const ITER: u32 = 4096;
36
37    // Parse will detect language and check mnemonic valid length
38    // 12, 15, 18, 21, 24 are valid mnemonic length
39    let mnemonic = Mnemonic::parse(mnemonic).map_err(|_| Errno::InvalidMnemonic)?;
40
41    let entropy = mnemonic.to_entropy();
42
43    // This implementation follows SLIP-0023 - Cardano HD master node derivation
44    // from a master seed.
45    // https://github.com/satoshilabs/slips/blob/master/slip-0023.md
46    let mut pbkdf2_result = [0; 96];
47    let _ = pbkdf2::<Hmac<Sha512>>(passphrase.as_bytes(), &entropy, ITER, &mut pbkdf2_result);
48
49    Ok(XPrv::normalize_bytes_force3rd(pbkdf2_result))
50}

52/// Generate a new mnemonic.
53/// Number of bytes entropy required is calculated by `word_count` / 3 * 4.
54/// eg. 24 words will have 32 bytes entropy or 256 bits entropy.
55/// Number of checksum required is calculated by `word_count` / 3.
56/// eg. 24 words will have 8 checksum bits.
57///
58/// # Arguments
59///
60/// - `word_count`: The number of words in the mnemonic. Must be a multiple of 3 and in
61///   the range of 12 - 24.
62/// - `prefix`: A vector of strings representing the prefix, empty if no prefix.
63/// - `language`: The language of the mnemonic.
64///
65/// # Returns
66///
67/// Returns the mnemonic of type `String` as a `Result`.
68/// If the conversion is successful, it returns `Ok` with the mnemonic.
69/// If there is an error during the computation, it returns `Err` with an `Errno`.
70///
71/// # Errors
72///
73/// - `InvalidMnemonicLength`: If the word count is not a multiple of 3 or not in the
74///   range of 12 - 24.
75/// - `PrefixTooLong`: If the prefix is longer than the maximum allowed length, max is 3.
76/// - `WordNotFound`: If a word in the mnemonic is not found in the word list.
77pub(crate) fn generate_new_mnemonic(
78    word_count: usize,
79    prefix: Vec<String>,
80    language: Option<String>,
81) -> Result<Vec<String>, Errno> {
82    // Check word count
83    if is_invalid_word_count(word_count) {
84        return Err(Errno::InvalidMnemonicLength);
85    }
86
87    // Number of prefix word should be <= 3.
88    if prefix.len() > 3 {
89        return Err(Errno::PrefixTooLong);
90    }
91
92    let language = language.map_or_else(
93        || string_to_language("English"),
94        |lang| string_to_language(&lang),
95    )?;
96
97    let prefix_index_bits = get_prefix_index_bits(prefix, language)?;
98
99    // Create an vec that will hold binary conversion of entropy.
100    let mut entropy_bits = Vec::new();
101    // Add the prefix index bit to the bit entropy.
102    entropy_bits.extend_from_slice(&prefix_index_bits);
103
104    let entropy = generate_entropy(word_count)?;
105
106    // Convert bytes entropy to bits.
107    byte_to_bit(entropy, &mut entropy_bits, word_count);
108
109    let check_sum_bits = get_check_sum_bits(&entropy_bits, word_count);
110    // Add the checksum bits to the end of bit entropy.
111    entropy_bits.extend_from_slice(&check_sum_bits);
112
113    let word_indices = get_word_indices(&entropy_bits, word_count);
114    let mnemonic_list = get_mnemonic_from_indices(word_indices, language);
115
116    Ok(mnemonic_list)
117}

119/// Check if the word count is invalid.
120/// Valid word count is a multiple of 3 and in the range of 12 - 24.
121/// Returns true if the word count is invalid, otherwise false.
122fn is_invalid_word_count(word_count: usize) -> bool {
123    word_count < 12 || word_count % 3 != 0 || word_count > 24
124}

126/// Get the index bits of the prefix words from a BIP39 dictionary.
127fn get_prefix_index_bits(
128    prefix_list: Vec<String>,
129    language: Language,
130) -> Result<Vec<u8>, Errno> {
131    let mut prefix_index: Vec<u8> = Vec::new();
132    for word in prefix_list {
133        match language.find_word(&word) {
134            Some(index) => {
135                for b in decimal_to_binary_array(index) {
136                    prefix_index.push(b);
137                }
138            },
139            None => return Err(Errno::WordNotFound),
140        }
141    }
142    Ok(prefix_index)
143}

145/// Generate entropy bytes and return the value.
146///
147/// Calculation:
148/// Each word in a BIP39 mnemonic phrase represents 11 bits of information. The total
149/// number of bits in the mnemonic phrase (including both entropy and checksum) is
150/// therefore:
151///
152/// `total_bits` = `word_count` * 11
153///
154/// The total number of bits includes both the entropy bits and the checksum bits. The
155/// length of the checksum is defined as:
156///
157/// `checksum_len` = `entropy_bits` / 32
158///
159/// Therefore, the total number of bits can be written as:
160///
161/// `total_bits` = `entropy_bits` + `checksum_len`
162///              = `entropy_bits` + `entropy_bits` / 32
163///              = `entropy_bits` * (1 + 1/32)
164///              = `entropy_bits` * 33 / 32
165///
166/// Since `total_bits` is also equal to `word_count * 11`, we have:
167///
168/// `word_count` * 11 = `entropy_bits` * 33 / 32
169///
170/// Solving for `entropy_bits`, we get:
171///
172/// `entropy_bits` = `word_count` * 11 * 32 / 33
173///
174/// To find the number of entropy bytes, we need to divide the entropy bits by 8:
175///
176/// `total_entropy_bytes` = `entropy_bits` / 8
177///
178/// Simplifying further, we get:
179///
180/// `total_entropy_bytes` = (`word_count` * 11 * 32 / 33) / 8
181///                       = `word_count` * 11 * 4 / 33
182///                       = `word_count` * 4 / 3
183///
184/// However, since the number of mnemonic words is always a multiple of 3
185/// (in BIP39, valid word counts are 12, 15, 18, 21, or 24),
186/// we can simplify this to:
187///
188/// `total_entropy_bytes` = `word_count` * 4 / 3
189///
190/// Note that if entropy bits is needed, multiply the `total_entropy_bytes` by 8.
191fn generate_entropy(word_count: usize) -> Result<Vec<u8>, Errno> {
192    // Number of bytes entropy calculate from mnemonic word.
193    let total_entropy_bytes = word_count.saturating_mul(4).div_ceil(3);
194    // Maximum length of mnemonic is 24 words which is 32 bytes entropy.
195    let mut total_entropy_bytes_max = [0u8; 32];
196    // Random number
197    let mut rng = rand::rng();
198    // Fill the random number into total_entropy_bytes.
199    if let Some(slice) = total_entropy_bytes_max.get_mut(0..total_entropy_bytes) {
200        rng.fill_bytes(slice);
201    }
202
203    if let Some(slice) = total_entropy_bytes_max.get(0..total_entropy_bytes) {
204        Ok(slice.to_vec())
205    } else {
206        Err(Errno::GenerateEntropyFailed)
207    }
208}

210/// Generate checksum bits from entropy bits.
211#[allow(clippy::indexing_slicing)]
212fn get_check_sum_bits(
213    entropy_bits: &[u8],
214    word_count: usize,
215) -> Vec<u8> {
216    let checksum_len = word_count / 3;
217    // Convert entropy_bits to bytes, so it works with SHA256 hasher.
218    let entropy_bytes = bits_to_bytes(entropy_bits);
219
220    let hash_result = Sha256::digest(entropy_bytes);
221
222    // Retrieve the first `checksum_len` checksum bits from the hash result.
223    let mut checksum_bits = Vec::new();
224    for i in 0..checksum_len {
225        checksum_bits.push((hash_result[0] >> 7_usize.saturating_sub(i)) & 1);
226    }
227    checksum_bits
228}

230/// Get the word indices from the entropy bits.
231fn get_word_indices(
232    entropy_bits: &[u8],
233    word_count: usize,
234) -> Vec<u16> {
235    let mut word_index_vec = Vec::new();
236
237    // Separate entropy bits into 11 bits and convert to decimal.
238    // This decimal will be used to get the word index.
239    for i in 0..word_count {
240        let mut idx: u16 = 0;
241        for j in 0..11 {
242            if let Some(value) = entropy_bits.get(i.saturating_mul(11).saturating_add(j)) {
243                if *value > 0 {
244                    idx = idx.saturating_add(1 << (10_usize.saturating_sub(j)));
245                }
246            }
247        }
248        word_index_vec.push(idx);
249    }
250    word_index_vec
251}

253/// Get the mnemonic from the BIP39 dictionary using word indices.
254fn get_mnemonic_from_indices(
255    word_index_vec: Vec<u16>,
256    language: Language,
257) -> Vec<String> {
258    let word_list = language.word_list();
259    let mut mnemonic: Vec<String> = vec![];
260    for word in word_index_vec {
261        if let Some(word) = word_list.get(word as usize) {
262            mnemonic.push((*(word)).to_string());
263        }
264    }
265    mnemonic
266}

268/// Turns decimal into binary array of length 11.
269fn decimal_to_binary_array(decimal: u16) -> [u8; 11] {
270    let mut binary = [0u8; 11];
271    let mut n = decimal;
272    let mut index = 0;
273
274    while n > 0 {
275        if let Some(value) = binary.get_mut(index) {
276            let bit = n % 2;
277            index = index.saturating_add(1);
278            *value = bit as u8;
279            n /= 2;
280        }
281    }
282
283    // If the number of bits is less than 11, fill the remaining bits with 0.
284    while index < 11 {
285        if let Some(value) = binary.get_mut(index) {
286            *value = 0;
287        }
288        index = index.saturating_add(1);
289    }
290    binary.reverse();
291    binary
292}

294/// Turns bit of array to bytes of arrays
295fn bits_to_bytes(bits: &[u8]) -> Vec<u8> {
296    bits.chunks(8)
297        .map(|chunk| {
298            chunk.iter().enumerate().fold(0, |acc, (i, &bit)| {
299                acc | ((bit) << ((7_usize).saturating_sub(i)))
300            })
301        })
302        .collect()
303}

305/// Convert string to BIP39 language.
306fn string_to_language(s: &str) -> Result<Language, Errno> {
307    match s {
308        "English" => Ok(Language::English),
309        "SimplifiedChinese" => Ok(Language::SimplifiedChinese),
310        "TraditionalChinese" => Ok(Language::TraditionalChinese),
311        "Czech" => Ok(Language::Czech),
312        "French" => Ok(Language::French),
313        "Italian" => Ok(Language::Italian),
314        "Japanese" => Ok(Language::Japanese),
315        "Korean" => Ok(Language::Korean),
316        "Spanish" => Ok(Language::Spanish),
317        _ => Err(Errno::UnsupportedLanguage),
318    }
319}

321/// Convert bytes entropy to bits.
322///
323/// `word_count` must be a multiple of 3 and 12-24
324/// It will return garbage if that pre-condition does not hold.
325fn byte_to_bit(
326    entropy: Vec<u8>,
327    entropy_bits: &mut Vec<u8>,
328    word_count: usize,
329) {
330    for byte in entropy {
331        for j in (0..8).rev() {
332            // Should not exceed the word_count / 3 * 32
333            // which is number of entropy bits for the mnemonic word count.
334            if entropy_bits.len() >= word_count.div_ceil(3).saturating_mul(32) {
335                break;
336            }
337            entropy_bits.push((byte >> j) & 1);
338        }
339    }
340}
