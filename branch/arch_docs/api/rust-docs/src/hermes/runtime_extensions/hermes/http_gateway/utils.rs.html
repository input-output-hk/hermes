1//! Utility functions for HTTP gateway
2
3use hyper::{body::Body, HeaderMap, Response};
4
5use crate::runtime_extensions::hermes::http_gateway::event::HeadersKV;

7/// Extract headers from request into `HeadersKV` format.
8pub(crate) fn extract_headers_kv(headers: &HeaderMap) -> HeadersKV {
9    headers
10        .iter()
11        .map(|(name, value)| {
12            (name.to_string(), vec![value
13                .to_str()
14                .unwrap_or_default()
15                .to_string()])
16        })
17        .collect()
18}
19
20/// Build HTTP response from status code, headers, and body
21pub(crate) fn build_http_response<B>(
22    status_code: u16,
23    headers: Vec<(String, Vec<String>)>,
24    body: Vec<u8>,
25) -> anyhow::Result<Response<B>>
26where
27    B: Body + From<Vec<u8>>,
28{
29    let mut response_builder = Response::builder().status(status_code);
30    for (key, values) in headers {
31        for value in values {
32            response_builder = response_builder.header(&key, value);
33        }
34    }
35    Ok(response_builder.body(body.into())?)
36}
