23/// Main State for Cardano blockchain runtime extension.
24struct State {
25    /// Network resource storage.
26    network: ApplicationResourceStorage<Network, cardano_blockchain_types::Network>,
27    /// Lookup map for network resource.
28    network_lookup: DashMap<(ApplicationName, cardano_blockchain_types::Network), u32>,
29    /// Block resource storage.
30    block: ApplicationResourceStorage<Block, cardano_blockchain_types::MultiEraBlock>,
31    /// Transaction resource storage.
32    // Use (MultiEraBlock, slot) instead of pallas::ledger::traverse::MultiEraTx due to life time
33    // issue
34    transaction:
35        ApplicationResourceStorage<Transaction, (cardano_blockchain_types::MultiEraBlock, u16)>,
36    /// Subscription ID resource storage.
37    subscription_id: ApplicationResourceStorage<SubscriptionId, cardano_blockchain_types::Network>,
38    /// Active subscription ID to its subscription type and network handler.
39    subscriptions: DashMap<u32, (SubscriptionType, network::Handle)>,
40    /// Chain sync state of a specific network.
41    #[allow(clippy::struct_field_names)]
42    sync_state: DashMap<cardano_blockchain_types::Network, tokio::task::JoinHandle<()>>,
43}
44
45/// Block subscription type.
46#[derive(PartialEq)]
47pub(crate) enum SubscriptionType {
48    /// Normal block subscription.
49    Block,
50    /// Immutable roll forward block subscription.
51    ImmutableRollForward,
52}
53
54/// Initialize state
55static STATE: once_cell::sync::Lazy<State> = once_cell::sync::Lazy::new(|| {
56    State {
57        network: ApplicationResourceStorage::new(),
58        network_lookup: DashMap::new(),
59        block: ApplicationResourceStorage::new(),
60        transaction: ApplicationResourceStorage::new(),
61        subscription_id: ApplicationResourceStorage::new(),
62        subscriptions: DashMap::new(),
63        sync_state: DashMap::new(),
64    }
65});
66
67/// Advise Runtime Extensions of a new context
68pub(crate) fn new_context(ctx: &crate::runtime_context::HermesRuntimeContext) {
69    STATE.block.add_app(ctx.app_name().clone());
70    STATE.network.add_app(ctx.app_name().clone());
71    STATE.transaction.add_app(ctx.app_name().clone());
72    STATE.subscription_id.add_app(ctx.app_name().clone());
73}

75/// Cardano Error.

83/// Global multi-threaded Tokio runtime for background tasks.
84#[allow(clippy::expect_used)]
85pub(crate) static TOKIO_RUNTIME: once_cell::sync::Lazy<Runtime> =
86    once_cell::sync::Lazy::new(|| {
87        tokio::runtime::Builder::new_multi_thread()
