13/// `ResourceStorage` struct.
14/// - `WitType` represents the type from the wit file definitions and which will appear in
15///   the `wasmtime::component::Resource<WitType>` object.
16/// - `RustType` actually the type which is bound to the `WitType` and holds all the data
17///   needed for the `WitType`.
18pub(crate) struct ResourceStorage<WitType, RustType> {
19    /// Map of id to resource object.
20    state: DashMap<u32, RustType>,
21    /// Next available address id of the resource.
22    available_address: AtomicU32,
23    /// `WitType` type phantom.
24    _phantom: std::marker::PhantomData<WitType>,
25}
26
27impl<WitType, RustType> ResourceStorage<WitType, RustType>
28where WitType: 'static
29{
30    /// Creates new `ResourceStorage` instance.
31    pub(crate) fn new() -> Self {
32        Self {
33            state: DashMap::new(),
34            available_address: AtomicU32::default(),
35            _phantom: std::marker::PhantomData,
36        }
37    }
38
39    /// Creates a new owned resource from the given object.
40    /// Stores a resources link to the original object in the resource manager.
41    pub(crate) fn create_resource(
42        &self, object: RustType,
43    ) -> wasmtime::component::Resource<WitType> {
44        let available_address = self.available_address.load(Ordering::Acquire);
45        self.state.insert(available_address, object);
46
47        // Increment the value of the available address to 1,
48        // so that it can be used for the next resource.
49        // Under the assumption that `ResourceStorage` will not handle too many resources at once,
50        // and will not hold resources for too long, saturating increment is safe.
51        self.available_address
52            .store(available_address.saturating_add(1), Ordering::Release);
53
54        wasmtime::component::Resource::new_own(available_address)
55    }
56
57    /// Creates a new owned resource from the given object.
58    /// Stores a resources link to the original object in the resource manager.
59    ///
60    /// NOTE: `&mut self` is used to enable the `rustc` borrow checker and prevent
61    /// potential deadlocking of the `DashMap`.
62    /// As `ResourceStorage` not supposed to be used in any other cases rather than as a
63    /// field of `ApplicationResourceStorage` (so no any `static` variable of this
64    /// type will be created), it is fine to add `&mut self` for this method.
65    pub(crate) fn get_object<'a>(
66        &'a mut self, resource: &wasmtime::component::Resource<WitType>,
67    ) -> wasmtime::Result<impl DerefMut<Target = RustType> + 'a> {
68        self.state
69            .get_mut(&resource.rep())
70            .ok_or(Self::resource_not_found_err())
71    }
72
73    /// Removes the resource from the resource manager.
74    /// Similar to the `drop` function, resource is releasing and consumed by this
75    /// function, thats why it is passed by value.
76    #[allow(clippy::needless_pass_by_value)]
77    pub(crate) fn delete_resource(
78        &self, resource: wasmtime::component::Resource<WitType>,
79    ) -> anyhow::Result<RustType> {
80        self.delete_resource_rep(resource.rep())
81    }
82
83    /// Removes the resource from the resource manager by its representation.
84    /// The resource is properly removed from internal storage.
85    /// Returns an error if the resource was not found.
86    pub(crate) fn delete_resource_rep(&self, rep: u32) -> anyhow::Result<RustType> {
87        self.state
88            .remove(&rep)
89            .map(|(_, v)| v)
90            .ok_or(Self::resource_not_found_err())
91    }
92
93    /// Resource not found error message.
94    fn resource_not_found_err() -> wasmtime::Error {
95        let msg = format!(
96            "Resource <{}, {}> not found, need to add resource first by calling `add_resource`",
97            type_name::<WitType>(),
98            type_name::<RustType>()
99        );
100        wasmtime::Error::msg(msg)
101    }
102}

104/// `ApplicationResourceStorage` struct.
105/// - `WitType` represents the type from the wit file definitions and which will appear in
106///   the `wasmtime::component::Resource<WitType>` object.
107/// - `RustType` actually the type which is bound to the `WitType` and holds all the data
108///   needed for the `WitType`.
109pub(crate) struct ApplicationResourceStorage<WitType, RustType> {
110    /// Map of app name to resources.
111    state: DashMap<ApplicationName, ResourceStorage<WitType, RustType>>,
112}
113
114impl<WitType, RustType> ApplicationResourceStorage<WitType, RustType>
115where WitType: 'static
116{
117    /// Creates new `ApplicationResourceStorage` instance.
118    pub(crate) fn new() -> Self {
119        Self {
120            state: DashMap::new(),
121        }
122    }
123
124    /// Adds new application to the resource manager.
125    /// If the application state already exists, do nothing.
126    pub(crate) fn add_app(&self, app_name: ApplicationName) {
127        if !self.state.contains_key(&app_name) {
128            self.state.insert(app_name, ResourceStorage::new());
129        }
130    }
131
132    /// Get application state from the resource manager.
133    /// To increase performance and reduce locking time, it's better to call
134    /// `drop(app_state)` immediately when the `app_state` is not needed anymore and don't
135    /// wait until it will be released by the compiler.
136    ///
137    /// **Locking behavior:** May deadlock if called when holding any sort of reference
138    /// into the map.
139    pub(crate) fn get_app_state<'a>(
140        &'a self, app_name: &ApplicationName,
141    ) -> anyhow::Result<impl DerefMut<Target = ResourceStorage<WitType, RustType>> + 'a> {
142        self.state
143            .get_mut(app_name)
144            .ok_or_else(|| anyhow::anyhow!(Self::app_not_found_err()))
145    }
146
147    /// Removes application and all associated resources from the resource manager.
148    #[allow(dead_code)]
149    pub(crate) fn remove_app(&self, app_name: &ApplicationName) {
150        self.state.remove(app_name);
151    }
152
153    /// Application not found error message.
154    fn app_not_found_err() -> wasmtime::Error {
155        let msg = format!(
156        "Application not found for resource <{}, {}>, need to add application first by calling `add_app`",
157        type_name::<WitType>(),
158        type_name::<RustType>()
159    );
160        wasmtime::Error::msg(msg)
161    }
162}

164#[cfg(test)]
165mod tests {
166    use super::*;
167
168    struct WitType;
169
170    #[test]
171    fn test_resource_storage() {
172        let mut resource_manager = ResourceStorage::<WitType, u32>::new();
173
174        let object = 100;
175        let resource = resource_manager.create_resource(object);
176        let copied_resource = wasmtime::component::Resource::new_borrow(resource.rep());
177
178        assert_eq!(*resource_manager.get_object(&resource).unwrap(), object);
179        assert!(resource_manager.delete_resource(resource).is_ok());
180        assert!(resource_manager.get_object(&copied_resource).is_err());
181        assert!(resource_manager.delete_resource(copied_resource).is_err());
182    }
183
184    #[test]
185    fn test_app_resource_storage() {
186        let resource_manager = ApplicationResourceStorage::<WitType, u32>::new();
187        let app_name_1 = ApplicationName("app_1".to_string());
188
189        {
190            assert!(resource_manager.get_app_state(&app_name_1).is_err());
191            resource_manager.add_app(app_name_1.clone());
192            assert!(resource_manager.get_app_state(&app_name_1).is_ok());
193            resource_manager.remove_app(&app_name_1);
194            assert!(resource_manager.get_app_state(&app_name_1).is_err());
195        }
196
197        {
198            // Check preserving app state when instantiating the same app twice
199            let object = 100;
200            resource_manager.add_app(app_name_1.clone());
201            let app_state = resource_manager.get_app_state(&app_name_1).unwrap();
202            let res = app_state.create_resource(object);
203
204            drop(app_state);
205            resource_manager.add_app(app_name_1.clone());
206            let mut app_state = resource_manager.get_app_state(&app_name_1).unwrap();
207            assert!(app_state.get_object(&res).is_ok());
208        }
209    }
