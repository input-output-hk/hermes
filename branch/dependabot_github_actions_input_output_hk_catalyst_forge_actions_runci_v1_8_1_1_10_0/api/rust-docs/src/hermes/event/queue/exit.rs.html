1//! Implementation of exit status retrieval after event queue shutdown.
2
3use std::{
4    process::ExitCode,
5    sync::{Arc, Condvar, Mutex},
6    time::Duration,
7};
8
9/// Exit status.
10#[derive(Copy, Clone, Debug, thiserror::Error)]
11pub enum Exit {
12    /// An application requested runtime abort.
13    #[error(
14        "Event queue closed: an application requested runtime abort \
15         with exit code ({exit_code:?})"
16    )]
17    Done {
18        /// Exit code provided by an application.
19        exit_code: ExitCode,
20    },
21    /// All event senders are dropped.
22    #[error("Event queue closed: all event senders are dropped")]
23    QueueClosed,
24    /// Another task panicked inside.
25    #[error("Event queue poisoned: another task panicked inside")]
26    QueuePoisoned,
27    /// Timeout elapsed.
28    #[error("Event queue closed: timeout")]
29    Timeout,
30}
31
32impl Exit {
33    /// Returns an exit code if `Self` is [`Self::Done`].
34    #[must_use]
35    pub fn get_exit_code(self) -> Option<ExitCode> {
36        if let Self::Done { exit_code } = self {
37            Some(exit_code)
38        } else {
39            None
40        }
41    }
42}
43
44impl From<ExitCode> for Exit {
45    fn from(exit_code: ExitCode) -> Self {
46        Self::Done { exit_code }
47    }
48}
49
50/// Lock to [`Exit`] that event queue sets on shutdown.
51// To satisfy the guarantees [`ExitLock`] should never implement or derive `Clone`.
52pub struct ExitLock(Arc<(Condvar, Mutex<Option<Exit>>)>);
53
54impl ExitLock {
55    /// Initializes the lock and return two clones of it.
56    pub(super) fn new_pair() -> (Self, Self) {
57        let condvar = Condvar::new();
58        // Default value doesn't matter as it's bound to be awaited to be changed.
59        let payload = Mutex::new(None);
60        let inner = Arc::new((condvar, payload));
61        (Self(inner.clone()), Self(inner))
62    }
63
64    /// Set the [`Exit`] value. This will notify the waiting thread.
65    ///
66    /// This shouldn't exposed the waiting thread.
67    pub(super) fn set(self, exit: Exit) {
68        let (condvar, payload) = &*self.0;
69        // It doesn't matter if the lock is poisoned
70        // because the condvar would catch it anyway.
71        let Ok(mut payload) = payload.lock() else {
72            return;
73        };
74        payload.get_or_insert(exit);
75        condvar.notify_one();
76    }
77
78    /// Blocks until the [`Exit`] value is set.
79    #[must_use]
80    pub fn wait(self) -> Exit {
81        let (condvar, payload) = &*self.0;
82        let Ok(exit) = payload.lock() else {
83            return Exit::QueuePoisoned;
84        };
85        condvar
86            .wait_while(exit, |opt| opt.is_none())
87            .as_deref()
88            .copied()
89            .unwrap_or(Some(Exit::QueuePoisoned))
90            // `None` is guaranteed to be unreachable since we are waiting for `Some` on condvar.
91            .unwrap_or(Exit::QueuePoisoned)
92    }
93
94    /// Blocks until either the [`Exit`] value is set or the timeout elapses.
95    #[must_use]
96    pub fn wait_timeout(self, dur: Duration) -> Exit {
97        let (condvar, payload) = &*self.0;
98        let Ok(exit) = payload.lock() else {
99            return Exit::QueuePoisoned;
100        };
101        condvar
102            .wait_timeout_while(exit, dur, |opt| opt.is_none())
103            .map(|(exit, timeout)| {
104                if timeout.timed_out() {
105                    Some(Exit::Timeout)
106                } else {
107                    *exit
108                }
109            })
110            .unwrap_or(Some(Exit::QueuePoisoned))
111            // `None` is guaranteed to be unreachable since we are waiting for `Some` on condvar.
112            .unwrap_or(Exit::QueuePoisoned)
113    }
114}
