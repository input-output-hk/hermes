1//! A signature payload object for author.cose Hermes WASM module package.
2//! Defined at `https://input-output-hk.github.io/hermes/architecture/08_concepts/hermes_packaging_requirements/wasm_modules/#wasm-component-module-signatures`.
3
4use super::super::{
5    hash::Blake2b256, schema_validation::SchemaValidator, sign::signature::SignaturePayloadEncoding,
6};
7
8/// A signature payload object.
9#[derive(Clone, Debug, PartialEq, Eq)]
10pub(crate) struct SignaturePayload {
11    /// Hash of the metadata JSON file.
12    metadata: Blake2b256,
13    /// Hash of the WASM component file.
14    component: Blake2b256,
15    /// Config instance.
16    config: Option<SignaturePayloadConfig>,
17    /// Settings instance.
18    settings: Option<SignaturePayloadSettings>,
19    /// Hash of the share directory content.
20    share: Option<Blake2b256>,
21}
22
23/// A `SignaturePayload` config object.
24#[derive(Clone, Debug, PartialEq, Eq)]
25struct SignaturePayloadConfig {
26    /// Hash of the config JSON file.
27    file: Option<Blake2b256>,
28    /// Hash of the config schema JSON file.
29    schema: Blake2b256,
30}
31
32/// A `SignaturePayload` settings object.
33#[derive(Clone, Debug, PartialEq, Eq)]
34struct SignaturePayloadSettings {
35    /// Hash of the settings schema JSON file.
36    schema: Blake2b256,
37}
38
39/// `SignaturePayload` builder object.
40pub(crate) struct SignaturePayloadBuilder {
41    /// Hash of the metadata JSON file.
42    metadata: Blake2b256,
43    /// Hash of the WASM component file.
44    component: Blake2b256,
45    /// Hash of the config JSON file.
46    config_file: Option<Blake2b256>,
47    /// Hash of the config schema JSON file.
48    config_schema: Option<Blake2b256>,
49    /// Hash of the settings schema JSON file.
50    settings_schema: Option<Blake2b256>,
51    /// Hash of the share directory content.
52    share: Option<Blake2b256>,
53}
54
55impl SignaturePayloadBuilder {
56    /// Create a new `SignaturePayloadBuilder`.
57    pub(crate) fn new(metadata: Blake2b256, component: Blake2b256) -> Self {
58        Self {
59            metadata,
60            component,
61            config_file: None,
62            config_schema: None,
63            settings_schema: None,
64            share: None,
65        }
66    }
67
68    /// Set the config file hash.
69    pub(crate) fn with_config_file(&mut self, file: Blake2b256) {
70        self.config_file = Some(file);
71    }
72
73    /// Set the config schema hash.
74    pub(crate) fn with_config_schema(&mut self, schema: Blake2b256) {
75        self.config_schema = Some(schema);
76    }
77
78    /// Set the settings schema hash.
79    pub(crate) fn with_settings_schema(&mut self, schema: Blake2b256) {
80        self.settings_schema = Some(schema);
81    }
82
83    /// Set the share directory hash.
84    pub(crate) fn with_share(&mut self, share: Blake2b256) {
85        self.share = Some(share);
86    }
87
88    /// Create a new `SignaturePayload`.
89    pub(crate) fn build(self) -> SignaturePayload {
90        SignaturePayload {
91            metadata: self.metadata,
92            component: self.component,
93            config: self.config_schema.map(|schema| {
94                SignaturePayloadConfig {
95                    file: self.config_file,
96                    schema,
97                }
98            }),
99            settings: self
100                .settings_schema
101                .map(|schema| SignaturePayloadSettings { schema }),
102            share: self.share,
103        }
104    }
105}
106
107/// WASM module cose signature payload JSON schema.
108const SIGNATURE_PAYLOAD_SCHEMA: &str =
109    include_str!("../../../../schemas/hermes_module_cose_author_payload.schema.json");
110
111impl SignaturePayloadEncoding for SignaturePayload {
112    fn to_json(&self) -> serde_json::Value {
113        let mut json = serde_json::Map::new();
114        json.insert("metadata".to_string(), self.metadata.to_hex().into());
115        json.insert("component".to_string(), self.component.to_hex().into());
116        if let Some(config) = &self.config {
117            let mut config_json = serde_json::Map::new();
118
119            config_json.insert("schema".to_string(), config.schema.to_hex().into());
120            if let Some(file) = &config.file {
121                config_json.insert("file".to_string(), file.to_hex().into());
122            }
123
124            json.insert("config".to_string(), config_json.into());
125        }
126        if let Some(settings) = &self.settings {
127            json.insert(
128                "settings".to_string(),
129                serde_json::json!({
130                    "schema": settings.schema.to_hex()
131                }),
132            );
133        }
134        if let Some(share) = &self.share {
135            json.insert("share".to_string(), share.to_hex().into());
136        }
137
138        json.into()
139    }
140
141    fn from_json(json: serde_json::Value) -> anyhow::Result<Self>
142    where Self: Sized {
143        let schema_validator = SchemaValidator::from_str(SIGNATURE_PAYLOAD_SCHEMA)?;
144        schema_validator.validate(&json)?;
145
146        let json = json
147            .as_object()
148            .ok_or_else(|| anyhow::anyhow!("Signature payload JSON is not an object"))?;
149
150        let metadata = json
151            .get("metadata")
152            .ok_or_else(|| anyhow::anyhow!("Missing metadata field"))?
153            .as_str()
154            .ok_or_else(|| anyhow::anyhow!("Invalid metadata field"))
155            .map(Blake2b256::from_hex)??;
156
157        let component = json
158            .get("component")
159            .ok_or_else(|| anyhow::anyhow!("Missing component field"))?
160            .as_str()
161            .ok_or_else(|| anyhow::anyhow!("Invalid component field"))
162            .map(Blake2b256::from_hex)??;
163
164        let config = json
165            .get("config")
166            .and_then(|val| val.as_object())
167            .map(|config| -> anyhow::Result<_> {
168                let file = config
169                    .get("file")
170                    .map(|file| {
171                        file.as_str()
172                            .ok_or_else(|| anyhow::anyhow!("Invalid file field"))
173                            .map(Blake2b256::from_hex)?
174                    })
175                    .transpose()?;
176                let schema = config
177                    .get("schema")
178                    .ok_or_else(|| anyhow::anyhow!("Missing schema field"))?
179                    .as_str()
180                    .ok_or_else(|| anyhow::anyhow!("Invalid schema field"))
181                    .map(Blake2b256::from_hex)??;
182                Ok(SignaturePayloadConfig { file, schema })
183            })
184            .transpose()?;
185
186        let settings = json
187            .get("settings")
188            .and_then(|val| val.as_object())
189            .map(|settings| -> anyhow::Result<_> {
190                let schema = settings
191                    .get("schema")
192                    .ok_or_else(|| anyhow::anyhow!("Missing schema field"))?
193                    .as_str()
194                    .ok_or_else(|| anyhow::anyhow!("Invalid schema field"))
195                    .map(Blake2b256::from_hex)??;
196                Ok(SignaturePayloadSettings { schema })
197            })
198            .transpose()?;
199
200        let share = json
201            .get("share")
202            .and_then(|val| val.as_str())
203            .map(Blake2b256::from_hex)
204            .transpose()?;
205
206        Ok(SignaturePayload {
207            metadata,
208            component,
209            config,
210            settings,
211            share,
212        })
213    }
214}

216#[cfg(test)]
217mod tests {
218    use super::*;
219
220    #[test]
221    fn signature_payload_encoding_test() {
222        let hash = Blake2b256::hash(b"test");
223        let schema_validator = SchemaValidator::from_str(SIGNATURE_PAYLOAD_SCHEMA).unwrap();
224
225        {
226            let payload_builder = SignaturePayloadBuilder::new(hash.clone(), hash.clone());
227            let payload = payload_builder.build();
228
229            let json = payload.to_json();
230            schema_validator.validate(&json).unwrap();
231
232            let expected_json = serde_json::json!({
233                "metadata": hash.to_hex(),
234                "component": hash.to_hex(),
235            });
236            assert_eq!(json, expected_json);
237
238            let payload = SignaturePayload::from_json(json).unwrap();
239            assert_eq!(payload, payload);
240        }
241
242        {
243            let mut payload_builder = SignaturePayloadBuilder::new(hash.clone(), hash.clone());
244            payload_builder.with_config_file(hash.clone());
245            payload_builder.with_config_schema(hash.clone());
246            payload_builder.with_settings_schema(hash.clone());
247            payload_builder.with_share(hash.clone());
248            let payload = payload_builder.build();
249
250            let json = payload.to_json();
251            schema_validator.validate(&json).unwrap();
252
253            let expected_json = serde_json::json!({
254                "metadata": hash.to_hex(),
255                "component": hash.to_hex(),
256                "config": {
257                    "file": hash.to_hex(),
258                    "schema": hash.to_hex(),
259                },
260                "settings": {
261                    "schema": hash.to_hex(),
262                },
263                "share": hash.to_hex(),
264            });
265            assert_eq!(json, expected_json);
266
267            let payload = SignaturePayload::from_json(json).unwrap();
268            assert_eq!(payload, payload);
269        }
270    }
