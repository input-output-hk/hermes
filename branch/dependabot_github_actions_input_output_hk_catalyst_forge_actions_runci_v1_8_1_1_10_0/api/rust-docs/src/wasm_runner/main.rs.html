27/// An example Wasm Runner
28///
29/// Loads each wasm component from the arguments as a standalone Hermes application
30/// consisting of a single module with a temporary VFS.
31///
32/// This allows to run Wasm components without properly packaging them.
33///
34/// Returns an exit code that can be inspected for custom values issued by Wasm
35/// components.
36#[derive(Debug, clap::Parser)]
37pub struct Arguments {
38    /// Wasm components to load as apps in this example.
39    components: Vec<PathBuf>,
40}

42/// Init the logger
43#[allow(dead_code)]
44fn init_logger() -> Result<(), SetGlobalDefaultError> {
45    let subscriber = FmtSubscriber::builder()
46        .json()
47        .with_level(true)
48        .with_thread_names(true)
49        .with_thread_ids(true)
50        .with_file(true)
51        .with_line_number(true)
52        .with_timer(time::UtcTime::rfc_3339())
53        .with_max_level(LevelFilter::INFO)
54        .finish();
55
56    tracing::subscriber::set_global_default(subscriber)
57}

59/// Initialize the IPFS node
60fn init_ipfs(temp_dir: &TempDir) -> anyhow::Result<()> {
61    let ipfs_dir = create_temp_dir_child(temp_dir, Path::new("ipfs"))?;
62    // disable bootstrapping the IPFS node to default addresses for testing
63    let default_bootstrap = false;
64    hermes::ipfs::bootstrap(ipfs_dir.as_path(), default_bootstrap)
65}

67/// Get the timeout value from env.
68fn get_timeout_ms() -> anyhow::Result<u64> {
69    env::var(ENV_TIMEOUT_MS).map_or(Ok(DEFAULT_ENV_TIMEOUT_MS), |s| {
70        s.parse().map_err(anyhow::Error::from)
71    })
72}

74fn main() -> ExitCode {
75    let internal_failure = ExitCode::from(101);
76    main_internal()
77        .inspect_err(|err| {
78            // Use `tracing::error!` once logger initialization is fixed.
79            eprintln!("Failed to run applications: {err}");
80        })
81        .map_or(internal_failure, |exit| {
82            exit.unwrap_exit_code_or(internal_failure)
83        })
84}
85
86/// "Regular" main function returning a [`Result`] that is mapped to an [`ExitCode`] in
87/// [`main()`]
88fn main_internal() -> anyhow::Result<Exit> {
89    // This is necessary otherwise the logging functions inside hermes are silent during the
90    // test run.
91    // init_logger()?;
92    // This causes issues with normal test runs, so comment out for now.
93    // info!("Starting Hermes WASM integration tests");
94
95    let timeout_ms = get_timeout_ms()?;
96    let args = Arguments::try_parse()?;
97    let temp_dir = TempDir::new()?;
98
99    let apps = collect_apps(&args, &temp_dir)?;
100    ensure!(!apps.is_empty(), "At least one app is required to run");
101
102    init_ipfs(&temp_dir)?;
103    let exit_lock = reactor::init()?;
104
105    for app in apps {
106        reactor::load_app(app)?;
107    }
108
109    let exit = exit_lock.wait_timeout(Duration::from_millis(timeout_ms));
110
111    Ok(exit)
112}

114/// Collects `.wasm` files in the current directory or sub-directories of the current
115/// directory. Return a [`String`] module name along with each compiled [`Module`].
116fn collect_modules(args: &Arguments) -> anyhow::Result<Vec<(String, Module)>> {
117    // All wasm components in a directory.
118    let mut modules = Vec::new();
119
120    // Collect component files
121    for file_path in &args.components {
122        let name = file_path
123            .file_name()
124            .and_then(OsStr::to_str)
125            .ok_or_else(|| anyhow!("Provided path is invalid: {}", file_path.display()))?
126            .to_string();
127        let wasm_buf = fs::read(file_path)?;
128        let module = Module::from_bytes(&wasm_buf)?;
129        modules.push((name, module));
130    }
131
132    Ok(modules)
133}

135/// Create one-module application with temp directory VFS.
136fn create_one_module_app(
137    name: String, vfs_dir_path: &Path, module: Module,
138) -> anyhow::Result<Application> {
139    let vfs_name = [name.as_str(), "_vfs"].concat();
140    let vfs = VfsBootstrapper::new(vfs_dir_path, vfs_name).bootstrap()?;
141    let app = Application::new(name, vfs, vec![module]);
142
143    Ok(app)
144}
145
146/// Create a temp subdirectory.
147fn create_temp_dir_child(temp_dir: &TempDir, child_path: &Path) -> anyhow::Result<PathBuf> {
148    let child_absolute_path = temp_dir.path().join(child_path);
149    fs::create_dir_all(child_absolute_path.as_path())?;
150    Ok(child_absolute_path)
151}
152
153/// Collects `.wasm` files in the current directory or sub-directories of the current
154/// directory. Then creates one-module applications out of each of them.
155fn collect_apps(args: &Arguments, temp_dir: &TempDir) -> anyhow::Result<Vec<Application>> {
156    let modules = collect_modules(args)?;
157    let mut apps = Vec::with_capacity(modules.len());
158    for (module_name, module) in modules {
159        let vfs_dir_path =
160            create_temp_dir_child(temp_dir, Path::new("vfs").join(&module_name).as_path())?;
161        let app = create_one_module_app(module_name, vfs_dir_path.as_path(), module)?;
162        apps.push(app);
163    }
164    Ok(apps)
165}
