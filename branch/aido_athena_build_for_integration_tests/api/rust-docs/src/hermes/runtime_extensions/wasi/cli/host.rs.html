44impl cli::exit::Host for HermesRuntimeContext {
45    fn exit(
46        &mut self,
47        status: Result<(), ()>,
48    ) -> wasmtime::Result<()> {
49        warn!("Exiting an application is not supported");
50        status.map_err(|()| anyhow::anyhow!(""))
51    }
52
53    /// Exit the current instance and any linked instances, reporting the
54    /// specified status code to the host.
55    ///
56    /// The meaning of the code depends on the context, with 0 usually meaning
57    /// "success", and other values indicating various types of failure.
58    ///
59    /// This function does not return; the effect is analogous to a trap, but
60    /// without the connotation that something bad has happened.
61    ///
62    /// # Warning
63    ///  
64    /// Unstable WASI feature! Should never be linked.
65    /// See details of how to handle unstable features
66    /// [here](https://github.com/bytecodealliance/wasmtime/issues/8645).
67    fn exit_with_code(
68        &mut self,
69        status: u8,
70    ) -> wasmtime::Result<()> {
71        self.exit(if status == 0 { Ok(()) } else { Err(()) })
72    }
73}

75impl cli::stdin::Host for HermesRuntimeContext {
76    fn get_stdin(&mut self) -> wasmtime::Result<wasmtime::component::Resource<InputStream>> {
77        warn!("Stdin is not supported");
78        let app_state = get_input_streams_state().get_app_state(self.app_name())?;
79        Ok(app_state.create_resource(Box::new(std::io::empty())))
80    }
81}
