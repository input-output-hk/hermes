1//! `SQLite` host implementation for WASM runtime.
2
3use crate::{
4    runtime_context::HermesRuntimeContext,
5    runtime_extensions::{
6        bindings::hermes::sqlite::api::{Errno, Host, Sqlite},
7        hermes::sqlite::{
8            connection::core::close,
9            kernel,
10            state::{connection::DbHandle, resource_manager},
11        },
12    },
13};
14
15impl Host for HermesRuntimeContext {
16    /// Opens a connection to a new or existing `SQLite` database.
17    ///
18    /// ## Parameters
19    ///
20    /// - `readonly`: If set to true, the database is opened in read-only mode. An error
21    ///   is returned if the database doesn't already exist.
22    /// - `memory`: If set to true, the database will be opened as an in-memory database.
23    ///
24    /// ## Returns
25    ///
26    /// If the database is opened (and/or created) successfully, then the `sqlite3` object
27    /// is returned. Otherwise an error code is returned.
28    fn open(
29        &mut self,
30        readonly: bool,
31        memory: bool,
32    ) -> wasmtime::Result<Result<wasmtime::component::Resource<Sqlite>, Errno>> {
33        let db_handle = DbHandle::from_readonly_and_memory(readonly, memory);
34
35        if let Some(resource) =
36            resource_manager::try_get_connection_resource(self.app_name(), db_handle)
37        {
38            return Ok(Ok(resource));
39        }
40
41        match kernel::open(readonly, memory, self.app_name().clone()) {
42            Ok(db_ptr) => {
43                match resource_manager::create_connection_resource(
44                    self.app_name(),
45                    db_handle,
46                    db_ptr as _,
47                ) {
48                    Ok(resource) => Ok(Ok(resource)),
49                    Err(err) => {
50                        if let Err(errno) = close(db_ptr) {
51                            anyhow::bail!(
52                            "failed to create connection resource: {err}, also failed to close the connection with errno: {errno}..."
53                        )
54                        }
55                        anyhow::bail!(
56                            "failed to create connection resource: {err}, closing the connection..."
57                        )
58                    },
59                }
60            },
61            Err(err) => Ok(Err(err)),
62        }
63    }
64}
