3use cardano_chain_follower::{turbo_downloader::DlConfig, ChainSyncConfig};
4
5use crate::runtime_extensions::hermes::cardano::{STATE, TOKIO_RUNTIME};
6
7/// Spawns a chain sync task for a given network
8pub(crate) fn spawn_chain_sync_task(chain: cardano_blockchain_types::Network) {
9    tracing::info!(chain = %chain, "Spawning chain sync task");
10    let dl_config = DlConfig::default();
11    let mut sync_cfg = ChainSyncConfig::default_for(chain);
12    sync_cfg.mithril_cfg = sync_cfg.mithril_cfg.with_dl_config(dl_config);
13
14    // If chain sync for a specific chain already exists, do not spawn a new one
15    if !STATE.sync_state.contains_key(&chain) {
16        let handle = TOKIO_RUNTIME.handle();
17        let join_handle = handle.spawn(async move {
18            if let Err(error) = sync_cfg.run().await {
19                tracing::error!(chain = %chain, error = %error, "Chain sync failed");
20            }
21        });
22        STATE.sync_state.insert(chain, join_handle);
23    }
24}
