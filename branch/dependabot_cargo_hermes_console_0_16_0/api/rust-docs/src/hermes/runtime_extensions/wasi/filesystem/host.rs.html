26    /// Return a stream for reading from a file, if available.
27    ///
28    /// May fail with an error-code describing why the file cannot be read.
29    ///
30    /// Multiple read, write, and append streams may be active on the same open
31    /// file and they do not interfere with each other.
32    ///
33    /// Note: This allows using `read-stream`, which is similar to `read` in POSIX.
34    fn read_via_stream(
35        &mut self, res: wasmtime::component::Resource<WasiDescriptor>, offset: Filesize,
36    ) -> wasmtime::Result<Result<wasmtime::component::Resource<InputStream>, ErrorCode>> {
37        let mut fs_app_state = get_state().get_app_state(self.app_name())?;
38        let Ok(descriptor) = fs_app_state.get_object(&res) else {
39            return Ok(Err(ErrorCode::BadDescriptor));
40        };
41        let mut file = match &*descriptor {
42            Descriptor::File(f) => f.clone(),
43            Descriptor::Dir(_) => return Ok(Err(ErrorCode::IsDirectory)),
44        };
45        file.seek(SeekFrom::Start(offset))?;
46
47        let input_streams_app_state = get_input_streams_state().get_app_state(self.app_name())?;
48        Ok(Ok(input_streams_app_state.create_resource(Box::new(file))))
49    }

250    /// Return the attributes of an open file or directory.
251    ///
252    /// Note: This is similar to `fstat` in POSIX, except that it does not return
253    /// device and inode information. For testing whether two descriptors refer to
254    /// the same underlying filesystem object, use `is-same-object`. To obtain
255    /// additional data that can be used do determine whether a file has been
256    /// modified, use `metadata-hash`.
257    ///
258    /// Note: This was called `fd_filestat_get` in earlier versions of WASI.
259    fn stat(
260        &mut self, res: wasmtime::component::Resource<WasiDescriptor>,
261    ) -> wasmtime::Result<Result<DescriptorStat, ErrorCode>> {
262        let mut app_state = get_state().get_app_state(self.app_name())?;
263        let Ok(descriptor) = app_state.get_object(&res) else {
264            return Ok(Err(ErrorCode::BadDescriptor));
265        };
266
267        let f = match &*descriptor {
268            Descriptor::File(f) => f,
269            Descriptor::Dir(_) => todo!(),
270        };
271
272        let Ok(size) = f
273            .size()
274            .and_then(|size| TryInto::<u64>::try_into(size).map_err(|e| anyhow::anyhow!(e)))
275        else {
276            return Ok(Err(ErrorCode::Io));
277        };
278
279        Ok(Ok(DescriptorStat {
280            type_: DescriptorType::RegularFile,
281            link_count: 0,
282            size,
283            data_access_timestamp: None,
284            data_modification_timestamp: None,
285            status_change_timestamp: None,
286        }))
287    }

328    /// Open a file or directory.
329    ///
330    /// The returned descriptor is not guaranteed to be the lowest-numbered
331    /// descriptor not currently open/ it is randomized to prevent applications
332    /// from depending on making assumptions about indexes, since this is
333    /// error-prone in multi-threaded contexts. The returned descriptor is
334    /// guaranteed to be less than 2**31.
335    ///
336    /// If `flags` contains `descriptor-flags::mutate-directory`, and the base
337    /// descriptor doesn't have `descriptor-flags::mutate-directory` set,
338    /// `open-at` fails with `error-code::read-only`.
339    ///
340    /// If `flags` contains `write` or `mutate-directory`, or `open-flags`
341    /// contains `truncate` or `create`, and the base descriptor doesn't have
342    /// `descriptor-flags::mutate-directory` set, `open-at` fails with
343    /// `error-code::read-only`.
344    ///
345    /// Note: This is similar to `openat` in POSIX.
346    fn open_at(
347        &mut self, res: wasmtime::component::Resource<WasiDescriptor>, _path_flags: PathFlags,
348        path: String, open_flags: OpenFlags, _flags: DescriptorFlags,
349    ) -> wasmtime::Result<Result<wasmtime::component::Resource<WasiDescriptor>, ErrorCode>> {
350        let mut app_state = get_state().get_app_state(self.app_name())?;
351        let Ok(descriptor) = app_state.get_object(&res) else {
352            return Ok(Err(ErrorCode::BadDescriptor));
353        };
354        let dir = match &*descriptor {
355            Descriptor::Dir(dir) => dir,
356            Descriptor::File(_) => return Ok(Err(ErrorCode::NotDirectory)),
357        };
358
359        let create = open_flags.contains(OpenFlags::CREATE);
360        let exclusive = open_flags.contains(OpenFlags::EXCLUSIVE);
361        let f = match dir.get_file(Path::from_str(&path)) {
362            Ok(f) => {
363                if create && exclusive {
364                    return Ok(Err(ErrorCode::Exist));
365                }
366
367                f
368            },
369            Err(_) => {
370                if create {
371                    if let Ok(f) = dir.create_file(Path::from_str(&path)) {
372                        f
373                    } else {
374                        return Ok(Err(ErrorCode::Io));
375                    }
376                } else {
377                    return Ok(Err(ErrorCode::NoEntry));
378                }
379            },
380        };
381
382        let f = if open_flags.contains(OpenFlags::TRUNCATE) {
383            if dir.remove_file(Path::from_str(&path)).is_err() {
384                return Ok(Err(ErrorCode::Io));
385            }
386
387            match dir.create_file(Path::from_str(&path)) {
388                Ok(f) => f,
389                Err(_) => return Ok(Err(ErrorCode::Io)),
390            }
391        } else {
392            f
393        };
394        drop(descriptor);
395        Ok(Ok(app_state.create_resource(Descriptor::File(f))))
396    }

445    /// Unlink a filesystem object that is not a directory.
446    ///
447    /// Return `error-code::is-directory` if the path refers to a directory.
448    /// Note: This is similar to `unlinkat(fd, path, 0)` in POSIX.
449    fn unlink_file_at(
450        &mut self, res: wasmtime::component::Resource<WasiDescriptor>, path: String,
451    ) -> wasmtime::Result<Result<(), ErrorCode>> {
452        let mut app_state = get_state().get_app_state(self.app_name())?;
453        let Ok(descriptor) = app_state.get_object(&res) else {
454            return Ok(Err(ErrorCode::BadDescriptor));
455        };
456
457        match &*descriptor {
458            Descriptor::Dir(dir) => {
459                let path: Path = path.into();
460
461                if dir.get_file(path.clone()).is_err() {
462                    return Ok(Err(ErrorCode::NoEntry));
463                }
464
465                if dir.remove_file(path).is_err() {
466                    Ok(Err(ErrorCode::Io))
467                } else {
468                    Ok(Ok(()))
469                }
470            },
471            Descriptor::File(_) => Ok(Err(ErrorCode::NotDirectory)),
472        }
473    }
