1//! Wall clock host implementation for WASM runtime.
2
3use crate::{
4    runtime_context::HermesRuntimeContext,
5    runtime_extensions::{
6        bindings::wasi::clocks::wall_clock::{Datetime, Host},
7        wasi::clocks::state::{wall_clock_now, wall_clock_res},
8    },
9};
10
11impl Host for HermesRuntimeContext {
12    /// Read the current value of the clock.
13    ///
14    /// This clock is not monotonic, therefore calling this function repeatedly
15    /// will not necessarily produce a sequence of non-decreasing values.
16    ///
17    /// The returned timestamps represent the number of seconds since
18    /// 1970-01-01T00:00:00Z, also known as [POSIX\'s Seconds Since the Epoch],
19    /// also known as [Unix Time].
20    ///
21    /// The nanoseconds field of the output is always less than 1000000000.
22    ///
23    /// [POSIX\'s Seconds Since the Epoch]: https://pubs.opengroup.org/onlinepubs/9699919799/xrat/V4_xbd_chap04.html#tag_21_04_16
24    /// [Unix Time]: https://en.wikipedia.org/wiki/Unix_time
25    fn now(&mut self) -> wasmtime::Result<Datetime> {
26        wall_clock_now()
27    }
28
29    /// Query the resolution of the clock.
30    ///
31    /// The nanoseconds field of the output is always less than 1000000000.
32    fn resolution(&mut self) -> wasmtime::Result<Datetime> {
33        Ok(wall_clock_res())
34    }
35}
