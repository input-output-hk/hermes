18/// Crate error type.
19#[derive(Debug, Error)]
20pub enum Error {
21    /// Data encoding/decoding error.
22    #[error("Codec error: {0:?}")]
23    Codec(pallas::ledger::traverse::Error),
24    /// Client connection error.
25    #[error("Client error: {0:?}")]
26    Client(pallas::network::facades::Error),
27    /// Blockfetch protocol error.
28    #[error("Blockfetch error: {0:?}")]
29    Blockfetch(pallas::network::miniprotocols::blockfetch::ClientError),
30    /// Chainsync protocol error.
31    #[error("Chainsync error: {0:?}")]
32    Chainsync(chainsync::ClientError),
33    /// Follower failed to set its read pointer.
34    #[error("Failed to set follower read pointer")]
35    SetReadPointer,
36    /// Follower background follow task has stopped.
37    #[error("Follower follow task is not running")]
38    FollowTaskNotRunning,
39    /// Mithril snapshot error.
40    #[error("Failed to read block(s) from Mithril snapshot")]
41    MithrilSnapshot,
42    /// Failed to parse
43    #[error("Failed to parse network")]
44    ParseNetwork,
45    /// Internal Error
46    #[error("Internal error")]
47    InternalError,
48}

68/// CBOR encoded data of a multi-era block.
69#[derive(Clone, PartialEq, Eq, Hash)]
70pub struct MultiEraBlockData(Vec<u8>);
71
72impl MultiEraBlockData {
73    /// Decodes the data into a multi-era block.
74    ///
75    /// # Errors
76    ///
77    /// Returns Err if the block's era couldn't be decided or if the encoded data is
78    /// invalid.
79    pub fn decode(&self) -> Result<MultiEraBlock<'_>> {
80        let block = MultiEraBlock::decode(&self.0).map_err(Error::Codec)?;
81
82        Ok(block)
83    }
84
85    /// Consumes the [`MultiEraBlockData`] returning the block data raw bytes.
86    #[must_use]
87    pub fn into_raw_data(self) -> Vec<u8> {
88        self.0
89    }
90}
91
92impl AsRef<[u8]> for MultiEraBlockData {
93    fn as_ref(&self) -> &[u8] {
94        &self.0
95    }
96}

98/// Enum of possible Cardano networks.
99#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
100pub enum Network {
101    /// Cardano mainnet network.
102    Mainnet,
103    /// Cardano pre-production network.
104    Preprod,
105    /// Cardano preview network.
106    Preview,
107    /// Cardano testnet network.
108    Testnet,
109}
110
111/// The human readable name of the Cardano mainnet network.
112const MAINNET_NAME: &str = "mainnet";
113/// The human readable name of the Cardano pre-production network.
114const PREPROD_NAME: &str = "preprod";
115/// The human readable name of the Cardano preview network.
116const PREVIEW_NAME: &str = "preview";
117/// The human readable name of a Cardano local testnet network.
118const TESTNET_NAME: &str = "testnet";
119
120impl FromStr for Network {
121    type Err = Error;
122
123    fn from_str(input: &str) -> std::result::Result<Network, Self::Err> {
124        match input {
125            MAINNET_NAME => Ok(Network::Mainnet),
126            PREPROD_NAME => Ok(Network::Preprod),
127            PREVIEW_NAME => Ok(Network::Preview),
128            TESTNET_NAME => Ok(Network::Testnet),
129            _ => Err(Error::ParseNetwork),
130        }
131    }
132}
133
134impl std::fmt::Display for Network {
135    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
136        match self {
137            Network::Mainnet => write!(f, "{MAINNET_NAME}"),
138            Network::Preprod => write!(f, "{PREPROD_NAME}"),
139            Network::Preview => write!(f, "{PREVIEW_NAME}"),
140            Network::Testnet => write!(f, "{TESTNET_NAME}"),
141        }
142    }
143}
144
145impl From<Network> for u64 {
146    fn from(network: Network) -> Self {
147        match network {
148            Network::Mainnet => MAINNET_MAGIC,
149            Network::Preprod => PRE_PRODUCTION_MAGIC,
150            Network::Preview => PREVIEW_MAGIC,
151            Network::Testnet => TESTNET_MAGIC,
152        }
153    }
154}

156/// Return genesis values for given network
157#[must_use]
158pub fn network_genesis_values(network: &Network) -> Option<GenesisValues> {
159    match network {
160        Network::Mainnet => GenesisValues::from_magic(MAINNET_MAGIC),
161        Network::Preprod => GenesisValues::from_magic(PRE_PRODUCTION_MAGIC),
162        Network::Preview => GenesisValues::from_magic(PREVIEW_MAGIC),
163        Network::Testnet => GenesisValues::from_magic(TESTNET_MAGIC),
164    }
165}
