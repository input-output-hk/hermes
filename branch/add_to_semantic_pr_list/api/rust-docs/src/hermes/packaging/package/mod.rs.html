1//! Implementation of the generalized Hermes package struct as a HDF5 package.
2
3use std::{collections::BTreeMap, io::Read, ops::Deref};
4
5use super::hash::{Blake2b256, Blake2b256Hasher};
6use crate::hdf5::{Dir, File, Path};
7
8/// Hermes package object.
9/// Wrapper over HDF5 file object.
10pub(crate) struct Package(Dir);
11
12impl Deref for Package {
13    type Target = Dir;
14
15    fn deref(&self) -> &Self::Target {
16        &self.0
17    }
18}
19
20impl Package {
21    /// Create a new `Package` instance mounting a `dir` as a root for this package.
22    pub(crate) fn mount(dir: Dir) -> Self {
23        Self(dir)
24    }
25
26    /// Create new `Package` instance from path.
27    pub(crate) fn create<P: AsRef<std::path::Path>>(path: P) -> anyhow::Result<Self> {
28        let hdf5_file = hdf5::File::create(&path).map_err(|_| {
29            anyhow::anyhow!(
30                "Failed to create package. Package at {0} could be already exists.",
31                path.as_ref().display()
32            )
33        })?;
34        Ok(Self(Dir::new(hdf5_file.as_group()?)))
35    }
36
37    /// Open existing `Package` instance from path.
38    pub(crate) fn open<P: AsRef<std::path::Path>>(path: P) -> anyhow::Result<Self> {
39        let hdf5_file = hdf5::File::open_rw(&path).map_err(|_| {
40            anyhow::anyhow!(
41                "Failed to open package. Package at {0} could not be found.",
42                path.as_ref().display()
43            )
44        })?;
45        Ok(Self(Dir::new(hdf5_file.as_group()?)))
46    }
47
48    /// Calculates file hash, if present
49    pub(crate) fn calculate_file_hash(&self, path: Path) -> anyhow::Result<Option<Blake2b256>> {
50        let mut hasher = Blake2b256Hasher::new();
51        let Ok(mut file) = self.0.get_file(path) else {
52            return Ok(None);
53        };
54        calculate_file_hash(&mut file, &mut hasher)?;
55        Ok(Some(hasher.finalize()))
56    }
57
58    /// Calculates recursively package directory contents hash including file contents and
59    /// file names.
60    pub(crate) fn calculate_dir_hash(&self, path: &Path) -> anyhow::Result<Option<Blake2b256>> {
61        let mut hasher = Blake2b256Hasher::new();
62        let Ok(dir) = self.0.get_dir(path) else {
63            return Ok(None);
64        };
65        calculate_dir_hash(&dir, &mut hasher)?;
66        Ok(Some(hasher.finalize()))
67    }
68}
69
70/// Buffer size for hash calculation.
71/// 1024 * 1024 = 1MB.
72const BUFFER_SIZE: usize = 1024 * 1024;
73
74/// Calculates file hash with the provided hasher.
75#[allow(clippy::indexing_slicing)]
76fn calculate_file_hash(file: &mut File, hasher: &mut Blake2b256Hasher) -> anyhow::Result<()> {
77    let mut buf = vec![0; BUFFER_SIZE];
78
79    loop {
80        let len = file.read(&mut buf)?;
81        if len == 0 {
82            break;
83        }
84        hasher.update(&buf[..len]);
85    }
86
87    Ok(())
88}
89
90/// Calculates recursively directory contents hash with the provided hasher
91/// including file contents.
92fn calculate_dir_hash(dir: &Dir, hasher: &mut Blake2b256Hasher) -> anyhow::Result<()> {
93    // order all package directory content by names
94    // to have consistent hash result not depending on the order.
95    let files: BTreeMap<_, _> = dir
96        .get_files(&Path::default())?
97        .into_iter()
98        .map(|file| (file.name(), file))
99        .collect();
100    let dirs: BTreeMap<_, _> = dir
101        .get_dirs(&Path::default())?
102        .into_iter()
103        .map(|dir| (dir.name(), dir))
104        .collect();
105    for (path_str, mut file) in files {
106        hasher.update(path_str.as_bytes());
107        calculate_file_hash(&mut file, hasher)?;
108    }
109    for (path_str, dir) in dirs {
110        hasher.update(path_str.as_bytes());
111        calculate_dir_hash(&dir, hasher)?;
112    }
113
114    Ok(())
115}

117#[cfg(test)]
118mod tests {
119    use temp_dir::TempDir;
120
121    use super::*;
122    use crate::hdf5::resources::FsResource;
123
124    #[test]
125    fn calculate_file_hash_test() {
126        let tmp_dir = TempDir::new().unwrap();
127        let file_content = "test".as_bytes();
128
129        let package_name = tmp_dir.child("test.hdf5");
130        let package = Package::create(package_name).unwrap();
131
132        let file_1_name = "file_1";
133        let file_1 = tmp_dir.child(file_1_name);
134        std::fs::write(&file_1, file_content).unwrap();
135
136        package
137            .copy_resource_file(&FsResource::new(file_1), file_1_name.into())
138            .unwrap();
139
140        let hash = package
141            .calculate_file_hash(file_1_name.into())
142            .unwrap()
143            .unwrap();
144
145        let expected_hash = Blake2b256::hash(file_content);
146        assert_eq!(expected_hash, hash);
147    }
148
149    #[test]
150    fn calculate_dir_hash_test() {
151        let tmp_dir = TempDir::new().unwrap();
152        let file_content = "test".as_bytes();
153
154        let package_name = tmp_dir.child("test.hdf5");
155        let package = Package::create(package_name).unwrap();
156
157        let dir_name = "dir";
158        let dir = tmp_dir.child(dir_name);
159        std::fs::create_dir(&dir).unwrap();
160
161        let file_1_name = "file_1";
162        let file_1 = dir.join(file_1_name);
163        std::fs::write(file_1, file_content).unwrap();
164
165        let file_2_name = "file_2";
166        let file_2 = dir.join(file_2_name);
167        std::fs::write(file_2, file_content).unwrap();
168
169        let child_dir_name = "child_dir";
170        let child_dir = dir.join(child_dir_name);
171        std::fs::create_dir(&child_dir).unwrap();
172
173        let file_3_name = "file_3";
174        let file_3 = child_dir.join(file_3_name);
175        std::fs::write(file_3, file_content).unwrap();
176
177        let root_dir_name = "root_dir";
178        let root_dir = package.create_dir(root_dir_name.into()).unwrap();
179        root_dir.create_dir(dir_name.into()).unwrap();
180        root_dir
181            .copy_resource_dir(&FsResource::new(dir), &dir_name.into())
182            .unwrap();
183
184        let hash = package
185            .calculate_dir_hash(&format!("{root_dir_name}/{dir_name}").into())
186            .unwrap()
187            .unwrap();
188
189        let mut hasher = Blake2b256Hasher::new();
190        hasher.update(file_1_name.as_bytes());
191        hasher.update(file_content);
192        hasher.update(file_2_name.as_bytes());
193        hasher.update(file_content);
194        hasher.update(child_dir_name.as_bytes());
195        hasher.update(file_3_name.as_bytes());
196        hasher.update(file_content);
197        let expected_hash = hasher.finalize();
198
199        assert_eq!(expected_hash, hash);
200    }
