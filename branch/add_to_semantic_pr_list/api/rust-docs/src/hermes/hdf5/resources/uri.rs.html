1//! URI hermes specific parsing implementation.
2
3// cspell: words splitn
4
5/// URI resource definition.
6/// This definition mainly based on the [URI RFC](https://tools.ietf.org/html/rfc3986),
7/// but the implementation is not compliant with it and conforms with our needs.
8/// The parsing pattern is as follows:
9/// `[schema] :// [host] / [path]`
10#[derive(Debug, PartialEq, Eq)]
11pub(crate) struct Uri {
12    /// URI schema component.
13    pub(crate) schema: Option<String>,
14    /// URI host component.
15    pub(crate) host: Option<String>,
16    /// URI path component.
17    pub(crate) path: Option<String>,
18}
19
20impl Uri {
21    /// Parse URI from string with the following pattern:
22    /// `[schema] :// [host] / [path]`
23    #[allow(clippy::indexing_slicing)]
24    pub(crate) fn parse_from_str(s: &str) -> Self {
25        let schema_and_host_and_path = s.splitn(2, "://").collect::<Vec<_>>();
26        let mut schema = None;
27        let mut host = None;
28        let mut path = None;
29
30        if schema_and_host_and_path.len() == 2 {
31            schema = Some(schema_and_host_and_path[0].to_string());
32
33            let host_and_path = schema_and_host_and_path[1]
34                .splitn(2, '/')
35                .collect::<Vec<_>>();
36            if host_and_path.len() == 2 {
37                host = Some(host_and_path[0].to_string());
38                path = Some(host_and_path[1].to_string());
39            } else {
40                host = Some(host_and_path[0].to_string());
41            }
42        } else {
43            path = Some(schema_and_host_and_path[0].to_string());
44        }
45
46        Self {
47            schema: schema.filter(|s| !s.is_empty()),
48            host: host.filter(|s| !s.is_empty()),
49            path: path.filter(|s| !s.is_empty()),
50        }
51    }
52}

54#[cfg(test)]
55mod tests {
56    use super::*;
57
58    #[test]
59    fn uri_parsing_test() {
60        assert_eq!(
61            Uri::parse_from_str("https://www.google.com/file.txt"),
62            Uri {
63                schema: Some("https".to_string()),
64                host: Some("www.google.com".to_string()),
65                path: Some("file.txt".to_string())
66            }
67        );
68        assert_eq!(Uri::parse_from_str("://www.google.com/file.txt"), Uri {
69            schema: None,
70            host: Some("www.google.com".to_string()),
71            path: Some("file.txt".to_string())
72        });
73        assert_eq!(Uri::parse_from_str("www.google.com/file.txt"), Uri {
74            schema: None,
75            host: None,
76            path: Some("www.google.com/file.txt".to_string()),
77        });
78        assert_eq!(Uri::parse_from_str("file://www.google.com"), Uri {
79            schema: Some("file".to_string()),
80            host: Some("www.google.com".to_string()),
81            path: None
82        });
83        assert_eq!(Uri::parse_from_str("file:///../file.txt"), Uri {
84            schema: Some("file".to_string()),
85            host: None,
86            path: Some("../file.txt".to_string())
87        });
88        assert_eq!(Uri::parse_from_str("file:///~/file.txt"), Uri {
89            schema: Some("file".to_string()),
90            host: None,
91            path: Some("~/file.txt".to_string())
92        });
93        assert_eq!(Uri::parse_from_str("file:///file.txt"), Uri {
94            schema: Some("file".to_string()),
95            host: None,
96            path: Some("file.txt".to_string())
97        });
98        assert_eq!(Uri::parse_from_str("file.txt"), Uri {
99            schema: None,
100            host: None,
101            path: Some("file.txt".to_string())
102        });
103        assert_eq!(Uri::parse_from_str("/file.txt"), Uri {
104            schema: None,
105            host: None,
106            path: Some("/file.txt".to_string())
107        });
108        assert_eq!(Uri::parse_from_str("./file.txt"), Uri {
109            schema: None,
110            host: None,
111            path: Some("./file.txt".to_string())
112        });
113        assert_eq!(Uri::parse_from_str("~/file.txt"), Uri {
114            schema: None,
115            host: None,
116            path: Some("~/file.txt".to_string())
117        });
118        assert_eq!(Uri::parse_from_str("../file.txt"), Uri {
119            schema: None,
120            host: None,
121            path: Some("../file.txt".to_string())
122        });
123        assert_eq!(Uri::parse_from_str(""), Uri {
124            schema: None,
125            host: None,
126            path: None,
127        });
128    }
129}
