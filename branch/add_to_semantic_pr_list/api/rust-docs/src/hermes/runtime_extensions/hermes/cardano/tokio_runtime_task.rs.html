1//! The Tokio Runtime task is responsible for executing a Tokio runtime inside
2//! a OS thread so that Futures can be executed by the Cardano Runtime Extension.

12/// Command data that can be send to the Tokio runtime background thread.
13enum Command {
14    /// Instructs the Tokio runtime background thread to spawn a new chain follower.
15    SpawnFollower {
16        /// Name of the app that the follower will be tied to.
17        app_name: ApplicationName,
18        /// ID of the module that the follower will be tied to.
19        module_id: ModuleId,
20        /// Cardano blockchain that the follower will connect to.
21        chain_id: CardanoBlockchainId,
22        /// Follower's starting point.
23        follow_from: cardano_chain_follower::PointOrTip,
24        /// Response channel sender.
25        response_tx: tokio::sync::oneshot::Sender<
26            Result<(
27                super::chain_follower_task::Handle,
28                cardano_chain_follower::Point,
29            )>,
30        >,
31    },
32    /// Instructs the Tokio runtime background thread to read a block using some follower.
33    ReadBlock {
34        /// Cardano blockchain from which the block will be fetched.
35        chain_id: CardanoBlockchainId,
36        /// Chain point at which the block is to be fetched.
37        at: cardano_chain_follower::PointOrTip,
38        /// Response channel sender.
39        response_tx:
40            tokio::sync::oneshot::Sender<Result<cardano_chain_follower::MultiEraBlockData>>,
41    },
42}
43
44/// Tokio runtime handle command channel sender type.
45type CommandSender = tokio::sync::mpsc::Sender<Command>;
46
47/// Tokio runtime handle command channel receiver type.
48type CommandReceiver = tokio::sync::mpsc::Receiver<Command>;
49
50/// Handle used for communicating with the Tokio runtime background thread.
51pub struct Handle {
52    /// Commands channel sender.
53    cmd_tx: CommandSender,
54}
55
56impl Handle {
57    /// Spawns a new chain follower in the background Tokio runtime.
58    ///
59    /// # Errors
60    ///
61    /// Returns Err if the chain follower executor task could not be spawned.
62    pub fn spawn_follower_sync(
63        &self, app_name: ApplicationName, module_id: ModuleId, chain_id: CardanoBlockchainId,
64        follow_from: cardano_chain_follower::PointOrTip,
65    ) -> Result<(
66        super::chain_follower_task::Handle,
67        cardano_chain_follower::Point,
68    )> {
69        let (response_tx, response_rx) = tokio::sync::oneshot::channel();
70        let cmd = Command::SpawnFollower {
71            app_name,
72            module_id,
73            chain_id,
74            follow_from,
75            response_tx,
76        };
77
78        self.cmd_tx.blocking_send(cmd)?;
79
80        response_rx.blocking_recv()?
81    }
82
83    /// Reads a block from a Cardano network.
84    ///
85    /// # Errors
86    ///
87    /// Return Err if there were any errors while fetching the block.
88    pub fn read_block(
89        &self, chain_id: CardanoBlockchainId, at: cardano_chain_follower::PointOrTip,
90    ) -> Result<cardano_chain_follower::MultiEraBlockData> {
91        let (response_tx, response_rx) = tokio::sync::oneshot::channel();
92
93        let cmd = Command::ReadBlock {
94            chain_id,
95            at,
96            response_tx,
97        };
98
99        self.cmd_tx.blocking_send(cmd)?;
100
101        response_rx.blocking_recv()?
102    }
103}

105/// Spawns a OS thread running the Tokio runtime task.
106pub fn spawn() -> Handle {
107    let (cmd_tx, cmd_rx) = tokio::sync::mpsc::channel(1);
108    std::thread::spawn(move || {
109        executor(cmd_rx);
110    });
111
112    Handle { cmd_tx }
113}
114
115/// Runs the Cardano Runtime Extension Tokio runtime.
116#[instrument(skip(cmd_rx))]
117fn executor(mut cmd_rx: CommandReceiver) {
118    let res = tokio::runtime::Builder::new_current_thread()
119        .enable_io()
120        .enable_time()
121        .build();
122
123    let rt = match res {
124        Ok(rt) => rt,
125        Err(err) => {
126            error!(error = ?err, "Failed to start Cardano Runtime Extension background thread");
127            return;
128        },
129    };
130
131    trace!("Created Tokio runtime");
132
133    rt.block_on(async move {
134        while let Some(cmd) = cmd_rx.recv().await {
135            match cmd {
136                Command::SpawnFollower {
137                    app_name,
138                    module_id,
139                    chain_id,
140                    follow_from,
141                    response_tx,
142                } => {
143                    let res = spawn_follower(app_name, module_id, chain_id, follow_from).await;
144                    drop(response_tx.send(res));
145                },
146                Command::ReadBlock {
147                    chain_id,
148                    at,
149                    response_tx,
150                } => {
151                    let res = read_block(chain_id, at).await;
152                    drop(response_tx.send(res));
153                },
154            }
155        }
156    });
157}

159/// Spawns a follower which will follow the given chain and sets its starting point to the
160/// given point.
161async fn spawn_follower(
162    app_name: ApplicationName, module_id: ModuleId, chain_id: CardanoBlockchainId,
163    follow_from: cardano_chain_follower::PointOrTip,
164) -> Result<(
165    super::chain_follower_task::Handle,
166    cardano_chain_follower::Point,
167)> {
168    trace!("Spawning chain follower executor");
169
170    let config = cardano_chain_follower::FollowerConfigBuilder::default().build();
171    let network = chain_id.into();
172
173    let follower = cardano_chain_follower::Follower::connect(
174        follower_connect_address(network),
175        network,
176        config,
177    )
178    .await?;
179
180    trace!("Started chain follower");
181
182    let point = follower
183        .set_read_pointer(follow_from)
184        .await?
185        .ok_or(anyhow::anyhow!("Failed to locate follower starting point"))?;
186
187    trace!("Set chain follower starting point");
188
189    let handle = super::chain_follower_task::spawn(follower, app_name, module_id, chain_id);
190
191    Ok((handle, point))
192}

194/// Reads a block from the given chain at the given point.
195async fn read_block(
196    chain_id: CardanoBlockchainId, at: cardano_chain_follower::PointOrTip,
197) -> Result<cardano_chain_follower::MultiEraBlockData> {
198    trace!("Reading block");
199
200    let network = chain_id.into();
201
202    if let Some(reader) = STATE.readers.get(&network) {
203        let block_data = reader.read_block(at).await?;
204
205        Ok(block_data)
206    } else {
207        // Limit the follower's buffer size. This does not really matter
208        // since we'll not poll the
209        // follower's future so the following process will
210        // not be executed.
211        let cfg = cardano_chain_follower::FollowerConfigBuilder::default()
212            .chain_update_buffer_size(1)
213            .build();
214
215        let reader = cardano_chain_follower::Follower::connect(
216            follower_connect_address(network),
217            network,
218            cfg,
219        )
220        .await?;
221
222        let block_data = reader.read_block(at).await?;
223
224        Ok(block_data)
225    }
226}

228/// Returns the peer address used to connect to each Cardano network.
229const fn follower_connect_address(network: cardano_chain_follower::Network) -> &'static str {
230    match network {
231        cardano_chain_follower::Network::Mainnet => "backbone.cardano-mainnet.iohk.io:3001",
232        cardano_chain_follower::Network::Preprod => "preprod-node.play.dev.cardano.org:3001",
233        cardano_chain_follower::Network::Preview => "preview-node.play.dev.cardano.org:3001",
234        cardano_chain_follower::Network::Testnet => todo!(),
235    }
