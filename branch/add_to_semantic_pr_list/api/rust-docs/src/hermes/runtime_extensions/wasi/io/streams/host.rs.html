39    /// Read bytes from a stream, after blocking until at least one byte can
40    /// be read. Except for blocking, behavior is identical to `read`.
41    fn blocking_read(
42        &mut self, resource: wasmtime::component::Resource<InputStream>, len: u64,
43    ) -> wasmtime::Result<Result<Vec<u8>, StreamError>> {
44        let mut app_state = get_input_streams_state().get_app_state(self.app_name())?;
45        let Ok(mut stream) = app_state.get_object(&resource) else {
46            return Ok(Err(StreamError::Closed));
47        };
48
49        let Ok(len) = usize::try_from(len) else {
50            return Ok(Err(StreamError::Closed));
51        };
52        let mut buf = vec![0u8; len];
53        let Ok(read_len) = stream.read(&mut buf) else {
54            return Ok(Err(StreamError::Closed));
55        };
56        buf.truncate(read_len);
57
58        Ok(Ok(buf))
59    }
60

62    ///
63    /// Behaves identical to `read`, except instead of returning a list
64    /// of bytes, returns the number of bytes consumed from the stream.

67    ) -> wasmtime::Result<Result<u64, StreamError>> {
68        todo!()
69    }
70
71    /// Skip bytes from a stream, after blocking until at least one byte
72    /// can be skipped. Except for blocking behavior, identical to `skip`.
73    fn blocking_skip(
74        &mut self, rep: wasmtime::component::Resource<InputStream>, len: u64,
75    ) -> wasmtime::Result<Result<u64, StreamError>> {
76        let skipped = match self.blocking_read(rep, len)? {
77            Ok(buf) => buf.len(),
78            Err(e) => return Ok(Err(e)),
79        };
80
81        Ok(Ok(skipped as u64))
82    }
83
84    fn drop(&mut self, rep: wasmtime::component::Resource<InputStream>) -> wasmtime::Result<()> {
85        let app_state = get_input_streams_state().get_app_state(self.app_name())?;
86        app_state.delete_resource(rep)?;
87        Ok(())
88    }

93    ///
94    /// Returns the number of bytes permitted for the next call to `write`,
95    /// or an error. Calling `write` with more bytes than this function has
96    /// permitted will trap.

99    /// become ready when this function will report at least 1 byte, or an
100    /// error.
101    fn check_write(

104        /// The number of bytes permitted for the next call to `write`.

122    /// Perform a write of up to 4096 bytes, and then flush the stream. Block
123    /// until all of these operations are complete, or an error occurs.
124    ///
125    /// This is a convenience wrapper around the use of `check-write`,
126    /// `subscribe`, `write`, and `flush`, and is implemented with the
127    /// following pseudo-code:
128    ///
129    /// ```text
130    /// let pollable = this.subscribe();
131    /// while !contents.is_empty() {
132    /// // Wait for the stream to become writable
133    /// poll-one(pollable);
134    /// let Ok(n) = this.check-write(); // eliding error handling
135    /// let len = min(n, contents.len());
136    /// let (chunk, rest) = contents.split_at(len);
137    /// this.write(chunk  );            // eliding error handling
138    /// contents = rest;
139    /// }
140    /// this.flush();
141    /// // Wait for completion of `flush`
142    /// poll-one(pollable);
143    /// // Check for any errors that arose during `flush`
144    /// let _ = this.check-write();         // eliding error handling
145    /// ```
146    fn blocking_write_and_flush(
147        &mut self, res: wasmtime::component::Resource<OutputStream>, contents: Vec<u8>,
148    ) -> wasmtime::Result<Result<(), StreamError>> {
149        let mut app_state = get_output_streams_state().get_app_state(self.app_name())?;
150        let mut stream = app_state.get_object(&res)?;
151
152        if stream.write_all(&contents).is_err() {
153            return Ok(Err(StreamError::Closed));
154        }
155
156        Ok(Ok(()))
157    }
