2//!
3//! Processes HTTP requests through WASM modules using MPSC channels.
4//! Validates redirect URLs against configurable security policies.

19// ============================================================================
20// Type Aliases
21// ============================================================================
22
23/// HTTP status code (200, 404, 500, etc.)
24type Code = u16;
25
26/// Headers as key-value pairs, supporting multiple values per key
27pub type HeadersKV = Vec<(String, Vec<String>)>;
28
29/// URL path string
30type Path = String;
31
32/// HTTP method string (GET, POST, etc.)
33type Method = String;
34
35/// Request/response body as bytes
36type Body = Vec<u8>;
37
38// ============================================================================
39// Message Types
40// ============================================================================
41
42/// MPSC message types for HTTP event communication
43#[derive(Serialize, Deserialize)]
44pub(crate) enum HTTPEventMsg {
45    /// Receiver acknowledgment
46    HTTPEventReceiver,
47    /// Event response: (`status_code`, `headers`, `body`)
48    HttpEventResponse((Code, HeadersKV, Body)),
49}
50
51// ============================================================================
52// HTTP Event Structure
53// ============================================================================
54
55/// HTTP request event to be processed by WASM modules
56pub(crate) struct HTTPEvent {
57    /// HTTP headers as key-value pairs
58    pub(crate) headers: HeadersKV,
59    /// HTTP method (GET, POST, etc.)
60    pub(crate) method: Method,
61    /// Request URL path
62    pub(crate) path: Path,
63    /// Request body content
64    pub(crate) body: Bytes,
65    /// Channel to send response back to client
66    pub(crate) sender: Sender<HTTPEventMsg>,
67}

69// ============================================================================
70// Redirect Configuration

73/// Security configuration for validating internal redirects
74#[derive(Clone)]
75pub struct RedirectConfig {
76    /// Allowed URL schemes (e.g., "https")
77    pub schemes: HashSet<String>,
78    /// Allowed host names (e.g., "api.example.com")
79    pub hosts: HashSet<String>,
80    /// Allowed path prefixes (e.g., "/api/v1")
81    pub path_prefixes: Vec<String>,
82}
83
84impl Default for RedirectConfig {
85    fn default() -> Self {
86        Self {
87            schemes: ["https"].iter().map(ToString::to_string).collect(),
88            hosts: ["app.dev.projectcatalyst.io"]
89                .iter()
90                .map(ToString::to_string)
91                .collect(),
92            path_prefixes: vec!["/api/gateway".to_string()],
93        }
94    }
95}
96
97impl RedirectConfig {
98    /// Load from environment variables:
99    /// - `REDIRECT_ALLOWED_SCHEMES`
100    /// - `REDIRECT_ALLOWED_HOSTS`
101    /// - `REDIRECT_ALLOWED_PATH_PREFIXES`
102    pub fn from_env() -> Self {
103        let schemes = Self::parse_env_list("REDIRECT_ALLOWED_SCHEMES", "https");
104        let hosts = Self::parse_env_list("REDIRECT_ALLOWED_HOSTS", "app.dev.projectcatalyst.io");
105        let path_prefixes = Self::parse_env_list("REDIRECT_ALLOWED_PATH_PREFIXES", "/api/gateway")
106            .into_iter()
107            .collect();
108
109        Self {
110            schemes,
111            hosts,
112            path_prefixes,
113        }
114    }
115
116    /// Parse comma-separated environment variable
117    fn parse_env_list(
118        env_var: &str,
119        default: &str,
120    ) -> HashSet<String> {
121        env::var(env_var)
122            .unwrap_or_else(|_| default.to_string())
123            .split(',')
124            .map(|s| s.trim().to_string())
125            .collect()
126    }
127}

129// ============================================================================
130// Redirect Validation
131// ============================================================================
132
133/// Validates redirect URL against security policies
134fn validate_redirect_location(
135    location: &str,
136    config: &RedirectConfig,
137) -> anyhow::Result<()> {
138    let url =
139        Url::parse(location).map_err(|_| anyhow::anyhow!("Invalid redirect URL: {}", location))?;
140
141    validate_scheme(&url, config)?;
142    validate_host(&url, config)?;
143    validate_path(&url, config)?;
144
145    Ok(())
146}
147
148/// Validates URL scheme against allowed schemes
149fn validate_scheme(
150    url: &Url,
151    config: &RedirectConfig,
152) -> anyhow::Result<()> {
153    if !config.schemes.contains(url.scheme()) {
154        return Err(anyhow::anyhow!(
155            "Redirect scheme '{}' not allowed. Allowed schemes: {:?}",
156            url.scheme(),
157            config.schemes
158        ));
159    }
160    Ok(())
161}
162
163/// Validates URL host against allowed hosts
164fn validate_host(
165    url: &Url,
166    config: &RedirectConfig,
167) -> anyhow::Result<()> {
168    let host = url
169        .host_str()
170        .ok_or_else(|| anyhow::anyhow!("No host in redirect URL"))?;
171
172    if !config.hosts.contains(host) {
173        return Err(anyhow::anyhow!(
174            "Redirect host '{}' not allowed. Allowed hosts: {:?}",
175            host,
176            config.hosts
177        ));
178    }
179    Ok(())
180}
181
182/// Validates URL path against allowed prefixes
183fn validate_path(
184    url: &Url,
185    config: &RedirectConfig,
186) -> anyhow::Result<()> {
187    let path = url.path();
188    let path_allowed = config
189        .path_prefixes
190        .iter()
191        .any(|prefix| path.starts_with(prefix));
192
193    if !path_allowed {
194        return Err(anyhow::anyhow!(
195            "Redirect path '{}' not allowed. Allowed prefixes: {:?}",
196            path,
197            config.path_prefixes
198        ));
199    }
200    Ok(())
201}

207impl HermesEventPayload for HTTPEvent {
208    fn event_name(&self) -> &'static str {
209        "http-event"
210    }
211
212    fn execute(
213        &self,
214        module: &mut crate::wasm::module::ModuleInstance,
215    ) -> anyhow::Result<()> {
216        let event_response = module.instance.hermes_http_gateway_event().call_reply(
217            &mut module.store,
218            &self.body.as_ref().to_vec(),
219            &self.headers,
220            &self.path,
221            &self.method,
222        )?;
223
224        match event_response {
225            Some(HttpGatewayResponse::Http(resp)) => {
226                self.send_http_response(resp.code, resp.headers, resp.body)
227            },
228            Some(HttpGatewayResponse::InternalRedirect(location)) => {
229                self.handle_internal_redirect(location)
230            },
231            None => Ok(()),
232        }
233    }
234}
235
236impl HTTPEvent {
237    /// Send HTTP response back to client via MPSC channel
238    fn send_http_response(
239        &self,
240        code: Code,
241        headers: HeadersKV,
242        body: Body,
243    ) -> anyhow::Result<()> {
244        Ok(self
245            .sender
246            .send(HTTPEventMsg::HttpEventResponse((code, headers, body)))?)
247    }
248
249    /// Handle internal redirect with security validation
250    fn handle_internal_redirect(
251        &self,
252        location: String,
253    ) -> anyhow::Result<()> {
254        let config = RedirectConfig::from_env();
255
256        if let Err(e) = validate_redirect_location(&location, &config) {
257            error!("Invalid redirect location: {}", e);
258            return self.send_error_response(403, "Forbidden: Invalid redirect location");
259        }
260
261        self.spawn_redirect_request(location);
262        Ok(())
263    }
264
265    /// Spawn background thread for redirect request
266    fn spawn_redirect_request(
267        &self,
268        location: String,
269    ) {
270        let headers = self.headers.clone();
271        let method = self.method.clone();
272        let body = self.body.clone();
273        let sender = self.sender.clone();
274
275        std::thread::spawn(move || {
276            let client = blocking::Client::new();
277            let request = Self::build_request(&client, &location, &headers, &method, &body);
278
279            match request.send() {
280                Ok(response) => Self::process_response(response, &sender),
281                Err(e) => {
282                    error!("HTTP request failed: {:?}", e);
283                    Self::send_error_via_sender(&sender, 500, "Internal Server Error")
284                },
285            }
286        });
287    }
288
289    /// Build HTTP request for redirect (excludes Host header)
290    fn build_request(
291        client: &blocking::Client,
292        location: &str,
293        headers: &HeadersKV,
294        method: &str,
295        body: &Bytes,
296    ) -> blocking::RequestBuilder {
297        let mut request = client.request(
298            request::from_bytes(method.as_bytes()).unwrap_or(request::GET),
299            location,
300        );
301
302        // Add headers from original request, excluding Host
303        for (key, values) in headers {
304            if key.to_lowercase() != "host" {
305                for value in values {
306                    request = request.header(key, value);
307                }
308            }
309        }
310
311        // Add body if present
312        if !body.is_empty() {
313            request = request.body(body.to_vec());
314        }
315
316        request
317    }
318
319    /// Process HTTP response and forward to client
320    fn process_response(
321        response: blocking::Response,
322        sender: &Sender<HTTPEventMsg>,
323    ) -> anyhow::Result<()> {
324        let status_code = response.status().as_u16();
325        let headers: HeadersKV = response
326            .headers()
327            .iter()
328            .map(|(name, value)| {
329                (name.to_string(), vec![value
330                    .to_str()
331                    .unwrap_or("")
332                    .to_string()])
333            })
334            .collect();
335
336        match response.bytes() {
337            Ok(body) => {
338                sender.send(HTTPEventMsg::HttpEventResponse((
339                    status_code,
340                    headers,
341                    body.to_vec(),
342                )))?;
343            },
344            Err(e) => {
345                error!("Failed to read response body: {:?}", e);
346                Self::send_error_via_sender(sender, 500, "Internal Server Error")?;
347            },
348        }
349        Ok(())
350    }
351
352    /// Send error response to client
353    fn send_error_response(
354        &self,
355        code: Code,
356        message: &str,
357    ) -> anyhow::Result<()> {
358        self.sender.send(HTTPEventMsg::HttpEventResponse((
359            code,
360            vec![],
361            message.as_bytes().to_vec(),
362        )))?;
363        Ok(())
364    }
365
366    /// Helper to send error via sender channel
367    fn send_error_via_sender(
368        sender: &Sender<HTTPEventMsg>,
369        code: Code,
370        message: &str,
371    ) -> anyhow::Result<()> {
372        sender.send(HTTPEventMsg::HttpEventResponse((
373            code,
374            vec![],
375            message.as_bytes().to_vec(),
376        )))?;
377        Ok(())
378    }
379}
