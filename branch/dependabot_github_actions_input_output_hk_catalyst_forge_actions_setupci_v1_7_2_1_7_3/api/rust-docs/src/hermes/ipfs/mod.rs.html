32/// Hermes IPFS Internal Node
33///
34/// This is a wrapper around `HermesIpfsNode` which provides a singleton instance of the
35/// IPFS state.
36///
37/// This is done to ensure the IPFS Node is initialized only once when the
38/// `HermesIpfsNode` is first used. This is done to avoid any issues that may arise if
39/// the IPFS Node is initialized multiple times.
40///
41/// The IPFS Node is initialized in a separate thread and the sender channel is stored in
42/// the `HermesIpfsNode`.
43pub(crate) static HERMES_IPFS: OnceCell<HermesIpfsNode> = OnceCell::new();
44
45/// Bootstrap `HERMES_IPFS` node.
46///
47/// ## Errors
48///
49/// Returns errors if IPFS node fails to start.
50pub fn bootstrap(base_dir: &Path, default_bootstrap: bool) -> anyhow::Result<()> {
51    let ipfs_data_path = base_dir.join("ipfs");
52    let ipfs_node = HermesIpfsNode::init(
53        IpfsBuilder::new()
54            .with_default()
55            .set_default_listener()
56            .set_disk_storage(ipfs_data_path.clone()),
57        default_bootstrap,
58    )?;
59    HERMES_IPFS
60        .set(ipfs_node)
61        .map_err(|_| anyhow::anyhow!("failed to start IPFS node"))?;
62    Ok(())
63}

73impl HermesIpfsNode {
74    /// Create, initialize, and bootstrap a new `HermesIpfsNode`
75    pub(crate) fn init(builder: IpfsBuilder, default_bootstrap: bool) -> anyhow::Result<Self> {
76        let runtime = Builder::new_current_thread().enable_all().build()?;
77        let (sender, receiver) = mpsc::channel(1);
78        let _handle = std::thread::spawn(move || {
79            // Build and start IPFS node
80            let _unused = runtime.block_on(async move {
81                let node = builder.start().await?;
82                if default_bootstrap {
83                    // Add default addresses for bootstrapping
84                    let addresses = node.default_bootstrap().await?;
85                    // Connect to bootstrap nodes.
86                    node.bootstrap().await?;
87                    tracing::debug!(
88                        "Bootstrapped IPFS node with default addresses: {:?}",
89                        addresses
90                    );
91                }
92                let hermes_node: HermesIpfs = node.into();
93                let h = tokio::spawn(ipfs_command_handler(hermes_node, receiver));
94                let (..) = tokio::join!(h);
95                Ok::<(), anyhow::Error>(())
96            });
97            std::process::exit(0);
98        });
99        Ok(Self {
100            sender: Some(sender),
101            apps: AppIpfsState::new(),
102        })
103    }

105    /// Add file
106    ///
107    /// Returns the IPFS path of the added file
108    ///
109    /// ## Parameters
110    /// - `contents`: The content to add
111    ///
112    /// ## Errors
113    /// - `Errno::FileAddError`: Failed to add the content
114    fn file_add(&self, contents: IpfsFile) -> Result<hermes_ipfs::IpfsPath, Errno> {
115        let (cmd_tx, cmd_rx) = oneshot::channel();
116        self.sender
117            .as_ref()
118            .ok_or(Errno::FileAddError)?
119            .blocking_send(IpfsCommand::AddFile(
120                AddIpfsFile::Stream((None, contents)),
121                cmd_tx,
122            ))
123            .map_err(|_| Errno::FileAddError)?;
124        cmd_rx.blocking_recv().map_err(|_| Errno::FileAddError)?
125    }

263/// IPFS app state
264struct AppIpfsState {
265    /// List of pinned files per app.
266    pinned_files: DashMap<ApplicationName, HashSet<Cid>>,
267    /// List of DHT values per app.
268    dht_keys: DashMap<ApplicationName, HashSet<DhtKey>>,
269    /// List of subscriptions per app.
270    topic_subscriptions: DashMap<PubsubTopic, HashSet<ApplicationName>>,
271    /// Collection of stream join handles per topic subscription.
272    subscriptions_streams: DashMap<PubsubTopic, JoinHandle<()>>,
273    /// List of evicted peers per app.
274    evicted_peers: DashMap<ApplicationName, HashSet<PeerId>>,
275}
276
277impl AppIpfsState {
278    /// Create new `AppIpfsState`
279    fn new() -> Self {
280        Self {
281            pinned_files: DashMap::default(),
282            dht_keys: DashMap::default(),
283            topic_subscriptions: DashMap::default(),
284            subscriptions_streams: DashMap::default(),
285            evicted_peers: DashMap::default(),
286        }
287    }
288
289    /// Keep track of `ipfs_path` of file pinned by an app.
290    fn pinned_file(&self, app_name: ApplicationName, ipfs_path: &IpfsPath) -> Result<(), Errno> {
291        let ipfs_path: BaseIpfsPath = ipfs_path.parse().map_err(|_| Errno::InvalidIpfsPath)?;
292        let cid = ipfs_path.root().cid().ok_or(Errno::InvalidCid)?;
293        self.pinned_files
294            .entry(app_name)
295            .or_default()
296            .value_mut()
297            .insert(*cid);
298        Ok(())
299    }
300
301    /// Un-pin a file with `ipfs_path` pinned by an app.
302    fn unpinned_file(&self, app_name: &ApplicationName, ipfs_path: &IpfsPath) -> Result<(), Errno> {
303        let ipfs_path: BaseIpfsPath = ipfs_path.parse().map_err(|_| Errno::InvalidIpfsPath)?;
304        let cid = ipfs_path.root().cid().ok_or(Errno::InvalidCid)?;
305        self.pinned_files
306            .entry(app_name.clone())
307            .or_default()
308            .value_mut()
309            .remove(cid);
310        self.pinned_files.remove_if(app_name, |_, v| v.is_empty());
311        Ok(())
312    }
313
314    #[allow(dead_code)]
315    /// List of pinned files by an app.
316    pub(crate) fn list_pinned_files(&self, app_name: &ApplicationName) -> Vec<String> {
317        self.pinned_files.get(app_name).map_or(vec![], |cids| {
318            cids.iter().map(std::string::ToString::to_string).collect()
319        })
320    }
321
322    /// Keep track of `dht_key` of DHT value added by an app.
323    fn added_dht_key(&self, app_name: ApplicationName, dht_key: DhtKey) {
324        self.dht_keys
325            .entry(app_name)
326            .or_default()
327            .value_mut()
328            .insert(dht_key);
329    }
330
331    /// Keep track of `topic` subscription added by an app.
332    fn added_app_topic_subscription(&self, app_name: ApplicationName, topic: PubsubTopic) {
333        self.topic_subscriptions
334            .entry(topic)
335            .or_default()
336            .value_mut()
337            .insert(app_name);
338    }
339
340    /// Keep track of `topic` stream handle.
341    fn added_topic_stream(&self, topic: PubsubTopic, handle: JoinHandle<()>) {
342        self.subscriptions_streams.entry(topic).insert(handle);
343    }
344
345    /// Check if a topic subscription already exists.
346    fn topic_subscriptions_contains(&self, topic: &PubsubTopic) -> bool {
347        self.topic_subscriptions.contains_key(topic)
348    }
349
350    /// Returns a list of apps subscribed to a topic.
351    fn subscribed_apps(&self, topic: &PubsubTopic) -> Vec<ApplicationName> {
352        self.topic_subscriptions
353            .get(topic)
354            .map_or(vec![], |apps| apps.value().iter().cloned().collect())
355    }
356
357    /// Add `peer_id` of evicted peer by an app.
358    fn evicted_peer(&self, app_name: ApplicationName, peer_id: PeerId) {
359        self.evicted_peers
360            .entry(app_name)
361            .or_default()
362            .value_mut()
363            .insert(peer_id);
364    }
365}
