9/// Build application from the application package.
10pub(crate) fn build_app<P: AsRef<std::path::Path>>(
11    package: &ApplicationPackage, vfs_dir_path: P,
12) -> anyhow::Result<Application> {
13    let app_name = package.get_app_name()?;
14    let mut bootstrapper = VfsBootstrapper::new(vfs_dir_path, app_name.clone());
15    mount_to_vfs(package, &mut bootstrapper)?;
16    let vfs = bootstrapper.bootstrap()?;
17
18    let mut modules = Vec::new();
19    for module_info in package.get_modules()? {
20        let module = module_info.get_component()?;
21        modules.push(module);
22    }
23    let app = Application::new(app_name, vfs, modules);
24
25    Ok(app)
26}

28/// Mount `ApplicationPackage` content to the `Vfs`
29fn mount_to_vfs(
30    package: &ApplicationPackage, bootstrapper: &mut VfsBootstrapper,
31) -> anyhow::Result<()> {
32    let root_path = "/".to_string();
33    bootstrapper.with_mounted_file(
34        root_path.clone(),
35        package.get_icon_file()?,
36        PermissionLevel::Read,
37    );
38    bootstrapper.with_mounted_file(
39        root_path.clone(),
40        package.get_metadata_file()?,
41        PermissionLevel::Read,
42    );
43    if let Some(share_dir) = package.get_share_dir() {
44        bootstrapper.with_mounted_dir(root_path.clone(), share_dir, PermissionLevel::Read);
45    }
46    if let Some(www_dir) = package.get_www_dir() {
47        bootstrapper.with_mounted_dir(root_path, www_dir, PermissionLevel::Read);
48    }
49
50    for module_info in package.get_modules()? {
51        let lib_module_dir_path = format!("{}/{}", Vfs::LIB_DIR, module_info.get_name());
52        bootstrapper.with_dir_to_create(lib_module_dir_path.clone(), PermissionLevel::Read);
53
54        bootstrapper.with_mounted_file(
55            lib_module_dir_path.clone(),
56            module_info.get_metadata_file()?,
57            PermissionLevel::Read,
58        );
59        bootstrapper.with_mounted_file(
60            lib_module_dir_path.clone(),
61            module_info.get_component_file()?,
62            PermissionLevel::Read,
63        );
64        if let Some(config_schema) = module_info.get_config_schema_file() {
65            bootstrapper.with_mounted_file(
66                lib_module_dir_path.clone(),
67                config_schema,
68                PermissionLevel::Read,
69            );
70        }
71        if let Some(config) = module_info.get_config_file() {
72            bootstrapper.with_mounted_file(
73                lib_module_dir_path.clone(),
74                config,
75                PermissionLevel::Read,
76            );
77        }
78        if let Some(settings_schema) = module_info.get_settings_schema_file() {
79            bootstrapper.with_mounted_file(
80                lib_module_dir_path.clone(),
81                settings_schema,
82                PermissionLevel::Read,
83            );
84        }
85        if let Some(share_dir) = module_info.get_share_dir() {
86            bootstrapper.with_mounted_dir(lib_module_dir_path, share_dir, PermissionLevel::Read);
87        }
88    }
89    Ok(())
90}
