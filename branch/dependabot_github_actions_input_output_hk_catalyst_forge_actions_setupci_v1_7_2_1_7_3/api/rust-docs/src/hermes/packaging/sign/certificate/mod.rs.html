11/// Certificate decoding from string error.
12#[derive(thiserror::Error, Debug)]
13#[error("Cannot decode certificate from string. Invalid PEM format.")]
14pub(crate) struct CertificateDecodingError;
15
16/// x.509 cert instance.
17/// Wrapper over `x509_cert::Certificate`
18#[derive(Clone, Debug)]
19pub(crate) struct Certificate(x509_cert::Certificate);
20
21impl Certificate {
22    /// Create new certificate from file decoded in PEM format
23    pub(crate) fn from_file<P: AsRef<Path>>(path: P) -> anyhow::Result<Self> {
24        let str = std::fs::read_to_string(&path).map_err(|_| FileError::from_path(&path, None))?;
25
26        Ok(Self::from_str(&str).map_err(|err| FileError::from_path(&path, Some(err)))?)
27    }
28
29    /// Create new certificate from string decoded in PEM format
30    pub(crate) fn from_str(str: &str) -> anyhow::Result<Self> {
31        let cert = x509_cert::Certificate::from_pem(str.as_bytes())
32            .map_err(|_| CertificateDecodingError)?;
33        Ok(Self(cert))
34    }
35
36    /// Get certificate's subject public key.
37    pub(crate) fn subject_public_key(&self) -> anyhow::Result<PublicKey> {
38        let subject_public_key = &self
39            .0
40            .tbs_certificate
41            .subject_public_key_info
42            .subject_public_key;
43
44        PublicKey::from_bytes(subject_public_key.raw_bytes()).map_err(|err| {
45            anyhow::anyhow!("Failed to decode certificate subject public key. {err}")
46        })
47    }
48
49    /// `Blake2b256` hash of the certificate DER encoded bytes.
50    pub(crate) fn hash(&self) -> anyhow::Result<Blake2b256> {
51        let der_bytes = self.0.to_der()?;
52        Ok(Blake2b256::hash(&der_bytes))
53    }
54}
