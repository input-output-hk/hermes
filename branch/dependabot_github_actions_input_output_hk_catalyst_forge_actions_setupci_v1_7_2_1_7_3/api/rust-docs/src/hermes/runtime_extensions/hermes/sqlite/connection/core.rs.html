15/// Checks if the provided SQL string contains a `PRAGMA` statement.
16/// Generally, `PRAGMA` is intended for internal use only.
17pub(crate) fn validate_sql(sql: &str) -> bool {
18    sql.to_uppercase().sz_find("PRAGMA ".as_bytes()).is_some()
19}
20
21/// Closes a database connection, destructor for `sqlite3`.
22pub(crate) fn close(db_ptr: *mut sqlite3) -> Result<(), Errno> {
23    let rc = unsafe { sqlite3_close(db_ptr) };
24
25    if rc == SQLITE_OK {
26        Ok(())
27    } else {
28        Err(Errno::Sqlite(rc))
29    }
30}

32/// Retrieves runtime status information about a single database connection.
33pub(crate) fn errcode(db_ptr: *mut sqlite3) -> Option<ErrorInfo> {
34    let (error_code, error_msg) = unsafe { (sqlite3_errcode(db_ptr), sqlite3_errmsg(db_ptr)) };
35
36    if error_code == SQLITE_OK {
37        return None;
38    }
39
40    let message = unsafe {
41        std::ffi::CStr::from_ptr(error_msg)
42            .to_str()
43            .map(std::borrow::ToOwned::to_owned)
44            .ok()
45    };
46
47    message.map(|message| {
48        ErrorInfo {
49            code: error_code,
50            message,
51        }
52    })
53}

55/// Compiles SQL text into byte-code that will do the work of querying or updating the
56/// database.
57pub(crate) fn prepare(db_ptr: *mut sqlite3, sql: &str) -> Result<*mut sqlite3_stmt, Errno> {
58    if validate_sql(sql) {
59        return Err(Errno::ForbiddenPragmaCommand);
60    }
61
62    let sql_cstring = std::ffi::CString::new(sql).map_err(|_| Errno::ConvertingCString)?;
63
64    let mut stmt_ptr: *mut sqlite3_stmt = std::ptr::null_mut();
65
66    let n_byte = sql_cstring.as_bytes_with_nul().len();
67    let n_byte = i32::try_from(n_byte).map_err(|_| Errno::ConvertingNumeric)?;
68
69    let rc = unsafe {
70        sqlite3_prepare_v3(
71            db_ptr,
72            sql_cstring.as_ptr(),
73            n_byte,
74            0,
75            &mut stmt_ptr,
76            std::ptr::null_mut(),
77        )
78    };
79
80    if rc == SQLITE_OK {
81        Ok(stmt_ptr)
82    } else {
83        Err(Errno::Sqlite(rc))
84    }
85}
86
87/// Executes an SQL query directly without preparing it into a statement and returns
88/// the result.
89pub(crate) fn execute(db_ptr: *mut sqlite3, sql: &str) -> Result<(), Errno> {
90    let sql_cstring = std::ffi::CString::new(sql).map_err(|_| Errno::ConvertingCString)?;
91
92    let rc = unsafe { sqlite3_exec(db_ptr, sql_cstring.as_ptr(), None, null_mut(), null_mut()) };
93
94    if rc == SQLITE_OK {
95        Ok(())
96    } else {
97        Err(Errno::Sqlite(rc))
98    }
99}

109    const TMP_DIR: &str = "tmp-dir";
110
111    fn init() -> Result<*mut sqlite3, Errno> {
112        let app_name = ApplicationName(String::from(TMP_DIR));
113
114        open(false, true, app_name)
115    }

117    #[test]
118    fn test_validate_pragma() {
119        let db_ptr = init().unwrap();
120
121        let sql = "PRAGMA page_size;";
122        let stmt_ptr = prepare(db_ptr, sql);
123
124        assert!(matches!(stmt_ptr, Err(Errno::ForbiddenPragmaCommand)));
125
126        let sql = "pragma page_size;";
127        let stmt_ptr = prepare(db_ptr, sql);
128
129        assert!(matches!(stmt_ptr, Err(Errno::ForbiddenPragmaCommand)));
130
131        close(db_ptr).unwrap();
132    }
