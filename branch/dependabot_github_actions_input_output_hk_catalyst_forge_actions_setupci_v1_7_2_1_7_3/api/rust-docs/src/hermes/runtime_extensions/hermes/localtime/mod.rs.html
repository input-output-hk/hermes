17/// Get `Tz` from an optional `Timezone`.
18///
19/// If present, `tz` is parsed into `Tz`, otherwise the system timezone is
20/// returned.
21///
22/// This function returns `Errno::UnknownTimezone` if `tz` cannot be parsed.
23///
24/// **Parameters**
25/// * `tz`: `Option<Timezone>`.
26///
27/// **Returns**
28/// `wasmtime::Result<Result<Tz, Errno>>`.
29pub(crate) fn get_tz(tz: Option<Timezone>) -> Result<Tz, Errno> {
30    let timezone_str = match tz {
31        Some(tz_str) => tz_str,
32        None => iana_time_zone::get_timezone().map_err(|_| Errno::UnknownTimezone)?,
33    };
34    timezone_str.parse().map_err(|_| Errno::UnknownTimezone)
35}

37impl TryFrom<chrono::DateTime<Tz>> for Localtime {
38    type Error = Errno;
39
40    fn try_from(dt: chrono::DateTime<Tz>) -> Result<Self, Self::Error> {
41        let localtime = Localtime {
42            year: dt.year().try_into().map_err(|_| Errno::YearOutOfRange)?,
43            month: dt.month().try_into().map_err(|_| Errno::InvalidLocaltime)?,
44            dow: dt
45                .weekday()
46                .number_from_monday()
47                .try_into()
48                .map_err(|_| Errno::InvalidLocaltime)?,
49            day: dt.day().try_into().map_err(|_| Errno::InvalidLocaltime)?,
50            hh: dt.hour().try_into().map_err(|_| Errno::InvalidLocaltime)?,
51            mm: dt
52                .minute()
53                .try_into()
54                .map_err(|_| Errno::InvalidLocaltime)?,
55            ss: dt
56                .second()
57                .try_into()
58                .map_err(|_| Errno::InvalidLocaltime)?,
59            ns: dt.nanosecond(),
60            tz: dt.offset().tz_id().to_string(),
61        };
62        Ok(localtime)
63    }
64}

66impl TryInto<chrono::DateTime<Tz>> for Localtime {
67    type Error = Errno;
68
69    fn try_into(self) -> Result<chrono::DateTime<Tz>, Self::Error> {
70        let Localtime {
71            year,
72            month,
73            dow: _,
74            day,
75            hh,
76            mm,
77            ss,
78            ns,
79            tz: orig_tz,
80        } = self;
81        let orig_tz = get_tz(Some(orig_tz))?;
82        let year: i32 = year.try_into().map_err(|_| Errno::YearOutOfRange)?;
83        let month: u32 = month.into();
84        let day: u32 = day.into();
85        let hh: u32 = hh.into();
86        let mm: u32 = mm.into();
87        let ss: u32 = ss.into();
88        let ns: u32 = ns;
89        let naive_date_time = NaiveDate::from_ymd_opt(year, month, day)
90            .ok_or(Errno::InvalidLocaltime)?
91            .and_hms_nano_opt(hh, mm, ss, ns)
92            .ok_or(Errno::InvalidLocaltime)?;
93        let date_time = orig_tz.from_utc_datetime(&naive_date_time);
94        Ok(date_time)
95    }
96}

98impl TryFrom<chrono::DateTime<Tz>> for Datetime {
99    type Error = Errno;
100
101    fn try_from(value: chrono::DateTime<Tz>) -> Result<Self, Self::Error> {
102        let seconds = value
103            .timestamp()
104            .try_into()
105            .map_err(|_| Errno::InvalidLocaltime)?;
106        let nanoseconds = value.nanosecond();
107        Ok(Datetime {
108            seconds,
109            nanoseconds,
110        })
111    }
112}
