6/// Get a block relative to `start` by `step`.
7pub(crate) fn get_block_relative(
8    network: Network,
9    start: Option<u64>,
10    step: i64,
11) -> anyhow::Result<MultiEraBlock> {
12    let handle = std::thread::spawn(move || -> anyhow::Result<MultiEraBlock> {
13        let point = if let Some(start_point) = start {
14            calculate_point_from_step(start_point, step)?
15        // If start is None, use the tip
16        } else {
17            Point::TIP
18        };
19
20        let rt = match tokio::runtime::Builder::new_current_thread()
21            .enable_time()
22            .enable_io()
23            .build()
24        {
25            Ok(rt) => rt,
26            Err(e) => {
27                return Err(anyhow::anyhow!("Failed to create Tokio runtime: {e}"));
28            },
29        };
30
31        let block = rt
32            .block_on(ChainFollower::get_block(network, point.clone()))
33            .ok_or_else(|| anyhow::anyhow!("Failed to fetch block at point {point}"))?;
34
35        Ok(block.data)
36    });
37
38    handle
39        .join()
40        .map_err(|e| anyhow::anyhow!("Thread panicked while getting block: {e:?}"))?
41}

43/// Calculate point from start and step
44fn calculate_point_from_step(
45    start: u64,
46    step: i64,
47) -> anyhow::Result<Point> {
48    let target = if step.is_negative() {
49        start
50            .checked_sub(step.unsigned_abs())
51            .ok_or_else(|| anyhow::anyhow!("Step causes underflow"))?
52    } else {
53        start
54            .checked_add(step.unsigned_abs())
55            .ok_or_else(|| anyhow::anyhow!("Step causes overflow"))?
56    };
57    Ok(Point::fuzzy(target.into()))
58}

60/// Retrieves the current tips of the blockchain for the specified network.
61pub(crate) fn get_tips(network: Network) -> anyhow::Result<(Slot, Slot)> {
62    let handle = std::thread::spawn(move || -> anyhow::Result<(Slot, Slot)> {
63        let rt = match tokio::runtime::Builder::new_current_thread()
64            .enable_time()
65            .enable_io()
66            .build()
67        {
68            Ok(rt) => rt,
69            Err(e) => {
70                return Err(anyhow::anyhow!("Failed to create Tokio runtime: {e}"));
71            },
72        };
73
74        let (immutable_tip, live_tip) = rt.block_on(ChainFollower::get_tips(network));
75        Ok((immutable_tip.slot_or_default(), live_tip.slot_or_default()))
76    });
77
78    handle
79        .join()
80        .map_err(|e| anyhow::anyhow!("Thread panicked while getting tips: {e:?}"))?
81}

83/// Checks if the block at the given slot is a rollback block or not.
84pub(crate) fn get_is_rollback(
85    network: Network,
86    slot: Slot,
87) -> anyhow::Result<Option<bool>> {
88    let handle = std::thread::spawn(move || -> anyhow::Result<Option<bool>> {
89        let rt = match tokio::runtime::Builder::new_current_thread()
90            .enable_time()
91            .enable_io()
92            .build()
93        {
94            Ok(rt) => rt,
95            Err(e) => {
96                return Err(anyhow::anyhow!("Failed to create Tokio runtime: {e}"));
97            },
98        };
99
100        let point = Point::fuzzy(slot);
101        let block = rt.block_on(ChainFollower::get_block(network, point));
102        match block {
103            // Block found
104            Some(block) => Ok(Some(block.kind == Kind::Rollback)),
105            // Block not found
106            None => Ok(None),
107        }
108    });
109
110    handle
111        .join()
112        .map_err(|e| anyhow::anyhow!("Thread panicked while getting block rollback: {e:?}"))?
113}

122    fn test_positive_step() -> Result<()> {
123        let start = 100;
124        let step = 50;
125        let point = calculate_point_from_step(start, step)?;
126        assert_eq!(point.slot_or_default(), 150.into());
127        Ok(())
128    }

130    #[test]

132        let start = 100;
133        let step = -30;

135        assert_eq!(point.slot_or_default(), 70.into());
136        Ok(())
137    }
138
139    #[test]
140    fn test_zero_step() -> Result<()> {
141        let start = 100;
142        let step = 0;
143        let point = calculate_point_from_step(start, step)?;
144        assert_eq!(point.slot_or_default(), 100.into());
145        Ok(())
146    }

148    #[test]

150        let start = 10;
151        let step = -100;
