24/// Create a delayed crontab entry.
25pub(crate) fn mkdelay_crontab(
26    duration: Instant,
27    tag: CronEventTag,
28) -> wasmtime::Result<CronJobDelay> {
29    // Add the delay to the current time.
30    let delayed = Utc::now()
31        .checked_add_signed(TimeDelta::nanoseconds(duration.try_into()?))
32        .ok_or(anyhow::anyhow!("Invalid Timestamp"))?;
33    let timestamp = delayed
34        .timestamp_nanos_opt()
35        .ok_or(anyhow::anyhow!("Invalid Timestamp"))?
36        .try_into()?;
37    let (month, day): (u8, u8) = (delayed.month().try_into()?, delayed.day().try_into()?);
38    let (hour, minute): (u8, u8) = (delayed.hour().try_into()?, delayed.minute().try_into()?);
39    let day_of_week = delayed.weekday().number_from_monday().try_into()?;
40    let when = mkcron_impl(
41        &vec![CronComponent::At(day_of_week)],
42        &vec![CronComponent::At(month)],
43        &vec![CronComponent::At(day)],
44        &vec![CronComponent::At(hour)],
45        &vec![CronComponent::At(minute)],
46    );
47    Ok(CronJobDelay {
48        timestamp,
49        event: OnCronEvent {
50            tag: CronTagged { when, tag },
51            last: true,
52        },
53    })
54}

56/// Convert `CronTime` arguments to a `CronSched`.
57pub(crate) fn mkcron_impl(
58    dow: &CronTime,
59    month: &CronTime,
60    day: &CronTime,
61    hour: &CronTime,
62    minute: &CronTime,
63) -> CronSched {
64    let dow_schedule: CronSched = cron_time_to_cron_sched_dow(dow);
65    let month_schedule: CronSched = cron_time_to_cron_sched_month(month);
66    let day_schedule: CronSched = cron_time_to_cron_sched_day(day);
67    let hour_schedule: CronSched = cron_time_to_cron_sched_hour(hour);
68    let minute_schedule: CronSched = cron_time_to_cron_sched_minute(minute);
69    // Return the merged schedule.
70    format!("{minute_schedule} {hour_schedule} {day_schedule} {month_schedule} {dow_schedule}",)
71}

73/// Convert a `CronTime` to a `CronSched`.
74///
75/// Silently clamps values within the specified `min_val..=max_val` range, removes
76/// duplicates, merges overlaps, and ensures that range values are in the right order:
77/// `first <= last`.
78///
79/// If the `CronTime` contains no components, returns `*`.
80/// If the `CronTime` contains `CronComponent::All`, returns `*`.
81/// If the `CronTime` contains `CronComponent::Range(first, last)`, returns `*`.
82/// If the `CronTime` contains overlapping components, it merges them.
83///
84/// Returns `CronSched`
85fn cron_time_to_cron_sched(
86    cron_time: &CronTime,
87    min_val: u8,
88    max_val: u8,
89) -> CronSched {
90    // If vec has no components or if it includes `CronComponent::All`, skip processing and
91    // return "*"
92    let cron_sched: CronSched = if cron_time.is_empty() {
93        format!("{}", CronComponent::All)
94    } else {
95        // Silently clamp values, and ensure that range values are in the right order: `first <=
96        // last`. For the case of finding `CronComponent::Range((final, last))`, it is
97        // replaced with `CronComponent::All`.
98        let clamped: Vec<CronComponent> = clamp_cron_time_values(cron_time, min_val, max_val);
99        // Merge overlapping components
100        let merged: Vec<CronComponent> = merge_cron_time_overlaps(&clamped);
101        // Return the merged cron schedule
102        merged
103            .into_iter()
104            .map(|s| s.to_string())
105            .collect::<Vec<CronSched>>()
106            .join(",")
107    };
108    cron_sched
109}

111/// Clamp values within the specified `min_val..=max_val` range
112///
113/// Returns `CronTime`
114fn clamp_cron_time_values(
115    cron_time: &[CronComponent],
116    min_val: u8,
117    max_val: u8,
118) -> CronTime {
119    cron_time
120        .iter()
121        .fold(BTreeSet::new(), |mut out, cron_component| {
122            out.insert(cron_component.clamp_inner(min_val, max_val));
123            out
124        })
125        .into_iter()
126        .collect()
127}
128
129/// Scan and merge components if they overlap.
130///
131/// Returns `CronTime`
132fn merge_cron_time_overlaps(cron_time: &CronTime) -> CronTime {
133    BTreeSet::from_iter(cron_time)
134        .iter()
135        .fold(Vec::new(), |mut out, &cron_component| {
136            let mut is_not_merged = true;
137            // For each item in the output vector, check if it overlaps with the current
138            // component. If it does, merge it with the current component, and set the
139            // `is_not_merged` flag to false.
140            for item in &mut out {
141                if let Some(merged_item) = item.merge(*cron_component) {
142                    is_not_merged = false;
143                    *item = merged_item;
144                }
145            }
146            if is_not_merged {
147                out.push(*cron_component);
148            }
149            out
150        })
151}

186impl CronComponent {
187    /// Represents all possible values, `"*"`.
188    const ALL_STR: &'static str = "*";
189    /// Maximum value for `Day`.
190    const MAX_DAY: u8 = 31;
191    /// Maximum value for `DayOfWeek`. Sunday.
192    const MAX_DOW: u8 = 7;
193    /// Maximum value for `Hour`.
194    const MAX_HOUR: u8 = 23;
195    /// Maximum value for `Minute`.
196    const MAX_MINUTE: u8 = 59;
197    /// Maximum value for `Month`.
198    const MAX_MONTH: u8 = 12;
199    /// Minimum value for `Day`.
200    const MIN_DAY: u8 = 1;
201    /// Minimum value for `DayOfWeek`. Monday.
202    const MIN_DOW: u8 = 1;
203    /// Minimum value for `Hour`.
204    const MIN_HOUR: u8 = 0;
205    /// Minimum value for `Minute`.
206    const MIN_MINUTE: u8 = 0;
207    /// Minimum value for `Month`.
208    const MIN_MONTH: u8 = 1;
209
210    /// Clamp inner values within the given range values. Returns `CronComponent`.
211    fn clamp_inner(
212        self,
213        first: u8,
214        last: u8,
215    ) -> Self {
216        /// Implement clamping inner values within the given range values.
217        fn clamp_val(
218            val: u8,
219            min_limit: u8,
220            max_limit: u8,
221        ) -> u8 {
222            min(max(val, min_limit), max_limit)
223        }
224
225        match self {
226            Self::All => self,
227            Self::At(when) => Self::At(clamp_val(when, first, last)),
228            Self::Range((a, b)) => {
229                // Clamp values.
230                let (c, d) = (clamp_val(a, first, last), clamp_val(b, first, last));
231                // Ensure lowest value is first.
232                let range = if c <= d { (c, d) } else { (d, c) };
233                // If the range is the set as clamping limits, return `All`.
234                if range == (first, last) {
235                    Self::All
236                } else if c == d {
237                    Self::At(c)
238                } else {
239                    // Return the range.
240                    Self::Range(range)
241                }
242            },
243        }
244    }
245
246    /// Merge two `CronComponent` values. Returns `Option<CronComponent>`.
247    ///
248    /// This method makes no checks to determine if the values are within
249    /// any limit.
250    fn merge(
251        self,
252        other: CronComponent,
253    ) -> Option<Self> {
254        match self {
255            Self::All => Some(self),
256            Self::At(when) => {
257                match other {
258                    Self::All => Some(Self::All),
259                    Self::At(w) if w == when => Some(self),
260                    Self::Range((a, b)) if (a..=b).contains(&when) => Some(other),
261                    _ => None,
262                }
263            },
264            Self::Range((first, last)) => {
265                match other {
266                    Self::All => Some(Self::All),
267                    Self::At(w) if (first..=last).contains(&w) => Some(self),
268                    Self::Range((a, b))
269                        if ((first..=last).contains(&a) || (first..=last).contains(&b))
270                            || ((a..=b).contains(&first) || (a..=b).contains(&last)) =>
271                    {
272                        Some(Self::Range((min(first, a), max(last, b))))
273                    },
274                    _ => None,
275                }
276            },
277        }
278    }
279}

281impl Display for CronComponent {
282    fn fmt(
283        &self,
284        f: &mut Formatter<'_>,
285    ) -> std::fmt::Result {
286        match self {
287            Self::All => write!(f, "{}", Self::ALL_STR),
288            Self::At(val) => write!(f, "{val}"),
289            Self::Range((start, end)) => write!(f, "{start}-{end}"),
290        }
291    }
292}

294impl PartialEq for CronComponent {
295    fn eq(
296        &self,
297        other: &Self,
298    ) -> bool {
299        match self {
300            Self::All => matches!(other, Self::All),
301            Self::At(when) => {
302                match other {
303                    Self::At(w) if w == when => true,
304                    Self::Range((a, b)) if (a..=b).contains(&when) => true,
305                    _ => false,
306                }
307            },
308            Self::Range((first, last)) => {
309                match other {
310                    Self::At(w) if first == w && last == w => true,
311                    Self::Range((a, b)) if first == a && last == b => true,
312                    _ => false,
313                }
314            },
315        }
316    }
317}

330/// Compare two `CronComponent`s.
331///
332/// `CronComponent`s are ordered in the following order, from greater to lesser:
333/// - `All`
334/// - `Range`
335/// - `At`
336impl Ord for CronComponent {
337    fn cmp(
338        &self,
339        other: &Self,
340    ) -> std::cmp::Ordering {
341        match self {
342            Self::All => {
343                match other {
344                    Self::All => std::cmp::Ordering::Equal,
345                    _ => std::cmp::Ordering::Greater,
346                }
347            },
348            Self::At(when) => {
349                match other {
350                    Self::At(w) => when.cmp(w),
351                    _ => std::cmp::Ordering::Less,
352                }
353            },
354            Self::Range((first, last)) => {
355                match other {
356                    Self::All => std::cmp::Ordering::Less,
357                    Self::At(_) => std::cmp::Ordering::Greater,
358                    Self::Range((start, end)) => first.cmp(start).then(last.cmp(end)),
359                }
360            },
361        }
362    }

451        // `All` is always greater than all other `CronComponent`s.
452        assert_eq!(CronComponent::All, CronComponent::All);
453        assert!(CronComponent::All > CronComponent::At(0));
454        assert!(CronComponent::All > CronComponent::Range((0, 0)));
