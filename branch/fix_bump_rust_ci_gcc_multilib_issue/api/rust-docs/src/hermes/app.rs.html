13/// Hermes App Name type
14#[derive(Debug, Clone, PartialEq, Eq, Hash)]
15pub(crate) struct ApplicationName(pub(crate) String);
16
17impl std::fmt::Display for ApplicationName {
18    fn fmt(
19        &self,
20        f: &mut std::fmt::Formatter<'_>,
21    ) -> std::fmt::Result {
22        self.0.fmt(f)
23    }
24}

26/// Hermes application
27pub(crate) struct Application {
28    /// Application name
29    name: ApplicationName,
30
31    /// WASM modules
32    indexed_modules: HashMap<ModuleId, Module>,
33
34    /// Application's `Vfs` instance
35    vfs: Arc<Vfs>,
36}
37
38impl Application {
39    /// Create a new Hermes app
40    #[must_use]
41    pub(crate) fn new(
42        app_name: String,
43        vfs: Vfs,
44        modules: Vec<Module>,
45    ) -> Self {
46        let indexed_modules = modules
47            .into_iter()
48            .map(|module| (module.id().clone(), module))
49            .collect();
50        Self {
51            name: ApplicationName(app_name),
52            indexed_modules,
53            vfs: Arc::new(vfs),
54        }
55    }
56
57    /// Get app name
58    pub(crate) fn name(&self) -> &ApplicationName {
59        &self.name
60    }
61
62    /// Get vfs
63    pub(crate) fn vfs(&self) -> &Vfs {
64        self.vfs.as_ref()
65    }
66
67    /// Dispatch event for all available modules.
68    pub(crate) fn dispatch_event(
69        &self,
70        event: &dyn HermesEventPayload,
71    ) -> anyhow::Result<()> {
72        for module in self.indexed_modules.values() {
73            module_dispatch_event(
74                module,
75                self.name.clone(),
76                module.id().clone(),
77                self.vfs.clone(),
78                event,
79            )?;
80        }
81        Ok(())
82    }
83
84    /// Dispatch event for the target module by the `module_id`.
85    pub(crate) fn dispatch_event_for_target_module(
86        &self,
87        module_id: ModuleId,
88        event: &dyn HermesEventPayload,
89    ) -> anyhow::Result<()> {
90        let module = self
91            .indexed_modules
92            .get(&module_id)
93            .ok_or(anyhow::anyhow!("Module {module_id} not found"))?;
94        module_dispatch_event(
95            module,
96            self.name.clone(),
97            module_id,
98            self.vfs.clone(),
99            event,
100        )
101    }
102}
103
104/// Dispatch event
105pub(crate) fn module_dispatch_event(
106    module: &Module,
107    app_name: ApplicationName,
108    module_id: ModuleId,
109    vfs: Arc<Vfs>,
110    event: &dyn HermesEventPayload,
111) -> anyhow::Result<()> {
112    let runtime_ctx = HermesRuntimeContext::new(
113        app_name,
114        module_id,
115        event.event_name().to_string(),
116        module.exec_counter(),
117        vfs,
118    );
119
120    // Advise Runtime Extensions of a new context
121    new_context(&runtime_ctx);
122
123    module.execute_event(event, runtime_ctx)?;
124    Ok(())
125}
