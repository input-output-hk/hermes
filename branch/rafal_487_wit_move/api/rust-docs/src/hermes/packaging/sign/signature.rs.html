19/// COSE signature object.
20#[derive(Debug, Clone, PartialEq)]
21pub(crate) struct Signature<T> {
22    /// Signatures collection.
23    cose_signatures: Vec<CoseSignature>,
24    /// Signature payload object.
25    payload: T,
26}
27
28impl<T: SignaturePayloadEncoding> Signature<T> {
29    /// Create new `Signature` object.
30    pub(crate) fn new(payload: T) -> Self {
31        Self {
32            cose_signatures: Vec::new(),
33            payload,
34        }
35    }
36
37    /// Get the payload object.
38    pub(crate) fn payload(&self) -> &T {
39        &self.payload
40    }
41
42    /// Add a new signature to the `CoseSign` object.
43    pub(crate) fn add_sign(
44        &mut self,
45        private_key: &PrivateKey,
46        certificate: &Certificate,
47    ) -> anyhow::Result<()> {
48        anyhow::ensure!(
49            private_key.public_key() == certificate.subject_public_key()?,
50            "Certificate subject public key doesn't associated with the signing private key."
51        );
52
53        self.add_cose_signature(private_key, certificate)?;
54
55        Ok(())
56    }
57
58    /// Verify the `Signature` object.
59    /// Validate all signatures, how they correspond to the certificate and public key.
60    /// Returns `Ok(())` if signature is valid.
61    ///
62    /// # Note:
63    /// Before verifying, all related to the added signatures should be added to the
64    /// certificate storage `certificate::storage::add_certificate()`.
65    pub(crate) fn verify(&self) -> anyhow::Result<()> {
66        let cose_sign = self.build_cose_sign()?;
67        anyhow::ensure!(!cose_sign.signatures.is_empty(), "Empty signatures list.");
68
69        let mut errors = Errors::new();
70        for (i, cose_signature) in cose_sign.signatures.iter().enumerate() {
71            Self::verify_cose_sign(&cose_sign, i, cose_signature)
72                .unwrap_or_else(errors.get_add_err_fn());
73        }
74        errors.return_result(())
75    }
76
77    /// Convert `Signature` object to CBOR decoded bytes.
78    pub(crate) fn to_bytes(&self) -> anyhow::Result<Vec<u8>> {
79        self.build_cose_sign()?
80            .to_vec()
81            .map_err(|e| anyhow::anyhow!(e.to_string()))
82    }
83
84    /// Create new `Signature` object from CBOR encoded bytes reader.
85    pub(crate) fn from_reader(mut reader: impl Read) -> anyhow::Result<Self> {
86        let mut bytes = Vec::new();
87        reader.read_to_end(&mut bytes)?;
88        Self::from_bytes(&bytes)
89    }
90
91    /// Create new `Signature` object from CBOR encoded bytes.
92    pub(crate) fn from_bytes(bytes: &[u8]) -> anyhow::Result<Self> {
93        let cose_sign = CoseSign::from_slice(bytes).map_err(|e| anyhow::anyhow!(e.to_string()))?;
94        Self::validate_cose_protected_header(&cose_sign)?;
95
96        let payload_bytes = cose_sign
97            .payload
98            .ok_or(anyhow::anyhow!("Missing signature payload."))?;
99
100        let json = serde_json::from_slice(payload_bytes.as_slice())?;
101        let payload = T::from_json(json)?;
102
103        Ok(Self {
104            cose_signatures: cose_sign.signatures,
105            payload,
106        })
107    }
108}
109
110impl<T: SignaturePayloadEncoding> Signature<T> {
111    /// Verify the `CoseSignature` object.
112    fn verify_cose_sign(
113        cose_sign: &CoseSign,
114        i: usize,
115        cose_signature: &CoseSignature,
116    ) -> anyhow::Result<()> {
117        let kid = &cose_signature.protected.header.key_id;
118
119        let cert_hash = Blake2b256::from_bytes(kid).map_err(|err| {
120            anyhow::anyhow!("Failed to decode signature `kid` value to `Blake2b256` hash. {err}",)
121        })?;
122        let cert = certificate::storage::get_certificate(&cert_hash).ok_or(anyhow::anyhow!(
123            "Cannot find certificate in the storage, cert hash `{}`.",
124            cert_hash.to_hex()
125        ))?;
126        let public_key = cert.subject_public_key()?;
127
128        cose_sign.verify_signature(i, &[], |signature_bytes, msg| {
129            public_key.verify(msg, signature_bytes)
130        })
131    }
132
133    /// Validate `CoseSign` protected header.
134    fn validate_cose_protected_header(cose_sign: &CoseSign) -> anyhow::Result<()> {
135        let header = &cose_sign.protected.header;
136        let expected_header = Self::build_cose_protected_header();
137
138        anyhow::ensure!(
139            header.alg == expected_header.alg,
140            "Invalid COSE signature protected header `alg` value."
141        );
142        anyhow::ensure!(
143            header.content_type == expected_header.content_type,
144            "Invalid COSE signature protected header `content_type` value."
145        );
146
147        Ok(())
148    }
149
150    /// Add a new signature to the `Self::cose_signatures` field.
151    fn add_cose_signature(
152        &mut self,
153        private_key: &PrivateKey,
154        certificate: &Certificate,
155    ) -> anyhow::Result<()> {
156        let empty_signature = Self::build_empty_cose_signature(certificate)?;
157
158        // check for duplicate
159        if self
160            .cose_signatures
161            .iter()
162            .any(|sign| sign.protected.header.key_id == empty_signature.protected.header.key_id)
163        {
164            return Ok(());
165        }
166
167        // sign with private key
168        let cose_sign = self
169            .prepare_cose_sign_builder()?
170            .add_created_signature(empty_signature, &[], |data| private_key.sign(data))
171            .build();
172        if let Some(new_signature) = cose_sign.signatures.into_iter().next() {
173            self.cose_signatures.push(new_signature);
174        }
175
176        Ok(())
177    }
178
179    /// Build empty `CoseSignature` object.
180    fn build_empty_cose_signature(certificate: &Certificate) -> anyhow::Result<CoseSignature> {
181        let protected_header = Self::build_cose_sign_protected_header(certificate)?;
182        let res = CoseSignatureBuilder::new()
183            .protected(protected_header)
184            .build();
185        Ok(res)
186    }
187
188    /// Return a protected header for `CoseSignatureBuilder`.
189    fn build_cose_sign_protected_header(certificate: &Certificate) -> anyhow::Result<Header> {
190        let certificate_hash_bytes = certificate.hash()?.to_bytes().to_vec();
191        let header = HeaderBuilder::new().key_id(certificate_hash_bytes).build();
192        Ok(header)
193    }
194
195    /// Prepare `CoseSignBuilder` object with necessary data.
196    fn prepare_cose_sign_builder(&self) -> anyhow::Result<CoseSignBuilder> {
197        let json = self.payload.to_json();
198        let payload_bytes = serde_json::to_vec(&json)?;
199        let protected_header = Self::build_cose_protected_header();
200
201        let builder = CoseSignBuilder::new()
202            .protected(protected_header)
203            .payload(payload_bytes);
204        Ok(builder)
205    }
206
207    /// Return a protected header for `CoseSignBuilder`.
208    fn build_cose_protected_header() -> Header {
209        HeaderBuilder::new()
210            .algorithm(iana::Algorithm::EdDSA)
211            .content_format(iana::CoapContentFormat::Json)
212            .build()
213    }
214
215    /// Build a `CoseSign` object.
216    fn build_cose_sign(&self) -> anyhow::Result<CoseSign> {
217        let builder = self.prepare_cose_sign_builder()?;
218
219        let res = self
220            .cose_signatures
221            .iter()
222            .fold(builder, |builder, signature| {
223                builder.add_signature(signature.clone())
224            })
225            .build();
226        Ok(res)
227    }
228}

230/// Signature payload encoding trait.
231/// Defines how to encode and decode signature payload object from JSON.
232pub(crate) trait SignaturePayloadEncoding {
233    /// Encode signature payload object to bytes.
234    fn to_json(&self) -> serde_json::Value;
235
236    /// Decode signature payload object from bytes.
237    fn from_json(json: serde_json::Value) -> anyhow::Result<Self>
238    where Self: Sized;
239}
240
241impl SignaturePayloadEncoding for serde_json::Value {
242    fn to_json(&self) -> serde_json::Value {
243        self.clone()
244    }
245
246    fn from_json(json: serde_json::Value) -> anyhow::Result<Self>
247    where Self: Sized {
248        Ok(json)
249    }
250}

259    #[test]
260    fn signature_serde_test() {
261        let payload = serde_json::json!({ "key": "value" });
262        let signature = Signature::new(payload);
263
264        let bytes = signature.to_bytes().unwrap();
265        let decoded_signature = Signature::<serde_json::Value>::from_bytes(&bytes).unwrap();
266        assert_eq!(signature, decoded_signature);
267
268        let cose_sign = CoseSign::from_slice(bytes.as_slice()).unwrap();
269        assert!(
270            Signature::<serde_json::Value>::from_bytes(&cose_sign.clone().to_vec().unwrap())
271                .is_ok()
272        );
273
274        let mut cose_sign_modified_alg = cose_sign.clone();
275        cose_sign_modified_alg.protected.original_data = None;
276        cose_sign_modified_alg.protected.header.alg = None;
277        assert!(Signature::<serde_json::Value>::from_bytes(
278            &cose_sign_modified_alg.clone().to_vec().unwrap()
279        )
280        .is_err());
281
282        cose_sign_modified_alg.protected.original_data = None;
283        cose_sign_modified_alg.protected.header.alg = Some(
284            coset::RegisteredLabelWithPrivate::Assigned(iana::Algorithm::ES256),
285        );
286        assert!(Signature::<serde_json::Value>::from_bytes(
287            &cose_sign_modified_alg.to_vec().unwrap()

345    #[test]
346    fn signature_verify_test() {
347        let payload = serde_json::json!({ "key": "value" });
348        let mut signature = Signature::new(payload);
349
350        let private_key = PrivateKey::from_str(&private_key_str()).unwrap();
351        let certificate = Certificate::from_str(&certificate_str()).unwrap();
352
353        assert!(
354            signature.verify().is_err(),
355            "Empty signature must be invalid."
356        );
357
358        signature.add_sign(&private_key, &certificate).unwrap();
359
360        assert!(
361            signature.verify().is_err(),
362            "Missing certificate in the storage."
363        );
364
365        certificate::storage::add_certificate(certificate).unwrap();
366        signature.verify().unwrap();
367
368        // corrupt signature
369        let bytes = signature.to_bytes().unwrap();
370        let mut cose_sign = CoseSign::from_slice(bytes.as_slice()).unwrap();
371        // change payload
372        cose_sign.payload = Some(
373            serde_json::json!("corrupted")
374                .to_string()
375                .as_bytes()
376                .to_vec(),
377        );
378        let signature =
379            Signature::<serde_json::Value>::from_bytes(&cose_sign.to_vec().unwrap()).unwrap();
380
381        assert!(signature.verify().is_err(), "Corrupted signature.");
382    }
