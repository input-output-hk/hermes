1//! Hermes Reactor implementation.
2
3use dashmap::{mapref::one::Ref, DashMap};
4use once_cell::sync::OnceCell;
5
6use crate::{
7    app::{Application, ApplicationName},
8    event::{self, queue::ExitLock},
9    runtime_extensions::hermes::init,
10};
11
12/// Global Hermes reactor state
13static REACTOR_STATE: OnceCell<Reactor> = OnceCell::new();
14
15/// Failed when reactor already been initialized.
16#[derive(thiserror::Error, Debug, Clone)]
17#[error("Reactor already been initialized.")]
18pub(crate) struct AlreadyInitializedError;
19
20/// Failed when event queue not been initialized.
21#[derive(thiserror::Error, Debug, Clone)]
22#[error("Reactor not been initialized. Call `HermesEventQueue::init` first.")]
23pub(crate) struct NotInitializedError;
24
25/// Hermes Reactor struct.
26/// This object orchestrates all Hermes apps within all core parts of the Hermes.
27struct Reactor {
28    /// Loaded hermes apps
29    apps: DashMap<ApplicationName, Application>,
30}
31
32/// Initialize Hermes Reactor.
33/// Setup and runs all necessary services.
34///
35/// [`ExitLock`] would contain shutdown information if awaited.
36///
37/// # Errors
38///
39/// - Queue already initialized.
40/// - Reactor already initialized.
41pub(crate) fn init() -> anyhow::Result<ExitLock> {
42    let exit_lock = event::queue::init()?;
43
44    REACTOR_STATE
45        .set(Reactor {
46            apps: DashMap::new(),
47        })
48        .map_err(|_| AlreadyInitializedError)?;
49
50    Ok(exit_lock)
51}
52
53/// Load Hermes application into the Hermes Reactor.
54///
55/// # Errors
56///
57/// - Reactor not initialized.
58/// - Cannot send initialization event to the application.
59pub(crate) fn load_app(app: Application) -> anyhow::Result<()> {
60    let reactor = REACTOR_STATE.get().ok_or(NotInitializedError)?;
61
62    let app_name = app.name().clone();
63    reactor.apps.insert(app_name.clone(), app);
64
65    init::emit_init_event(app_name)?;
66    Ok(())
67}
68
69/// Get Hermes application from the Hermes Reactor.
70pub(crate) fn get_app(
71    app_name: &ApplicationName
72) -> anyhow::Result<Ref<'_, ApplicationName, Application>> {
73    let reactor = REACTOR_STATE.get().ok_or(NotInitializedError)?;
74    reactor
75        .apps
76        .get(app_name)
77        .ok_or(anyhow::anyhow!("Application {app_name} not found"))
78}
79
80/// Get all available Hermes application names from the Hermes Reactor.
81pub(crate) fn get_all_app_names() -> anyhow::Result<Vec<ApplicationName>> {
82    let reactor = REACTOR_STATE.get().ok_or(NotInitializedError)?;
83    Ok(reactor.apps.iter().map(|val| val.key().clone()).collect())
84}
