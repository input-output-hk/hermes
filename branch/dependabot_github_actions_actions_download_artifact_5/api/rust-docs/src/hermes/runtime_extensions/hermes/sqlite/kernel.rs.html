23/// The default page size of `SQLite`.
24const PAGE_SIZE: u32 = 4_096;
25/// Max delay for sql query to retry.
26const MAX_DELAY_MS: u64 = 30000;
27/// Jitter to avoid all clients retrying in lock-step.
28const JITTER_MS: u64 = MAX_DELAY_MS * 2 / 10;
29
30/// Custom `SQLite` busy handler with exponential backoff and random jitter.
31///
32/// This handler is called whenever `SQLite` encounters a `SQLITE_BUSY` state
33/// (e.g. when the database file is locked by another transaction).
34///
35/// Behavior:
36/// - Uses exponential backoff: delays grow as 10, 20, 40, 80… ms.
37/// - Adds a random jitter (0–(`JITTER_MS` - 1) ms) to reduce lock-step retries across
38///   threads.
39/// - Clamps the maximum delay per attempt to `MAX_DELAY_MS` ms.
40/// - Always returns `1` to tell `SQLite` to retry, ensuring requests eventually complete.
41///
42/// Notes:
43/// - The `n` parameter is the number of times the handler has been invoked for the same
44///   lock.
45/// - `_data` is a user-provided pointer passed via `sqlite3_busy_handler`, unused here.
46extern "C" fn busy_handler(
47    _data: *mut c_void,
48    n: c_int,
49) -> c_int {
50    // add (`JITTER_MS` - 1) ms of randomness to avoid all clients retrying in lock-step
51    let jitter = random::<u64>() % JITTER_MS;
52    let exp: u32 = (n.saturating_sub(1)).try_into().unwrap_or(0);
53
54    // grows exponentially: 10, 20, 40, 80… milliseconds
55    let delay = 10u64.saturating_mul(2u64.saturating_pow(exp));
56
57    // add the random shift, but ensure the total wait is ≤ `MAX_DELAY_MS` ms
58    let wait_ms = delay.saturating_add(jitter).min(MAX_DELAY_MS);
59
60    std::thread::sleep(Duration::from_millis(wait_ms));
61    1
62}

64/// Configures the `SQLite` connection to use WAL mode with optimized settings for
65/// concurrency.
66///
67/// This function performs several critical setup steps:
68/// 1. Switches the journal mode to Write-Ahead Logging (WAL), which significantly
69///    improves concurrency by allowing readers to proceed while a writer is active.
70/// 2. Sets the synchronous level to NORMAL. This is a common performance optimization for
71///    WAL mode, reducing disk syncs for non-critical moments, which is generally safe
72///    against crashes but not power loss.
73/// 3. Configures the WAL auto-checkpoint threshold. A checkpoint is the process of
74///    transferring committed transactions from the WAL file back into the main database
75///    file.
76///
77/// # Parameters
78/// * `db_ptr`: A raw mutable pointer to an open `SQLite` database connection.
79///
80/// # Returns
81/// * `Ok(())` on success.
82/// * `Err(Errno)` if any of the configuration steps fail.
83fn enable_wal_mode(db_ptr: *mut sqlite3) -> Result<(), Errno> {
84    let pragmas = "PRAGMA journal_mode=WAL; PRAGMA synchronous=NORMAL;";
85    let c_pragmas = std::ffi::CString::new(pragmas).map_err(|_| Errno::ConvertingCString)?;
86
87    let rc = unsafe {
88        sqlite3_exec(
89            db_ptr,
90            c_pragmas.as_ptr(),
91            None,
92            std::ptr::null_mut(),
93            std::ptr::null_mut(),
94        )
95    };
96    if rc != SQLITE_OK {
97        return Err(Errno::Sqlite(rc));
98    }
99
100    // Set the WAL auto-checkpoint threshold to the default value of 1000 pages.
101    // This means a checkpoint will be triggered automatically when the WAL file
102    // grows to about 4MB (1000 pages * 4KB/page).
103    let rc = unsafe { sqlite3_wal_autocheckpoint(db_ptr, 1000) };
104    if rc != SQLITE_OK {
105        return Err(Errno::Sqlite(rc));
106    }
107
108    Ok(())
109}

111/// Opens a connection to a new or existing `SQLite` database.
112pub(super) fn open(
113    readonly: bool,
114    memory: bool,
115    app_name: ApplicationName,
116) -> Result<*mut sqlite3, Errno> {
117    let mut db_ptr: *mut sqlite3 = std::ptr::null_mut();
118
119    let (db_path, config) = if memory {
120        let in_memory_config =
121            get_app_in_memory_sqlite_db_cfg(app_name).ok_or(Errno::InvalidInMemoryConfig)?;
122
123        (":memory:".into(), in_memory_config)
124    } else {
125        let persistent_config =
126            get_app_persistent_sqlite_db_cfg(app_name).ok_or(Errno::InvalidPersistentConfig)?;
127
128        let db_name = persistent_config
129            .db_file
130            .clone()
131            .ok_or(Errno::MissingDatabaseNameForPersistentConfig)?;
132
133        (db_name, persistent_config)
134    };
135
136    // Setting SQLITE_OPEN_NOMUTEX is enough, without setting env var, since
137    // by default SQLITE_THREADSAFE=1 is used, and doc says:
138    //
139    // If single-thread mode has not been selected at compile-time or start-time,
140    // then individual database connections can be created as either multi-thread or
141    // serialized.
142    let flags = if readonly {
143        SQLITE_OPEN_READONLY
144    } else {
145        SQLITE_OPEN_CREATE | SQLITE_OPEN_READWRITE
146    } | SQLITE_OPEN_NOMUTEX;
147
148    let c_path =
149        CString::new(db_path.to_string_lossy().as_bytes()).map_err(|_| Errno::ConvertingCString)?;
150
151    let rc = unsafe { sqlite3_open_v2(c_path.as_ptr(), &raw mut db_ptr, flags, std::ptr::null()) };
152
153    if rc != SQLITE_OK {
154        return Err(Errno::Sqlite(rc));
155    } else if db_ptr.is_null() {
156        return Err(Errno::FailedOpeningDatabase);
157    }
158
159    let rc = unsafe { sqlite3_busy_handler(db_ptr, Some(busy_handler), std::ptr::null_mut()) };
160    if rc != SQLITE_OK {
161        return Err(Errno::Sqlite(rc));
162    }
163
164    if !memory && !readonly {
165        enable_wal_mode(db_ptr)?;
166    }
167
168    // config database size limitation
169    let rc = if memory {
170        let size_limit = i64::from(config.max_db_size);
171
172        unsafe { sqlite3_soft_heap_limit64(size_limit) };
173
174        SQLITE_OK
175    } else {
176        let page_size = config.max_db_size / PAGE_SIZE;
177        let pragma_stmt = format!("PRAGMA max_page_count = {page_size}");
178
179        let c_pragma_stmt =
180            std::ffi::CString::new(pragma_stmt).map_err(|_| Errno::ConvertingCString)?;
181
182        unsafe {
183            sqlite3_exec(
184                db_ptr,
185                c_pragma_stmt.as_ptr(),
186                None,
187                std::ptr::null_mut(),
188                std::ptr::null_mut(),
189            )
190        }
191    };
192
193    if rc != SQLITE_OK {
194        return Err(Errno::FailedSettingDatabaseSize);
195    }
196
197    Ok(db_ptr)
198}
