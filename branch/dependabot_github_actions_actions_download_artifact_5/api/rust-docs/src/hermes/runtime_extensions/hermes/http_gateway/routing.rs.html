49/// Main HTTP request router that processes incoming requests by hostname
50///
51/// Takes an HTTP request and routes it to the appropriate handler based on the
52/// hostname in the Host header. Each hostname corresponds to a different application.
53///
54/// ## Process:
55/// 1. Generates a unique request ID for tracking
56/// 2. Validates the hostname against allowed hosts
57/// 3. Routes valid requests to the Hermes application handler
58/// 4. Tracks connection state in the connection manager
59///
60///
61/// ## Example:
62/// A request with `Host: app.hermes.local` gets routed to the "app" application
63/// if "hermes.local" is in the valid hosts list.
64pub(crate) async fn router(
65    req: Request<Incoming>,
66    connection_manager: ConnectionManager,
67    ip: SocketAddr,
68    config: Config,
69) -> anyhow::Result<Response<Full<Bytes>>> {
70    let unique_request_id = EventUID(rusty_ulid::generate_ulid_string());
71    let client_ip = ClientIPAddr(ip);
72
73    // Register connection start
74    connection_manager.insert(
75        unique_request_id.clone(),
76        (client_ip.clone(), Processed(false), LiveConnection(true)),
77    );
78
79    #[cfg(debug_assertions)]
80    debug!("connection manager {:?}", connection_manager);
81
82    // Parse the Host header to extract the application name and resolved hostname
83    // This determines which application should handle the request
84    // For example: "app.hermes.local" -> app_name="app", resolved_host="hermes.local"
85    let (app_name, resolved_host) = host_resolver(req.headers())?;
86
87    // Check if the resolved hostname is in our list of valid/allowed hosts
88    // This is a security measure to prevent routing to unauthorized applications
89    let response = if config
90        .valid_hosts
91        .iter()
92        .any(|host| host.0 == resolved_host.0)
93    {
94        // If hostname is valid, route the request to the Hermes WASM runtime
95        // The app_name determines which specific application will handle it
96        route_to_hermes(req, app_name.clone()).await?
97    } else {
98        // If hostname is not in the valid list, reject with an error response
99        // This prevents potential security issues from unauthorized hosts
100        return error_response("Invalid hostname");
101    };
102
103    // Mark connection as processed
104    connection_manager.insert(
105        unique_request_id,
106        (client_ip, Processed(true), LiveConnection(false)),
107    );
108
109    #[cfg(debug_assertions)]
110    debug!("connection manager {connection_manager} app {app_name}");
111
112    Ok(response)
113}

133/// Extractor that resolves the hostname of the request.
134/// Hostname is resolved through the Host header
135pub(crate) fn host_resolver(headers: &HeaderMap) -> anyhow::Result<(ApplicationName, Hostname)> {
136    let host = headers
137        .get("Host")
138        .and_then(|h| h.to_str().ok())
139        .ok_or_else(|| anyhow!("Missing or invalid Host header"))?;
140
141    // Strip port if present (e.g., "app.hermes.local:5000" -> "app.hermes.local")
142    let host_without_port = host.split(':').next().unwrap_or(host);
143
144    // Parse <app.name>.hermes.local format
145    let (app, hostname) = host_without_port
146        .split_once('.')
147        .ok_or_else(|| anyhow!("Malformed Host header: expected format 'app.domain'"))?;
148
149    Ok((
150        ApplicationName(app.to_owned()),
151        Hostname(hostname.to_owned()),
152    ))
153}

155/// Main HTTP request router that processes incoming requests and delegates to appropriate
156/// handlers
157///
158/// This function serves as the central routing hub, taking validated HTTP requests and
159/// directing them to either WebAssembly modules or static file handlers based on the
160/// request path and routing rules.
161async fn route_to_hermes(
162    req: Request<Incoming>,
163    app_name: ApplicationName,
164) -> anyhow::Result<Response<Full<Bytes>>> {
165    // Extract the URI for route analysis - this contains path and query parameters
166    let uri = req.uri().to_owned();
167
168    // Analyze the request path and determine which type of handler should process it
169    // This applies our routing rules to classify the request appropriately
170    let route_type = determine_route(&uri)?;
171
172    // Delegate to the appropriate handler based on route classification
173    match route_type {
174        // API endpoints need WebAssembly module processing
175        // These requests go through the event queue to WASM components
176        RouteType::WebAssembly(path) => handle_webasm_request(req, path).await,
177        // Static files are served directly from the virtual file system
178        RouteType::StaticFile(path) => serve_static_web_content(&path, &app_name),
179    }
180}

182/// Route classification for incoming HTTP requests
183///
184/// This enum categorizes requests into two main types that require different handling:
185/// - API requests that need WebAssembly module processing
186/// - Static file requests that serve assets directly from the filesystem
187enum RouteType {
188    /// API endpoints that should be processed by WebAssembly modules
189    /// Contains the full path including query parameters for forwarding
190    WebAssembly(String),
191
192    /// Static file requests that serve assets (HTML, CSS, JS, images, etc.)
193    /// Contains the normalized file path for filesystem lookup
194    StaticFile(String),
195}
196
197/// Analyzes an HTTP request URI and determines the appropriate handler type
198///
199/// This function implements the core routing logic by examining the request path
200/// and applying routing rules to classify the request type.
201fn determine_route(uri: &hyper::Uri) -> anyhow::Result<RouteType> {
202    // Extract the full path including query parameters for accurate forwarding
203    // This preserves important data like ?asat=SLOT:95022059 that WASM modules need
204    let path = uri
205        .path_and_query()
206        .map_or(uri.path(), hyper::http::uri::PathAndQuery::as_str);
207
208    // Check if this is an API endpoint that needs WebAssembly processing
209    // API routes are identified by the /api prefix (exact match or with additional path)
210    if uri.path() == WEBASM_ROUTE || uri.path().starts_with(&format!("{WEBASM_ROUTE}/")) {
211        Ok(RouteType::WebAssembly(path.to_string()))
212    }
213    // Check if this is a valid static file path
214    // Uses regex validation to ensure path safety and prevent directory traversal
215    else if is_valid_path(uri.path()).is_ok() {
216        Ok(RouteType::StaticFile(path.to_string()))
217    }
218    // Reject invalid or potentially dangerous paths
219    else {
220        Err(anyhow!("Invalid route: {}", uri.path()))
221    }
222}

224/// Processes HTTP requests for WebAssembly module execution
225///
226/// Converts incoming HTTP requests into WASM-compatible format and handles
227/// the async communication with WebAssembly modules.
228///
229/// ## Returns:
230/// HTTP response from the WebAssembly module
231async fn handle_webasm_request(
232    req: Request<Incoming>,
233    path: String,
234) -> anyhow::Result<Response<Full<Bytes>>> {
235    let (lambda_send, lambda_recv_answer) = channel();
236    let method = req.method().to_string();
237
238    // Convert headers to multi-value format
239    let headers: HeadersKV = req
240        .headers()
241        .iter()
242        .map(|(name, value)| {
243            let key = name.to_string();
244            let values = vec![value.to_str().unwrap_or_default().to_string()];
245            (key, values)
246        })
247        .collect();
248
249    let (_parts, body) = req.into_parts();
250    let body_bytes = body.collect().await?.to_bytes();
251
252    compose_http_event(
253        method,
254        headers,
255        body_bytes,
256        path,
257        lambda_send,
258        &lambda_recv_answer,
259    )
260}

262/// Compose http event and send to global queue, await queue response and relay back to
263/// waiting receiver channel for HTTP response
264fn compose_http_event<B>(
265    method: String,
266    headers: HeadersKV,
267    body: Bytes,
268    path: String,
269    sender: Sender<HTTPEventMsg>,
270    receiver: &Receiver<HTTPEventMsg>,
271) -> anyhow::Result<Response<B>>
272where
273    B: Body + From<String>,
274{
275    let on_http_event = HTTPEvent {
276        headers,
277        method,
278        path,
279        body,
280        sender,
281    };
282    let event = HermesEvent::new(on_http_event, TargetApp::All, TargetModule::All);
283
284    crate::event::queue::send(event)?;
285
286    let timeout = Duration::from_secs(EVENT_TIMEOUT);
287    match receiver.recv_timeout(timeout)? {
288        HTTPEventMsg::HttpEventResponse(resp) => {
289            let body = serde_json::to_string(&resp)?.into();
290            Ok(Response::new(body))
291        },
292        HTTPEventMsg::HTTPEventReceiver => error_response("HTTP event message error"),
293    }
294}

296/// Compiled regex for validating static file request paths
297///
298/// Lazily compiled from `VALID_PATH` pattern for efficient path matching.
299/// Returns `None` if the regex pattern is invalid.
300static VALID_PATH_REGEX: LazyLock<Option<Regex>> = LazyLock::new(|| Regex::new(VALID_PATH).ok());
301
302/// Validates HTTP request paths for static file serving
303///
304/// Uses regex pattern matching to ensure paths are safe for VFS access.
305/// Allows root path "/" and absolute paths "/path/to/resource".
306///
307/// Primary security relies on VFS sandboxing; this provides basic validation.
308fn is_valid_path(path: &str) -> anyhow::Result<()> {
309    VALID_PATH_REGEX
310        .as_ref()
311        .ok_or_else(|| anyhow::anyhow!("Path validation unavailable: invalid regex pattern"))?
312        .is_match(path)
313        .then_some(())
314        .ok_or_else(|| anyhow::anyhow!("Invalid path format: {path:?}"))
315}

317/// Serves static web assets for Flutter applications with comprehensive error handling
318///
319/// This is the primary function for handling static file requests in the Hermes web
320/// server. It manages the complete lifecycle of serving web assets from the Virtual File
321/// System (VFS), including MIME type detection, caching headers, security policies, and
322/// fallback handling.
323///
324/// ## Core Functionality:
325///
326/// ### 1. Path Resolution & VFS Access
327/// - Converts HTTP request paths to VFS file paths using `resolve_static_file_path()`
328/// - Accesses the application's Virtual File System through the reactor
329/// - Handles both direct file requests and Flutter routing scenarios
330fn serve_static_web_content<B>(
331    path: &str,
332    app_name: &ApplicationName,
333) -> anyhow::Result<Response<B>>
334where
335    B: Body + From<Vec<u8>>,
336{
337    let file_path = resolve_static_file_path(path);
338    let app = reactor::get_app(app_name)?;
339
340    app.vfs()
341        .read(&file_path)
342        .and_then(|contents| serve_existing_asset(contents, &file_path))
343        .or_else(|_| handle_missing_asset(&file_path, &app))
344}

349/// Document root directory in the VFS where static web assets are stored
350const DOCUMENT_ROOT: &str = "www";

352/// Resolves incoming HTTP request paths to actual file paths within the Virtual File
353/// System (VFS)
354///
355/// This function performs path normalization and translation for serving static files in
356/// a Flutter web application. It implements the standard web server convention where the
357/// document root maps to a specific VFS directory.
358///
359/// ## Path Resolution Rules:
360///
361/// ### Root Path Handling:
362/// - **Input**: `"/"` (root/index request)
363/// - **Output**: `"www/index.html"`
364/// - **Purpose**: Serves the main Flutter application entry point
365/// - **Example**: `GET /` → serves `www/index.html` from VFS
366///
367/// ### Static File Path Translation:
368/// - **Input**: `"/assets/app.js"` (any non-root path)
369/// - **Process**: Remove leading `/`, prepend `www/`
370/// - **Output**: `"www/assets/app.js"`
371/// - **Purpose**: Maps URL paths to VFS file locations
372///
373/// ## VFS Structure Context:
374/// VFS Root
375/// |-- www/                    <- Document root directory
376///     |-- index.html         <- Main application entry point
377///     |-- flutter.js         <- Flutter framework loader
378///     |-- `flutter_service_worker.js`  <- Service worker
379///     |-- assets/            <- Static assets directory
380///         |-- fonts/         <- Font files
381///         |-- images/        <- Image assets
382///         |-- packages/      <- Dart package assets
383///
384/// ## Security Considerations:
385/// - **Path traversal protection**: This function doesn't validate against `../` attacks
386///   (assumes upstream validation by HTTP router/framework)
387/// - **VFS isolation**: All resolved paths are contained within the `www/` directory
388/// - **No filesystem access**: Works purely with VFS paths, not real filesystem paths
389fn resolve_static_file_path(path: &str) -> String {
390    if path == "/" {
391        DEFAULT_INDEX_PATH.to_string()
392    } else {
393        let clean_path = path.strip_prefix('/').unwrap_or(path);
394        format!("{DOCUMENT_ROOT}/{clean_path}")
395    }
396}

398/// Serves an existing asset file with appropriate headers
399fn serve_existing_asset<B>(
400    file_contents: Vec<u8>,
401    file_path: &str,
402) -> anyhow::Result<Response<B>>
403where
404    B: Body + From<Vec<u8>>,
405{
406    let mut response = Response::new(file_contents.into());
407
408    if let Some(extension) = get_file_extension(file_path) {
409        let content_type = get_flutter_content_type(extension);
410
411        response
412            .headers_mut()
413            .insert("Content-Type", content_type.parse()?);
414    }
415
416    add_security_headers(response)
417}

419/// Handles missing asset files with appropriate fallback logic
420fn handle_missing_asset<B>(
421    file_path: &str,
422    app: &Application,
423) -> anyhow::Result<Response<B>>
424where
425    B: Body + From<Vec<u8>>,
426{
427    // Critical assets should return 404
428    if is_critical_asset(file_path) {
429        error!("Critical asset missing: {}", file_path);
430        return Ok(Response::builder()
431            .status(StatusCode::NOT_FOUND)
432            .body("Asset not found".as_bytes().to_vec().into())?);
433    }
434
435    // Fall back to index.html for navigation routes
436    serve_index_html_fallback(file_path, app)
437}

440fn is_critical_asset(file_path: &str) -> bool {
441    matches!(
442        get_file_extension(file_path),
443        Some("js" | "wasm" | "json" | "css" | "map")
444    )

447/// Adds security headers to HTTP responses for Flutter web applications
448///
449/// This function implements Cross-Origin Isolation by setting two critical security
450/// headers that work together to create a secure browsing context.
451///
452/// ## Headers Added:
453///
454/// ### Cross-Origin-Opener-Policy (COOP): "same-origin"
455/// - **Purpose**: Isolates the browsing context from cross-origin windows
456/// - **Effect**: Prevents other origins from accessing this window object
457/// - **Security**: Protects against certain types of cross-origin attacks
458/// - **Compatibility**: Allows same-origin popups/windows to communicate normally
459///
460/// ### Cross-Origin-Embedder-Policy (COEP): "require-corp"
461/// - **Purpose**: Requires all embedded resources to explicitly opt-in to cross-origin
462///   loading
463/// - **Effect**: Blocks cross-origin resources without proper CORS or
464///   Cross-Origin-Resource-Policy headers
465/// - **Security**: Prevents malicious resource injection from untrusted origins
466/// - **Requirement**: All cross-origin assets need `crossorigin` attribute or CORP
467///   headers
468fn add_security_headers<B>(mut response: Response<B>) -> anyhow::Result<Response<B>>
469where B: Body {
470    let headers = response.headers_mut();
471
472    // Enable Cross-Origin Isolation for advanced web features
473    headers.insert(
474        "Cross-Origin-Opener-Policy",
475        "same-origin"
476            .parse()
477            .map_err(|e| anyhow!("Invalid COOP header value: {}", e))?,
478    );
479    headers.insert(
480        "Cross-Origin-Embedder-Policy",
481        "require-corp"
482            .parse()
483            .map_err(|e| anyhow!("Invalid COEP header value: {}", e))?,
484    );
485
486    Ok(response)
487}

494/// Returns the appropriate MIME type for Flutter web assets based on file extension
495///
496/// Maps common web asset file extensions to their corresponding MIME types
497/// for proper browser handling. Defaults to `application/octet-stream` for
498/// unknown extensions.
499fn get_flutter_content_type(extension: &str) -> &'static str {
500    match extension {
501        "html" => "text/html",
502        "js" => "application/javascript",
503        "wasm" => "application/wasm",
504        "css" => "text/css",
505        "json" => "application/json",
506        "png" => "image/png",
507        "jpg" | "jpeg" => "image/jpeg",
508        "svg" => "image/svg+xml",
509        "ico" => "image/x-icon",
510        "woff" | "woff2" => "font/woff2",
511        "ttf" => "font/ttf",
512        "otf" => "font/otf",
513        _ => "application/octet-stream",
514    }
515}

517/// HTTP Content-Type header value for HTML documents
518///
519/// Used when serving HTML files, particularly the index.html fallback for Flutter
520/// single-page applications. This MIME type tells browsers to parse the content
521/// as HTML markup.
522const CONTENT_TYPE_HTML: &str = "text/html";
523
524/// Cache-Control header directive to prevent caching
525///
526/// Instructs browsers and intermediate caches to:
527/// - `no-cache`: Always revalidate with server before using cached content
528/// - `no-store`: Never store the response in any cache
529/// - `must-revalidate`: Require revalidation of stale cache entries
530///
531/// Used for index.html to ensure users always get the latest version of the
532/// Flutter application, preventing issues with cached outdated app shells.
533const NO_CACHE_DIRECTIVE: &str = "no-cache, no-store, must-revalidate";
534
535/// Serves index.html as fallback for navigation routes
536fn serve_index_html_fallback<B>(
537    file_path: &str,
538    app: &Application,
539) -> anyhow::Result<Response<B>>
540where
541    B: Body + From<Vec<u8>>,
542{
543    match file_path {
544        "www/index.html" => not_found(),
545        _ => {
546            match app.vfs().read("www/index.html") {
547                Ok(index_contents) => {
548                    let mut response = Response::new(index_contents.into());
549                    response
550                        .headers_mut()
551                        .insert("Content-Type", CONTENT_TYPE_HTML.parse()?);
552                    response
553                        .headers_mut()
554                        .insert("Cache-Control", NO_CACHE_DIRECTIVE.parse()?);
555                    response = add_security_headers(response)?;
556                    Ok(response)
557                },
558                Err(_) => not_found(),
559            }
560        },
561    }
562}

571    #[test]
572    fn test_valid_paths_regex() {
573        // ^ and $: Match the entire string/line
574        // (/[a-zA-Z0-9-_]+)+: One or more directories, starting with slash, separated by
575        // slashes; each directory must consist of one or more characters of your charset.
576        // (...)+|/: Explicitly allow just a single slash
577        let regex = Regex::new(VALID_PATH).unwrap();
578
579        // valid
580        let example_one = "/abc/def";
581        let example_two = "/hello_1/world";
582        let example_three = "/three/directories/abc";
583        let example_four = "/";
584        let valid_path = vec![example_one, example_two, example_three, example_four];
585
586        for valid in valid_path {
587            if let Some(captures) = regex.captures(valid) {
588                assert_eq!(captures.get(0).unwrap().as_str(), valid);
589            }
590
591            assert!(is_valid_path(valid).is_ok());
592        }
593
594        // invalid
595        let example_one = "/abc/def/";
596        let example_two = "/abc//def";
597        let example_three = "//";
598        let example_four = "abc/def";
599        let example_five = "/abc/def/file.txt";
600        let invalids = vec![
601            example_one,
602            example_two,
603            example_three,
604            example_four,
605            example_five,
606        ];
607
608        for invalid in invalids {
609            if let Some(captures) = regex.captures(invalid) {
610                assert!(captures.len() == 0);
611            }
612
613            assert!(is_valid_path(invalid).is_err());
614        }
615    }
