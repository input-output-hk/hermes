1//! Permissions state management of the Hermes virtual file system.
2
3use std::sync::{
4    atomic::{AtomicBool, Ordering},
5    Arc,
6};
7
8use dashmap::DashMap;
9
10use crate::utils::parse_path;

12/// Permission level type.
13#[derive(Default, Clone, Copy, PartialEq, Eq)]
14#[cfg_attr(debug_assertions, derive(Debug))]
15pub(crate) enum PermissionLevel {
16    /// Read only permission level.
17    Read,
18    /// Read and write permission level.
19    #[default]
20    ReadAndWrite,
21}
22
23impl From<PermissionLevel> for bool {
24    fn from(level: PermissionLevel) -> Self {
25        match level {
26            PermissionLevel::Read => false,
27            PermissionLevel::ReadAndWrite => true,
28        }
29    }
30}
31
32impl From<bool> for PermissionLevel {
33    fn from(level: bool) -> Self {
34        if level {
35            PermissionLevel::ReadAndWrite
36        } else {
37            PermissionLevel::Read
38        }
39    }
40}
41
42/// VFS permissions state stored in the radix tree structure, where the
43/// each node relates to a single path element with the permission level.
44/// `PermissionLevel::ReadAndWrite` is a default permission level so if permission was not
45/// defined for the path explicitly `PermissionsState` will return the
46/// `PermissionLevel::ReadAndWrite` permission level for the asked path.
47pub(crate) struct PermissionsState {
48    /// Tree's root node.
49    root: PermissionNodeRef,
50}
51
52/// `PermissionsTree` node type.
53struct PermissionNode {
54    /// Node permission level.
55    permission: AtomicBool,
56    /// Node childs.
57    childs: DashMap<String, PermissionNodeRef>,
58}
59
60impl Default for PermissionNode {
61    fn default() -> Self {
62        Self {
63            permission: AtomicBool::new(PermissionLevel::default().into()),
64            childs: DashMap::new(),
65        }
66    }
67}
68
69/// Convenient type of the referenced `PermissionNode`.
70type PermissionNodeRef = Arc<PermissionNode>;
71
72impl PermissionsState {
73    /// Creates a new `PermissionsTree` instance with the root node which relates to the
74    /// "/" path with the `PermissionLevel::ReadAndWrite` default permission level.
75    pub(crate) fn new() -> Self {
76        let root = PermissionNodeRef::default();
77        Self { root }
78    }
79
80    /// Adds a new path to the `PermissionsTree` with the provided permission level.
81    pub(crate) fn add_permission(
82        &mut self,
83        path: &str,
84        permission: PermissionLevel,
85    ) {
86        let path_elements = parse_path(path);
87
88        let mut walk = self.root.clone();
89        for path_element in path_elements {
90            let node = walk.clone();
91            if let Some(child_node) = node.childs.get(&path_element) {
92                walk = child_node.value().clone();
93            } else {
94                let new_node = PermissionNodeRef::default();
95                node.childs.insert(path_element, new_node.clone());
96                walk = new_node;
97            };
98        }
99        // Update the last node with the provided permission
100        walk.permission.store(permission.into(), Ordering::Release);
101    }
102
103    /// Gets the permission level for the provided path.
104    pub(crate) fn get_permission(
105        &self,
106        path: &str,
107    ) -> PermissionLevel {
108        let mut permission: PermissionLevel = self.root.permission.load(Ordering::Acquire).into();
109
110        let path_elements = parse_path(path);
111
112        let mut walk = self.root.clone();
113        for path_element in path_elements {
114            let node = walk.clone();
115            if let Some(child_node) = node.childs.get(&path_element) {
116                permission = child_node.permission.load(Ordering::Acquire).into();
117
118                if permission == PermissionLevel::Read {
119                    break;
120                }
121
122                walk = child_node.clone();
123            } else {
124                break;
125            };
126        }
127
128        permission
129    }
130}

132#[cfg(all(test, debug_assertions))]
133mod tests {
134    use super::*;
135
136    #[test]
137    fn permission_level_test() {
138        assert!(!Into::<bool>::into(PermissionLevel::Read));
139        assert!(Into::<bool>::into(PermissionLevel::ReadAndWrite));
140
141        assert_eq!(Into::<PermissionLevel>::into(false), PermissionLevel::Read);
142        assert_eq!(
143            Into::<PermissionLevel>::into(true),
144            PermissionLevel::ReadAndWrite
145        );
146    }
147
148    #[test]
149    fn permission_tree_test() {
150        let mut tree = PermissionsState::new();
151
152        assert_eq!(tree.get_permission(""), PermissionLevel::ReadAndWrite);
153        assert_eq!(tree.get_permission("/a/b"), PermissionLevel::ReadAndWrite);
154
155        tree.add_permission("/a/b", PermissionLevel::Read);
156        tree.add_permission("c/b", PermissionLevel::Read);
157
158        assert_eq!(tree.get_permission("a"), PermissionLevel::ReadAndWrite);
159        assert_eq!(tree.get_permission("a/b"), PermissionLevel::Read);
160        assert_eq!(tree.get_permission("c"), PermissionLevel::ReadAndWrite);
161        assert_eq!(tree.get_permission("c/b"), PermissionLevel::Read);
162
163        tree.add_permission("/a/b", PermissionLevel::ReadAndWrite);
164        tree.add_permission("a", PermissionLevel::Read);
165
166        assert_eq!(tree.get_permission("a"), PermissionLevel::Read);
167        assert_eq!(tree.get_permission("a/b"), PermissionLevel::Read);
168    }
169}
