14/// Stores application data into parameters of the original SQL.
15pub(crate) fn bind(
16    stmt_ptr: *mut sqlite3_stmt,
17    index: i32,
18    value: Value,
19) -> Result<(), Errno> {
20    let rc = unsafe {
21        match value {
22            Value::Blob(value) => {
23                let bytes_len = i32::try_from(value.len()).map_err(|_| Errno::ConvertingNumeric)?;
24
25                sqlite3_bind_blob(
26                    stmt_ptr,
27                    index,
28                    value.as_ptr().cast::<std::ffi::c_void>(),
29                    bytes_len,
30                    SQLITE_TRANSIENT(),
31                )
32            },
33            Value::Double(value) => sqlite3_bind_double(stmt_ptr, index, value),
34            Value::Int32(value) => sqlite3_bind_int(stmt_ptr, index, value),
35            Value::Int64(value) => sqlite3_bind_int64(stmt_ptr, index, value),
36            Value::Null => sqlite3_bind_null(stmt_ptr, index),
37            Value::Text(value) => {
38                let c_value =
39                    std::ffi::CString::new(value).map_err(|_| Errno::ConvertingCString)?;
40
41                let n_byte = c_value.as_bytes_with_nul().len();
42                let n_byte = i32::try_from(n_byte).map_err(|_| Errno::ConvertingNumeric)?;
43
44                sqlite3_bind_text(
45                    stmt_ptr,
46                    index,
47                    c_value.as_ptr(),
48                    n_byte,
49                    SQLITE_TRANSIENT(),
50                )
51            },
52        }
53    };
54
55    if rc == SQLITE_OK {
56        Ok(())
57    } else {
58        Err(Errno::Sqlite(rc))
59    }
60}

62/// Advances a statement to the next result row or to completion.
63pub(crate) fn step(stmt_ptr: *mut sqlite3_stmt) -> Result<(), Errno> {
64    let rc = unsafe { sqlite3_step(stmt_ptr) };
65
66    if rc != SQLITE_DONE && rc != SQLITE_ROW {
67        Err(Errno::Sqlite(rc))
68    } else {
69        Ok(())
70    }
71}

73/// Returns information about a single column of the current result row of a query.
74pub(crate) fn column(
75    stmt_ptr: *mut sqlite3_stmt,
76    index: i32,
77) -> Result<Value, Errno> {
78    let value = unsafe {
79        let column_type = sqlite3_column_type(stmt_ptr, index);
80
81        match column_type {
82            SQLITE_BLOB => {
83                let blob_ptr = sqlite3_column_blob(stmt_ptr, index);
84                let blob_len = sqlite3_column_bytes(stmt_ptr, index);
85                let blob_len = usize::try_from(blob_len).map_err(|_| Errno::ConvertingNumeric)?;
86                let blob_slice = std::slice::from_raw_parts(blob_ptr.cast::<u8>(), blob_len);
87                Value::Blob(blob_slice.to_vec())
88            },
89            SQLITE_FLOAT => Value::Double(sqlite3_column_double(stmt_ptr, index)),
90            SQLITE_INTEGER => {
91                let int_value = sqlite3_column_int64(stmt_ptr, index);
92                if let Ok(int_value) = i32::try_from(int_value) {
93                    Value::Int32(int_value)
94                } else {
95                    Value::Int64(int_value)
96                }
97            },
98            SQLITE_NULL => Value::Null,
99            SQLITE_TEXT => {
100                let text_ptr = sqlite3_column_text(stmt_ptr, index);
101                let text_slice = std::ffi::CStr::from_ptr(text_ptr.cast::<c_char>());
102
103                let text_string = text_slice
104                    .to_str()
105                    .map(String::from)
106                    .map_err(|_| Errno::ConvertingCString)?;
107
108                Value::Text(text_string)
109            },
110            _ => return Err(Errno::UnknownColumnType),
111        }
112    };
113
114    Ok(value)
115}

117/// Destroys a prepared statement object. If the most recent evaluation of the
118/// statement encountered no errors or if the statement is never been evaluated,
119/// then the function results without errors. If the most recent evaluation of
120/// statement failed, then the function results the appropriate error code.
121pub(crate) fn finalize(stmt_ptr: *mut sqlite3_stmt) -> Result<(), Errno> {
122    let rc = unsafe { sqlite3_finalize(stmt_ptr) };
123
124    if rc == SQLITE_OK {
125        Ok(())
126    } else {
127        Err(Errno::Sqlite(rc))
128    }
129}

152    fn init_value(
153        db_ptr: *mut sqlite3,
154        db_value_type: &str,
155        value: Value,
156    ) -> Result<(), Errno> {
157        let sql = format!("CREATE TABLE Dummy(Id INTEGER PRIMARY KEY, Value {db_value_type});");
158
159        execute(db_ptr, sql.as_str())?;
160
161        let sql = "INSERT INTO Dummy(Value) VALUES(?);";
162
163        let stmt_ptr = prepare(db_ptr, sql)?;
164
165        bind(stmt_ptr, 1, value)?;
166        step(stmt_ptr)?;
167        finalize(stmt_ptr)?;
168
169        Ok(())
170    }
171
172    fn get_value(db_ptr: *mut sqlite3) -> Result<Value, Errno> {
173        let sql = "SELECT Value FROM Dummy WHERE Id = 1;";
174
175        let stmt_ptr = prepare(db_ptr, sql)?;
176        step(stmt_ptr)?;
177        let col_result = column(stmt_ptr, 0);
178        finalize(stmt_ptr)?;
179
180        col_result
181    }
