25/// The default page size of `SQLite`.
26const PAGE_SIZE: u32 = 4_096;
27/// Max delay for sql query to retry.
28const MAX_DELAY_MS: u64 = 30000;
29/// Jitter to avoid all clients retrying in lock-step.
30const JITTER_MS: u64 = MAX_DELAY_MS * 2 / 10;
31
32/// Custom `SQLite` busy handler with exponential backoff and random jitter.
33///
34/// This handler is called whenever `SQLite` encounters a `SQLITE_BUSY` state
35/// (e.g. when the database file is locked by another transaction).
36///
37/// Behavior:
38/// - Uses exponential backoff: delays grow as 10, 20, 40, 80… ms.
39/// - Adds a random jitter (0–(`JITTER_MS` - 1) ms) to reduce lock-step retries across
40///   threads.
41/// - Clamps the maximum delay per attempt to `MAX_DELAY_MS` ms.
42/// - Always returns `1` to tell `SQLite` to retry, ensuring requests eventually complete.
43///
44/// Notes:
45/// - The `n` parameter is the number of times the handler has been invoked for the same
46///   lock.
47/// - `_data` is a user-provided pointer passed via `sqlite3_busy_handler`, unused here.
48extern "C" fn busy_handler(
49    _data: *mut c_void,
50    n: c_int,
51) -> c_int {
52    // add (`JITTER_MS` - 1) ms of randomness to avoid all clients retrying in lock-step
53    let jitter = random::<u64>() % JITTER_MS;
54    let exp: u32 = (n.saturating_sub(1)).try_into().unwrap_or(0);
55
56    // grows exponentially: 10, 20, 40, 80… milliseconds
57    let delay = 10u64.saturating_mul(2u64.saturating_pow(exp));
58
59    // add the random shift, but ensure the total wait is ≤ `MAX_DELAY_MS` ms
60    let wait_ms = delay.saturating_add(jitter).min(MAX_DELAY_MS);
61
62    std::thread::sleep(Duration::from_millis(wait_ms));
63    1
64}

66/// Configures the `SQLite` connection to use WAL mode with optimized settings for
67/// concurrency.
68///
69/// This function performs several critical setup steps:
70/// 1. Switches the journal mode to Write-Ahead Logging (WAL), which significantly
71///    improves concurrency by allowing readers to proceed while a writer is active.
72/// 2. Sets the synchronous level to NORMAL. This is a common performance optimization for
73///    WAL mode, reducing disk syncs for non-critical moments, which is generally safe
74///    against crashes but not power loss.
75/// 3. Configures the WAL auto-checkpoint threshold. A checkpoint is the process of
76///    transferring committed transactions from the WAL file back into the main database
77///    file.
78///
79/// # Parameters
80/// * `db_ptr`: A raw mutable pointer to an open `SQLite` database connection.
81///
82/// # Returns
83/// * `Ok(())` on success.
84/// * `Err(Errno)` if any of the configuration steps fail.
85fn enable_wal_mode(db_ptr: *mut sqlite3) -> Result<(), Errno> {
86    let pragmas = "PRAGMA journal_mode=WAL; PRAGMA synchronous=NORMAL;";
87    let c_pragmas = std::ffi::CString::new(pragmas).map_err(|_| Errno::ConvertingCString)?;
88
89    let rc = unsafe {
90        sqlite3_exec(
91            db_ptr,
92            c_pragmas.as_ptr(),
93            None,
94            std::ptr::null_mut(),
95            std::ptr::null_mut(),
96        )
97    };
98    if rc != SQLITE_OK {
99        return Err(Errno::Sqlite(rc));
100    }
101
102    // Set the WAL auto-checkpoint threshold to the default value of 1000 pages.
103    // This means a checkpoint will be triggered automatically when the WAL file
104    // grows to about 4MB (1000 pages * 4KB/page).
105    let rc = unsafe { sqlite3_wal_autocheckpoint(db_ptr, 1000) };
106    if rc != SQLITE_OK {
107        return Err(Errno::Sqlite(rc));
108    }
109
110    Ok(())
111}

113/// Opens a connection to a new or existing `SQLite` database.
114pub(super) fn open(
115    readonly: bool,
116    memory: bool,
117    app_name: ApplicationName,
118) -> Result<*mut sqlite3, Errno> {
119    let mut db_ptr: *mut sqlite3 = std::ptr::null_mut();
120
121    let (db_path, config) = if memory {
122        let in_memory_config =
123            get_app_in_memory_sqlite_db_cfg(app_name).ok_or(Errno::InvalidInMemoryConfig)?;
124
125        (":memory:".into(), in_memory_config)
126    } else {
127        let persistent_config =
128            get_app_persistent_sqlite_db_cfg(app_name).ok_or(Errno::InvalidPersistentConfig)?;
129
130        let db_name = persistent_config
131            .db_file
132            .clone()
133            .ok_or(Errno::MissingDatabaseNameForPersistentConfig)?;
134
135        (db_name, persistent_config)
136    };
137
138    // Setting SQLITE_OPEN_NOMUTEX is enough, without setting env var, since
139    // by default SQLITE_THREADSAFE=1 is used, and doc says:
140    //
141    // If single-thread mode has not been selected at compile-time or start-time,
142    // then individual database connections can be created as either multi-thread or
143    // serialized.
144    let flags = if readonly {
145        SQLITE_OPEN_READONLY
146    } else {
147        SQLITE_OPEN_CREATE | SQLITE_OPEN_READWRITE
148    } | SQLITE_OPEN_NOMUTEX;
149
150    let c_path =
151        CString::new(db_path.to_string_lossy().as_bytes()).map_err(|_| Errno::ConvertingCString)?;
152
153    let rc = unsafe { sqlite3_open_v2(c_path.as_ptr(), &raw mut db_ptr, flags, std::ptr::null()) };
154
155    if rc != SQLITE_OK {
156        return Err(Errno::Sqlite(rc));
157    } else if db_ptr.is_null() {
158        return Err(Errno::FailedOpeningDatabase);
159    }
160
161    let rc = unsafe { sqlite3_busy_handler(db_ptr, Some(busy_handler), std::ptr::null_mut()) };
162    if rc != SQLITE_OK {
163        return Err(Errno::Sqlite(rc));
164    }
165
166    if !memory && !readonly {
167        enable_wal_mode(db_ptr)?;
168    }
169
170    // config database size limitation
171    let rc = if readonly {
172        SQLITE_OK
173    } else if memory {
174        let size_limit = i64::from(config.max_db_size);
175
176        unsafe { sqlite3_soft_heap_limit64(size_limit) };
177
178        SQLITE_OK
179    } else {
180        let page_size = config.max_db_size / PAGE_SIZE;
181        let pragma_stmt = format!("PRAGMA max_page_count = {page_size}");
182
183        let c_pragma_stmt =
184            std::ffi::CString::new(pragma_stmt).map_err(|_| Errno::ConvertingCString)?;
185
186        unsafe {
187            sqlite3_exec(
188                db_ptr,
189                c_pragma_stmt.as_ptr(),
190                None,
191                std::ptr::null_mut(),
192                std::ptr::null_mut(),
193            )
194        }
195    };
196
197    if rc != SQLITE_OK {
198        return Err(Errno::FailedSettingDatabaseSize);
199    }
200
201    Ok(db_ptr)
202}

204/// Same as [`open`], but even for in-memory connection, open on disk in a separate file.
205pub(super) fn open_with_persistent_memory(
206    readonly: bool,
207    memory: bool,
208    app_name: ApplicationName,
209) -> Result<*mut sqlite3, Errno> {
210    // Internally `kernel::open` derives db path from `ApplicationName`, treating it as no
211    // more than a string. So, it is okay to substitute it at `kernel::open` level to
212    // create another file. Once the pointer is obtained, however, it must be bound to the
213    // resource under the original `app_name`.
214    let db_name = if memory {
215        ApplicationName(format!("memory.{app_name}"))
216    } else {
217        app_name
218    };
219
220    let memory = false;
221
222    open(readonly, memory, db_name)
223}

225/// Files associated with a database.
226#[derive(Debug)]
227pub(super) struct DbPaths {
228    /// See <https://www2.sqlite.org/c3ref/filename_database.html>.
229    pub database: PathBuf,
230    /// See <https://www2.sqlite.org/c3ref/filename_database.html>.
231    pub journal: PathBuf,
232    /// See <https://www2.sqlite.org/c3ref/filename_database.html>.
233    pub wal: PathBuf,
234}
235
236impl DbPaths {
237    /// Returns **main** db paths associated with the connection.
238    /// The paths are owned by Rust and do not depend on sqlite connection after being
239    /// obtained.
240    ///
241    /// See <https://www2.sqlite.org/c3ref/db_name.html> and <https://www2.sqlite.org/c3ref/db_filename.html>.
242    pub(crate) fn main(db_ptr: *mut sqlite3) -> Result<DbPaths, Errno> {
243        unsafe fn c_str_as_nonempty_path<'a>(
244            ptr: *const std::ffi::c_char
245        ) -> Result<&'a Path, Errno> {
246            if ptr.is_null() {
247                return Err(Errno::ReturnedNullPointer);
248            }
249            let c_str = unsafe { std::ffi::CStr::from_ptr(ptr) };
250            // Most major platforms support Unicode (see crate-level comment at <https://docs.rs/camino/latest/camino>).
251            // If this platform doesn't, an error is safely returned.
252            if let Ok(utf8) = c_str.to_str() {
253                if utf8.is_empty() {
254                    Err(Errno::ConvertingCString)
255                } else {
256                    Ok(Path::new(utf8))
257                }
258            } else {
259                Err(Errno::ConvertingCString)
260            }
261        }
262        unsafe {
263            let db_name_ptr = sqlite3_db_name(db_ptr, 0);
264            c_str_as_nonempty_path(db_name_ptr)?;
265            let db_filename_ptr = sqlite3_db_filename(db_ptr, db_name_ptr);
266            c_str_as_nonempty_path(db_filename_ptr)?;
267
268            Ok(DbPaths {
269                database: c_str_as_nonempty_path(sqlite3_filename_database(db_filename_ptr))?
270                    .to_path_buf(),
271                journal: c_str_as_nonempty_path(sqlite3_filename_journal(db_filename_ptr))?
272                    .to_path_buf(),
273                wal: c_str_as_nonempty_path(sqlite3_filename_wal(db_filename_ptr))?.to_path_buf(),
274            })
275        }
276    }
277
278    /// Removes all existing persistent files.
279    /// Returns the first error encountered, even if some files where successfully
280    /// removed.
281    pub(crate) fn remove_all(&self) -> std::io::Result<()> {
282        [&self.database, &self.journal, &self.wal]
283            .into_iter()
284            .map(|path| {
285                if std::fs::exists(path)? {
286                    std::fs::remove_file(path)
287                } else {
288                    Ok(())
289                }
290            })
291            .fold(Ok(()), std::io::Result::and)
292    }
293}
